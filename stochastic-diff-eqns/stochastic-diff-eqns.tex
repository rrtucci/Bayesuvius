\chapter{Stochastic Differential Equations}
\label{ch-stochastic-diff-eqns}

This chapter is based mostly on Ref.\cite{sar-sol}.

{\bf Stochastic Differential Equations (SDE)}\footnote{The 
acronym SDE complements the acronyms ODE (Ordinary Differential Equation)
and PDE (Partial Differential Equation)}
are deterministic first order differential equations with additive external white noise. When discretized, they
can be modelled as dynamic bnets of type
DEN (Deterministic bnet with External Noise) (see Chapter \ref{ch-LDEN}).

This chapter deals with {\bf Classical Stochastic Calculus}. In that calculus,
the SDE have real valued solutions.
A theory of {\bf Quantum Stochastic Calculus} has also been developed (see,
for example,  Ref.\cite{part-quantum-sde}) that is
very similar to the classical
theory. In the quantum theory, the SDEs  have complex valued solutions. The quantum theory describes quantum
mechanical systems (such as lasers)
whereas the classical theory describes classical macroscopic systems such as a
pollen particle undergoing Brownian motion
while submerged in a liquid.

{\bf White noise} ${\dot{\rvB}}(t)
\in\RR^n$ for $t\geq 0$
is a random process with the
following properties:

\begin{enumerate}
\item Normally distributed
\beq
{\dot{\rvB}}(t) \sim \caln(\mu=0, Cov=Q)
\eeq

\item Independence at different times

${\dot{\rvB}}(t)$ and ${\dot{\rvB}}(s)$ are independent for
$t\neq s$

\item Zero Mean

\beq E[{\dot{\rvB}}(t)]=0
\eeq

\item Zero covariance at two different times
\beq
C_{\dot{\rvB}}(t, s) = \av{{\dot{\rvB}}(t), {\dot{\rvB}}^T(s)} = Q\delta(t-s)
\eeq



\end{enumerate}
{\bf Brownian motion} (a.k.a. {\bf Wiener process})
$\rvB(t)\in\RR^n$
for $t\geq 0$
is a random process with the
following properties:

\begin{enumerate}

\item Zero at time zero

\beq
\rvB(0)=0\footnote{If you wish to
consider  $\rvB(0)=\beta_0\neq 0$, replace
$\rvB$ by $\rvB-\beta_0$}
\eeq

\item Time derivative is 
white noise
\beq
\frac{
\Delta_{t_k}^{t_{k+1}}\rvB
}{\Delta_{t_k}^{t_{k+1}}t}
\sim \caln(\mu=0, Cov=Q)
\eeq

\beq
{\dot{\rvB}}(t_k)\sim  \caln(\mu=0, Cov=Q)
\eeq

\beq
\frac
{d \rvB}{dt} = {\dot{\rvB}}
\eeq

\beq
\rvB_t = \int_{t'=0}^t
dt'\; \dot{\rvB}_{t'}
\label{eq-int-white-noise}
\eeq
 Hence
$\rvB_t$ is an accumulation of white noise over the time
interval $[0, t]$.
Brownian motion acts as if it had a tail.

\item
Zero mean and covariance value from time 
overlap


\begin{claim}
\beq
E[\rvB_s] =0, \quad E[\rvB_s \rvB_t]=
\min(s,t)
\eeq

\beq
\av{\rvB_s, \rvB_t}=
\min(s,t)
\eeq
\end{claim}
\proof

$E[\rvB_t]=0$ is
obvious from
Eq.(\ref{eq-int-white-noise}) and the properties of white noise.

Assume $s<t$ (let $s$ stands for smaller).

\beq
\rvB_t-\rvB_s = \int_{s}^t
dt'\; \dot{\rvB}_{t'}
\eeq


\beqa
\av{\rvB_t, \rvB_s}
&=&
\underbrace{\av{(\rvB_t-\rvB_s),\rvB_s}}_0
+
\underbrace{\av{\rvB_s,\rvB_s}}_s
\eeqa
\qed

If $\rvB(t)\in\RR$ and
$[r, s]\cap [r', s']=\emptyset$, then

\beq
E[(\Delta^{s}_{r}\rvB) (\Delta^{s'}_{r'}\rvB)]=0
\label{eq-non-overlap-B}
\eeq and

\beq
E[|\Delta^{t}_{s}\rvB|^2] = |t-s|
\label{eq-overlap-B}
\eeq

For example,
\beqa
E[\Delta_1^4\rvB \Delta_3^6\rvB]
&=&
E[(\Delta_1^3\rvB + \Delta_3^4\rvB)
(\Delta_3^4\rvB + \Delta_4^6\rvB)
]
\\
&=&
E[(\Delta_3^4\rvB)^2]
\\
&=& |4-3|
\eeqa

Eqs.(\ref{eq-non-overlap-B}) and (\ref{eq-overlap-B}) can be
combined as follows

\beq
E[(\Delta^{s}_{r}\rvB) (\Delta^{s'}_{r'}\rvB)]=\; {\rm len}([r, s]\cap [r', s'])
\eeq
If  $(B_\mu)\in \RR^n$,
then covariance
matrix is proportional to the identity matrix

\beq
E[(\Delta^{s}_{r}\rvB) (\Delta^{s'}_{r'}\rvB)^T]
=I_n \; {\rm len}([r, s]\cap [r', s'])
\eeq

\end{enumerate}

Brownian Motion $\rvB_t$
is a special case of a {\bf Levy process}. A Levy process
$\rvx_t$ satisfies

\beq
P(\rvx_s-\rvx_t=x)= P(\rvx_{s-t}=x)
\eeq
3 examples of a Levy process are

\begin{enumerate}
\item {\bf Brownian Motion}

\beq
P(\rvx_s-\rvx_t=x)=\caln(x;
\mu=0, \s^2 )
\eeq
\item {\bf Poisson process}
\beq
P(\rvx_s-\rvx_t=x)= {\rm Poisson}(x;\lam=R(s-t))=
\frac{\lam^x e^{-\lam}}{x!}
\quad
\text{(Poisson Distri.)}
\eeq

For $\lam\ll 1$ and $x=1$,

\beq
P(\rvx_{s-t}=1)\approx R(s-t)
\eeq

\beq
P(\rvx_{s-t}=0)\approx 1-R(s-t)
\eeq

\item {\bf Cauchy process}

\beq
P(\rvx_s-\rvx_t=x)= 
{\rm Cauchy}(x;\mu=0, \gamma=s-t)=
\frac{1}{\pi\gamma}\frac{\gamma^2}{(x^2+\gamma^2)}
\quad
\text{(Cauchy Distri.)}
\eeq

\end{enumerate}

\section{Transition Probability Matrices}

Consider a sequence of random variables $\rvx_1, \rvx_2, \rvx_3, \dots$

The probability 

\beq
P(\rvx_b=x_b|\rvx_a=x_a)=P(x_b|x_a)
\eeq
for $a<b$ is called a {\bf transition probability matrix (TPM)} for a discrete 
process.
 The bnet
$$
\rvx_0\rarrow\rvx_1\rarrow \rvx_2\rarrow\cdots\rarrow\rvx_{N-1}$$
is called a {\bf Markov chain}. Its TPM satisfies

\beq
P(x_{i+1}| x_i, x_{i-1}, \ldots, x_0)=
P(x_{i+1}| x_i)
\eeq

For continuous instead of discrete time,
the definitions of TPM and Markov chain are generalized
as follows.
It's as if $\rvx(t_k)=\rvx_k$ for $k\in[N]$ and $t_{k-1}-t_k=\Delta t\to 0$.

Suppose  $t_0<t$ and $x_0, x\in \RR^n$.
We define
the {\bf event transition probability matrix (TPM)} as

\beq
P(\rvx_t=x|\rvx_{t_0}=x_0)=
=P(x(t)|x(t_0))=P(\XT{x}{t}|\XT{x_0}{t_0})
\eeq
For a {\bf Markovian process}

\beq
P(x(t)| x([0, s]))=
P(x(t)| x(s))
\eeq
\begin{claim}
A Markovian process $\rvx_t$ satisfies the
{\bf Chapman-Kolgomorov equation}:

\beq
\sum_{x_b}
P(\XT{x_c}{t_c}|\XT{x_b}{t_b})
P(\XT{x_b}{t_b}|\XT{x_a}{t_a})
=
P(\XT{x_c}{t_c}|\XT{x_a}{t_a})
\eeq
for $t_a\leq t_b\leq t_c$.
\end{claim}
\proof
\beqa
P(\XT{x_c}{t_c}|\XT{x_b}{t_b})
P(\XT{x_b}{t_b}|\XT{x_a}{t_a})
&=&
P(\XT{x_c}{t_c}|\XT{x_b}{t_b},\XT{x_a}{t_a})
P(\XT{x_b}{t_b}|\XT{x_a}{t_a})
\\
&=&
P(\XT{x_c}{t_c},\XT{x_b}{t_b}|\XT{x_c}{t_c})
\eeqa
Now apply $\sum_{x_b}$ to both sides.

Note that if $\rvx_t$ is not Markovian, $\sum_{x_b}
P(\XT{x_c}{t_c}|\XT{x_b}{t_b})
P(\XT{x_b}{t_b}|\XT{x_a}{t_a})$
is still a probability distribution in $x_c$, but a different one from $P(\XT{x_c}{t_c}|\XT{x_a}{t_a})$.
\beq
\sum_{x_b}
P(\XT{x_c}{t_c}|\XT{x_b}{t_b})
P(\XT{x_b}{t_b}|\XT{x_a}{t_a})=
Q(\XT{x_c}{t_c}|\XT{x_a}{t_a})
\eeq
\qed


In this chapter, we will
re-express many equations in Dirac notation
for greater clarity.
In Dirac notation, the TPM
becomes

\beq
P^\rvx(\XT{x}{t}|\XT{x_0}{t_0})=
\bra{\XT{x}{t}}P^\rvx_{t,t_0}\ket{\XT{x_0}{t_0}}
\eeq
and the Chapman-Kolgomorov equation 
for a Markovian process $\rvx_t$
becomes

\beq
\bra{x_a} P^\rvx_{t_a|t_b}
\underbrace{\left[\sum_{x_b} \ket{x_b}
\bra{x_b}\right] }_{1} P^\rvx_{t_b|t_c}
\ket{x_c}
=
\bra{x_a} P^\rvx_{t_a|t_c} \ket{x_c}
\eeq
Hence

\beq
\boxed{
P^\rvx_{t_a|t_b}
P^\rvx_{t_b|t_c} = 
P^\rvx_{t_a|t_c}}
\eeq

\section{Martingales}

Consider a discrete sequence
of random variables
$\rvx=[\rvx_1, \rvx_2, \rvx_3, \ldots]$.
$\rvx$ is a {\bf martingale}
if\footnote{In more rigorous
treatments, the finiteness constraint
$ \forall i\;
E[\;|\rvx_i|\;] < \infty 
$
is added to the definition
of martingale. We
won't bother with that constraint
because we don't aim to be 
fully rigorous.in this book.
}

\beq
E[\rvx_b|\rvx_a, \rvx_{<a}]=\rvx_a
\eeq
and
\beq
E[\rvx_b|\rvx_a=x, \rvx_{<a}]=x
\eeq
for any $a, b$ such that $a\leq b$.

A (continuous) random process $\rvx_t$ is a {\bf martingale} if
\beq
E[\rvx_t|\rvx_{t_0}, \rvx_{<t_0}]= \rvx_{t_0} 
\eeq
when $t_0\leq t$.
In particular, applying $\ket{x_0}$ to both sides,
\beq
E[\rvx_t|\rvx_{t_0}=x_0, \rvx_{<t_0}]=x_0
\eeq

Below, we will assume $t_0\leq t$.

\begin{claim}
If $\rvx_t$ is a martingale, then

\beq
\boxed {P^\rvx_{t|t_0}= P^\rvx_{t_0|t_0}=1}
\eeq

\end{claim}
\proof
\beqa
E[\rvx_t|\rvx_{t_0}=x_0]
&=&
\sum_{x}x\bra{x}P_{t|t_0}\ket{x_0}
\\
&=&
\rvx\sum_{x}\ket{x}\bra{x} P_{t|t_0}\ket{x_0}
\\
&=& \rvx \underbrace{P_{t|t_0}}_{1}\ket{x_0}
\\
&=&
x_0
\eeqa
\qed

\begin{claim}
Brownian motion is a martingale.
\beq
E[\rvB_t|\rvB_{t_0}=B_0]= B_0
\eeq
\end{claim}
\proof

\beqa
E[\rvB_t|\rvB_{t_0}=B_0]
&=&
\sum_B B \av{B|P_{t|t_0}|B_0}
\\
&=&
\int_{-\infty}^\infty dB\; B\;
\caln(B; \mu=B_0, \s^2=0)
\\
&=&
B_0
\eeqa
The mean of a normal distribution is
its mean.

\qed

Let
\beq
\phi(x)=\av{\phi|x}
\eeq

Let $\ket{x}_t$  be a copy of $\ket{x}$ for time $t$.
Likewise, let $\bra{x}_t$  be a copy of $\bra{x}$ for time $t$.
We shall represent a function $\phi(x,t)$
with explicit time dependence by

\beq
\phi(x,t)=\av{\phi|x}_t
\eeq

Note that $x_t=x(t)$ so $\ket{x_t}\neq \ket{x}_t= \ket{\rvx_t=x}$ (strictly speaking, although some people
might equate both).
I won't use $\ket{x_t}$.

\begin{claim}
 If $\phi(\cdot)$ is a smooth function 
and $\rvx_t$
is a martingale, then $\phi(\rvx_t,{t})$ is a martingale.
\beq
E[\phi(\rvx_t,{t})|\rvx_{t_0}=x_0]= \phi(x_0, t_0)
\eeq
\end{claim}
\proof

\beqa
E[\phi(\rvx_t,{t})|\rvx_{t_0}=x_0]
&=&
\bra{\phi}
\sum_{x}\ket{x}_t
\bra{x}_tP_{t|t_0}
\ket{x_0}_{t_0}
\\
&=&
\bra{\phi}P_{t|t_0}\ket{x_0}_{t_0}
\\
&=&
\av{\phi|x_0}_{t_0}
\eeqa
\qed




It\^{o} integrals $\int L(\XT{x}{t})d\rvB_t$ (discussed later)
are martingales too
(because they are semi-martingales without a drift term), but Stratonovich integrals
$\int L(\XT{x}{t})\circ d\rvB_t$
(discussed later)
aren't martingales (because they are semi-martingales with a drift term).





\section{Stochastic Integrals}



Define a {\bf semi-martingale}
as a random process $\rvx_t$ that satisfies the 
{\bf Stochastic Differential Equation} (SDE)
\beq
\frac{d\rvx_t}{dt}= f_t + L_t\circ_\alp \dot{\rvB}_t
\eeq
where $f_t$ and $L_t$ are reasonably  smooth
function of $t$ and $\rvx_t$. 
The definition of the product $\circ_\alp$ of $L_t$ and
the white noise $\dot{\rvB}_t$, requires 
some care and will be given later.
For now, just note that
a semi-martingale's time derivative is the 
sum of a smooth function ({\bf drift term}) and a zig-zaggy function ({\bf diffusion term, a martingale if $\alp=0$}).

An alternative differential expression for this SDE is

\beq
\begin{array}{lll}
\xymatrix{
\ul{dt}\ar[dr]|f
&&\ul{dB}\ar[dl]|{L\circ_\alp}
\\
&\ul{dx}
}&&
d\rvx_t= f_t dt + L_t\circ_\alp d\rvB_t
\end{array}
\eeq
For example, $d\rvx_t=  \mu dt
+ d\rvB_t$, Brownian motion with a drift,
is a semi-martingale.

This SDE is solved, at least formally once 
we define a stochastic integral, by

\beq
\rvx(T)-\rvx(0)=\int_{0}^T f_t dt  + 
\int_{t=0}^T L_t\circ_\alp d\rvB_t
\eeq


$f_t=f(\rvx_t, t)$ is called
the {\bf drift term} and 
$L_t=L(\rvx_t, t)$ is called the
{\bf diffusion coefficient}. When 
$L_t$ depends (resp., does not depend)
explicitly on $\rvx_t$,
we say the system has
{\bf multiplicative noise} 
(resp., {\bf additive}).
A system for which $f$ and $L$ are
both constant (i.e., independent of the event $(x,t)$) is said to have {\bf Constant Coefficients (CC)}\footnote{Ref.\cite{sar-sol},
on which most of this chapter is
based, calls systems with CC, LTI (linear, time-invariant) systems.}






For any $\alp\in[0, 1]$, define the {\bf General Stochastic $\alp$-anticipatory Integral} of the  semi-martingales $\rvx_t$and $\rvy_t$ by

\beq 
\int_{t=0}^T 
\rvx_t\circ_\alp d\rvy_t
\eqdef 
\lim_{\Delta t\to 0}\sum_i \rvx_{t_i+\alp\Delta t}\Delta_{t_i}^{t_{i+1}}\rvy
\eeq

\begin{itemize}
\item $\alp=0$: {\bf It\^{o} (zero anticipatory) Integral}

In this case we will omit the $\circ_\alp$
\beq
\rvx\circ_0 d\rvy =\rvx\; d\rvy,
\eeq
\item $\alp=1/2$: {\bf Stratonovitch (half anticipatory) integral}

In this case we will omit the $\alp$
\beq
\rvx\circ_{1/2} d\rvy =\rvx\circ d\rvy
\eeq
\item $\alp=1$: {\bf fully anticipatory integral}
\end{itemize}


Define the {\bf quadratic co-variation of $\rvx_t$ and $\rvy_t$} (or variation if
$\rvx_t=\rvy_t$)
by
\beq
\av{\rvx_t, \rvy_t}_0^T=
\lim_{\Delta t\to 0}\sum_i\left[\Delta_{t_i}^{t_i+\Delta t}\rvx\right]
\left[\Delta_{t_i}^{t_i+\Delta t}\rvy\right]
\eeq
For example, in a semi-martingale,
the smooth part  $f_t$ has
zero quadratic variation but the
zig-zaggy part $L_t\rvB_t$ has a non-zero one.

\beq
\av{f_t, f_t}_0^T=0,
\quad 
\av{L_t\rvB_t, L_t\rvB_t}_0^T\neq 0
\eeq


\begin{claim}
(Conversion from $\alp$-anticipatory to It\^{o})

If $\rvx_t$ and $\rvy_t$
are semi-martingales,
then

\beq
\int_{t=0}^T
\rvx_t \circ_\alp d\rvy_t
=
\int_{t=0}^T \rvx_t d \rvy_t
+\alp  \av{\rvx_t, \rvy_t}_0^T
\eeq
\label{cl-ito-conversion-eq}
\end{claim}
\proof

\beq
\rvx_{t_i +\alp \Delta t}
=
\rvx_{t_i} +\alp \Delta_{t_i}^{t_{i+1}}
\rvx
+
\calo
\left([\Delta_{t_i}^{t_{+1} } \rvx]^2\right)
\eeq

\beqa
\sum_i \rvx_{t_i+\alp\Delta t}\Delta_{t_i}^{t_{i+1}}\rvy
&=&
\sum_i  \rvx_{t_i}
\Delta_{t_i}^{t_{i+1}}\rvy
+
\alp\sum_i
\left(\Delta_{t_i}^{t_{i+1}}\rvx\right)
\left(\Delta_{t_i}^{t_{i+1}}\rvy\right)
\\
&\to&
\int_{t=0}^T \rvx_t d \rvy_t
+\alp  \av{\rvx_t, \rvy_t}_0^T
\eeqa
\qed

\begin{claim}
\label{cl-covariation-LM}
If
\beq
d\rvx_t = f_t dt + L_t d\rvB_t
\eeq
and
\beq
d\rvy_t = g_t dt + M_t d\rvB_t
\eeq
then
\beq
\av{\rvx_t, \rvy_t}_0^T=
\int_{t=0}^T L_t M_t\; dt
\eeq
\end{claim}
\proof
\beqa
\av{\rvx_t, \rvy_t}_0^T
&=&
\sum_i 
\left[\Delta_{t_i}^{t_i+\Delta t}\rvx_{t_i}\right]
\left[\Delta_{t_i}^{t_i+\Delta t}\rvy_{t_i}\right]

\left[L_{t_i}d\rvB_{t_i}\right]
\left[M_{t_i}d\rvB_{t_i}\right]
\\
&\to&
\int_{t=0}^T L_t M_t\; dt
\eeqa

\qed

\begin{claim}
The It\^{o} integral
is a martigale. The $\alp$-anticipatory
integral for $\alp\neq 0$ is not a martingale.
\end{claim}
\proof

\beq
M_\alp(T)= \int_{t=0}^T
\rvB_t \circ_\alp d\rvB_t
\eeq

\beq
E[
M_0(T)
\mid
\rvB_{[0, T]}
]
=
\frac{1}{2}E[\rvB_T^2 - T\mid \rvB_{[0, T]}]
=
\frac{1}{2}[\rvB_T^2 -T]
=M_0(T)
\eeq

For $\alp\neq 0$,
\beq
E[
M_\alp(T)
\mid
\rvB_{[0, T]}
]\neq M_\alp(T)
\eeq

\qed

The It\^o integral is  a martingale
but it violates the integration
rules of standard Calculus. The
Stratonovich integral does the opposite. It's not
a Martingale, but it 
preserves  the integration
rules of standard Calculus.

Martingale random processes respect Causality; i.e., 
their current expectation value does not
depend on the future. The
$\alp$ in a stochastic integral (SI)
measures how much the SI
violates Causality. SIs with
$\alp=0$ (It\^{o}) are the only truly causal ones.
Microscopic Physics should
of course obey Causality so
observable
quantities
should be 
expressible using Causal SI's only. 
However, there are
many situations
in which non-causal SI with fractional $\alp$
are useful. This
is true, for example,
if we want to impose future time
boundary conditions
(such as terminal $\rvx_T=k$
instead of the usual initial $\rvx_0=k$  conditions), or if we do some
temporal averaging that
includes future values.

\section{It\^{o}  Lemma}


Consider the $n$ dimensional case $\rvx, \dot{\rvB}\in \RR^n$. Let $\mu, \nu,\alpha, \beta\in [n]$. Suppose
\beq
d\rvx_\mu= f_\mu(\rvx, t)dt + L_{\mu,\nu}(\rvx, t)d\rvB_\nu(t)
\eeq

\beq
d\rvB_\alp d\rvB_\beta = Q_{\alp, \beta}\quad
\text{(In 1-dim, $Q_{\alp, \beta}\to q$)}
\eeq
and

\beq
\boxed
{R_{\mu, \nu} = \frac{1}{2} L_{\mu,\alpha}Q_{\alpha,\beta}L^T_{\beta,\nu}\quad \text{(In 1-dim, $R_{\mu,\nu}\to \frac{q}{2}L^2$)}}
\eeq



\begin{claim}(It\^{o} Lemma)

If $\ul{\phi}=\phi(\rvx, t)$ then
\beq
\begin{array}{c}
d\ul{\phi}=
\left[\pder{\phi}{t} +
 f_\mu\pder{\phi}{x_\mu} +
 R_{\mu, \nu}
\frac{\partial^2\phi}{\partial x_\mu\partial x_\nu}\right]dt
+
\pder{\phi}{x_\mu}L_{\mu,\nu}d\rvB_\nu
\\
\bcen
\xymatrix{
\ul{dt}\ar[dr]|{f_\mu}
&&\ul{dB}_\nu\ar[dl]|{L_{\mu\nu}}
\\
&\ul{dx}_\mu\ar[d]|{
\pder{\phi}{x_\mu}dx_\mu +\frac{1}{2}
dx_\mu dx_\nu\frac{\partial^2\phi}{\partial x_\mu\partial x_\nu}
}
\\
&\ul{d\phi}
}
\ecen
=
\bcen
\xymatrix@C=7pc{
\ul{dt}\ar[dr]|{\pder{\phi}{t} +
f_\mu \pder{\phi}{x_\mu} + R_{\mu, \nu}
\frac{\partial^2\phi}{\partial x_\mu\partial x_\nu}}
&&\ul{dB}_\nu\ar[dl]|{\pder{\phi}{x_\mu}L_{\mu,\nu}}
\\
&\ul{d\phi}
}
\ecen
\end{array}
\eeq
or, in the 1-dim case,

\beq
\begin{array}{c}
d\ul{\phi}=
\left[\partial_t \phi +
f \partial_x \phi + \frac{q}{2}L^2
\partial^2_x \phi\right]dt
+
L\partial_x \phi d\rvB
=\quad
dt \partial_t\phi
+dx \partial_x\phi
+ dt\frac{q}{2}L^2
\partial^2_x \phi
\\
\bcen
\xymatrix{
\ul{dt}\ar[dr]|{f}
&&\ul{dB}\ar[dl]|{L}
\\
&\ul{dx}\ar[d]|{
dx \partial_x \phi+\frac{1}{2}
(dx)^2\partial^2_x\phi
}
\\
&\ul{d\phi}
}
\ecen
=
\bcen
\xymatrix@C=7pc{
\ul{dt}\ar[dr]|{\partial_t \phi +
f \partial_x \phi + \frac{q}{2}L^2
\partial^2_x \phi}
&&\ul{dB}\ar[dl]|{L\partial_x\phi}
\\
&\ul{d\phi}
}
\ecen
\end{array}
\eeq






\end{claim}
\proof

The Taylor
expansion of $\phi(x,t)$, up to second
order derivatives, is

\beq
d\phi =
\pder{\phi}{t}dt
+
\sum_\mu \pder{\phi}{x_\mu} d\rvx_\mu
+\frac{1}{2}
\sum_\mu \sum_\nu \frac{\partial^2\phi}{\partial x_\mu\partial x_\nu}d\rvx_\mu d\rvx_\nu
\eeq


\beq
\pder{\phi}{x_\mu} d\rvx_\mu
=
\pder{\phi}{x_\mu}
\left[
f_\mu dt + L_{\mu, \nu}d\rvB_\nu
\right]
\eeq

\beqa
\frac{\partial^2\phi}{\partial x_\mu\partial x_\nu}d\rvx_\mu d\rvx_\nu
&=&\frac{\partial^2\phi}{\partial x_\mu\partial x_\nu}
\left[
f_\mu dt + L_{\mu, \alpha}d\rvB_\alpha
\right]
\left[
f_\nu dt + L_{\nu, \beta}d\rvB_\beta
\right]
\\
&=&
\frac{\partial^2\phi}{\partial x_\mu\partial x_\nu}
L_{\mu, \alpha}  L_{\nu, \beta} d\rvB_\alpha d\rvB_\beta
\\
&=&
\frac{\partial^2\phi}{\partial x_\mu\partial x_\nu}
L_{\mu, \alpha}  L_{\nu, \alp} Q_{\alpha,\alp} dt
\eeqa

\beqa
d\phi &=&
\pder{\phi}{t}dt
+
\pder{\phi}{x_\mu} [f_\mu dt + L_{\mu,\nu}d\rvB_\nu]
+ R_{\mu, \nu}
\frac{\partial^2\phi}{\partial x_\mu\partial x_\nu}dt
\\
&=&
\left[\pder{\phi}{t} +
f_\mu \pder{\phi}{x_\mu} + R_{\mu, \nu}
\frac{\partial^2\phi}{\partial x_\mu\partial x_\nu}\right]dt
+
\pder{\phi}{x_\mu}L_{\mu,\nu}d\rvB_\nu
\eeqa

\qed

\section{Fokker-Planck Equation}
The next claim defines the {\bf Fokker-Planck equation} (FP equation) (a.k.a. {\bf Fokker-Planck-Kolgomorov equation})
for the probability $P(x,t)$
of single event $(x,t)$.

Below, we will use $x$  for the vector with components $x_\mu$
and $x_t=x(t)$ for the vector with components $x_{t\mu}$.


\begin{claim}
(Forward FP equation)

If\footnote{Note that $\rvx_\mu$ does not
depend on $t$ explicitly but $P(x,t)$ does}
\beq
d\rvx_\mu(\rvx_t) = f_\mu(\rvx_t)dt + L_{\mu\nu}(\rvx_t)d\rvB_\nu
\eeq
then

\beq
	\pder{P(x, t)}{t}= \calf_x P(x, t)
	\eeq
with

\beq
\calf_x \bullet=
-\;\pder{}{x_\mu}
(f_\mu\bullet ) +
\frac{\partial^2}{ \partial x_\mu\partial x_\nu}( R_{\mu, \nu}\bullet)
\eeq
\end{claim}
\proof



Integration by parts

\beq
udv = d(uv)-(du)v
\eeq

\beq
\int_{-\infty}^{+\infty}udv=
\underbrace{uv|_{-\infty}^{+\infty}}_{0}
\quad
-
\int_{-\infty}^{+\infty}(du)v
\eeq

From It\^o's Lemma for $\phi(\cdot)$
that doesn't depend on $t$ explicitly,
\beq
\int dx^n\; P(x)\frac{d\phi(x_t)}{dt}=
\int dx^n\; P(x)
\left[
 f_\mu\pder{\phi}{x_\mu} + R_{\mu, \nu}
\frac{\partial^2\phi}{\partial x_\mu\partial x_\nu}\right]
\eeq
Now pull the time derivative
out of the integral on left hand side
and integrate by parts the right hand side.



\beq
\frac{d}{dt}\int dx^n\; \phi(x_t) P(x)=
\int dx^n\;  \phi(x_t)
\left[ -
\pder{(Pf_\mu)}{x_\mu} +
\frac{\partial^2 }{\partial x_\mu\partial x_\nu}(PR_{\mu, \nu})\right]
\eeq
\qed

\begin{claim}
	\beq
\pder{P(x,t)}{t}= \calf_x P(x,t)
	\eeq
	is solved formally by
	
	\beq
	P(x,t)=e^{(t-t_0)\calf_x}P(x, t_0)
	\eeq
\end{claim}
\proof
Just use the Taylor expansion of an exponential function and treat the operator $\calf_x$ as a constant.
\qed

Note that
\beq
\pder{P}{t} = -\pder{J_\mu}{x_\mu}
\label{eq-cons-prob}
\eeq
where

\beq
J_\mu = Pf_\mu - \pder{(PR_{\mu,\nu})}{x_\nu}
\eeq

Eq.(\ref{eq-cons-prob})
is the {\bf equation for conservation of probability}\footnote{In integral form
$$\pder{}{t}\int_V dV\;P=
-\int_V dV\;\nabla\cdot \vec{J}=
-\int_{\partial V} \vec{J}\cdot d\vec{S}$$
}. $J_\mu$ is called the {\bf probability
flux}, $f_\mu$ is called the {\bf drift},
and $R_{\mu, \nu}$ is called the
{\bf diffusion matrix} (or {\bf diffusion coefficient} if it's a scalar)

The forward FP equation when  $n=1$, $f=0$, $L=1$, $R=D>0$, is called the {\bf Diffusion Equation}.

\beq
dx = d\rvB
\eeq

\beq
\pder{P}{t} = D
\frac{\partial^2P}{\partial^2x}
\eeq

As another example of the forward FP,
consider the
{\bf Overdamped Langevin Equation}\footnote{
This equation, also known as Brownian dynamics (see
Ref.\cite{wiki-brownian-dyn}),  arises from
Newton's equation $m\ddot{x}= -\lam\dot{x}  -U'(x)$ when the acceleration $\ddot{x}$ is negligible, so the drag force and potential force cancel each other.}

\beq
d\rvx_\mu(\rvx_t) = -\;\frac{1}{2}\pder{U(\rvx_t)}{x_\mu}dt + d\rvB_\mu
\eeq

\begin{claim}
For the overdamped Langevin equation,
if $R_{\mu\nu}=\delta_{\mu\nu} \frac{1}{2\lam}>0$, the steady state
solution is

\beq
P(x) = \frac{e^{-\lam U(x)}}{Z}
\eeq
where
\beq
Z = \int dx^n\;e^{-\lam U(x)}
 \eeq

\end{claim}
\proof

The forward FP equation with
$\pder{P}{t}=0$ and
$R_{\mu\nu}=\delta_{\mu\nu} \frac{1}{2\lam}$ is
\beq
0= -
\pder{(Pf_\mu)}{x_\mu} + \frac{1}{2\lam}
\frac{\partial^2 P }{\partial x_\mu^2}
\eeq
If we substitute $f_\mu =-\; \frac{1}{2}\pder{U}{x_\mu}$ into this, we get


\beq
0= \frac{1}{2}
\pder{}{x_\mu}\left(P\pder{U}{x_\mu} + \frac{1}{\lam}
\pder{P }{x_\mu}
\right)
\eeq
If we now substitute the proposed value of $P$, we get $0=0$

\beq
0=
\pder{}{x_\mu}\left(\lam e^{-\lam U}\pder{U}{x_\mu}+
\pder{e^{-\lam U}}{x_\mu}
\right)
\eeq

\qed

Recall that

\beq
\calf_x \bullet=
-\;\pder{}{x_\mu}
( f_\mu\bullet) +
\frac{\partial^2}{ \partial x_\nu\partial x_\mu}( R_{\mu, \nu}\bullet)
\eeq

Define $\calb_x$ to be the same as $\calf_x$
but with every derivative 
$\partial_{x_i}$ 
replaced by its negative
and the order
of the operators
acting on $\bullet$ 
reversed. Hence

\beq
\calb_x \bullet=
f_\mu\pder{\bullet}{x_\mu}
 + R_{\mu, \nu}
\frac{\partial^2\bullet}{\partial x_\mu\partial x_\nu}
\eeq



\
The {\bf forward FP equation} for transition matrix $ P(\XT{x}{t}|\XT{x_0}{t_0})$
where $t_0<t$,
is  defined as
	
\beq
\pder{P(\XT{x}{t}|\XT{x_0}{t_0})}{t}=
\calf_x P(\XT{x}{t}|\XT{x_0}{t_0})
\label{eq-forward-fp-tm}
\eeq

The {\bf backward FP (or Kolgomorov) equation} for transition matrix $P(\XT{x_T}{T}|\XT{x}{t})$
where $t<T$,
is defined as

\beq
-\pder{P(\XT{x_T}{T}|\XT{x}{t})}{t}=
\calb_x P(\XT{x_T}{T}|\XT{x}{t})
\label{eq-backward-fp-tm}
\eeq


The forward FP equation is the
time reversed version of the
backward FP equation. Thus, they describe
the same stochastic process,
in opposite time directions.

shows the TPM arrows for the

in red and for the

\section{FP Equations in Dirac Notation}

When expressed in succinct
Dirac notation,
the derivation  of the forward and
backward FP equations
becomes very clear.
For simplicity,
we will consider only the
1-dim case. 

Suppose

\beq
d\rvx_t = f(\rvx_t)dt + L(\rvx_t)d\rvB_t
\eeq
Let

\beq
\Omega_x\bullet = 
\partial_x(f(x)\bullet)
+ \frac12 q \partial_x^2 (L(x)\bullet)
\eeq
and 
\beq
\Omega_x^\dagger\bullet =
-
f(x)\partial_x\bullet
+ \frac12 q L(x) \partial_x^2 \bullet
\eeq
with $qdt=d\rvB_t d\rvB_t$.


\begin{claim}(Forward FP equation)

If $\phi(x)$ is a smooth function
of $x$,
then

\beq
\partial_t \phi(x_t)= \Omega_x\phi(x_t)
\label{eq-pde-Omega-phi}
\eeq
\beq
\partial_tP(\XT{x}{t}|\XT{x_0}{t_0})=
\underbrace{\Omega_x^\dagger}_
{\calf_x}
P(\XT{x}{t}|\XT{x_0}{t_0})
\eeq
and
\beq
\partial_tP_{t|t_0}= \Omega_\rvx^\dagger P_{t|t_0}
\eeq
\end{claim}
\proof

Eq.(\ref{eq-pde-Omega-phi}) follows directly from It\^{o}'s Lemma
assuming $\phi(x)$ does not depend
explicitly on $t$.

Next, we express in Dirac notation, the expected value of
both sides of Eq.(\ref{eq-pde-Omega-phi}) 
(with $x_t$
replaced by $\rvx_t$).
\beqa
E[\phi(\rvx_t)|\rvx_0=x_0] &=&
\int dx \;\phi(x)P(\XT{x}{t}|\XT{x_0}{t_0})
\\
&=&\sum_x\av{\phi|x}\av{x|P_{t|t_0}|x_0}_{t_0}
\\
&=&
\av{\phi|P_{t|t_0}|x_0}_{t_0}
\eeqa
Note that the time dependence
in this expected value gets transferred from inside the $\phi(\cdot)$ to inside the $P(\cdot)$. This is a crucial observation. It's
the reason why $\partial_t$ and $E[\cdot]$
commute.



\beqa
E[\partial_t\phi(\rvx_t)|\rvx_0=x_0]
&=&
\int dx \;\phi(x)\partial_tP(\XT{x}{t}|\XT{x_0}{t_0})
\\&=&
\sum_x\av{\phi|x}\av{x|\partial_tP_{t|t_0}|x_0}_{t_0}
\\
&=&
\bra{\phi}
\partial_tP_{t|t_0}\ket{x_0}_{t_0}
\eeqa

\beqa
E[\Omega_\rvx\phi(\rvx_t)|\rvx_0=x_0]
&=&
\int dx [\Omega_x\phi(x)]
P(\XT{x}{t}|\XT{x_0}{t_0})
\\
&=&
\int dx\; \phi(x)\Omega_x^\dagger
P(\XT{x}{t}|\XT{x_0}{t_0})
\quad \text{(integrate by parts)}
\\
&=&\sum_x\av{\phi|x}\bra{x}
\Omega^\dagger_\rvx
P_{t|t_0}\ket{x_0}_{t_0}
\\
&=&\bra{\phi}
\Omega^\dagger_{\rvx}
P_{t|t_0}\ket{x_0}_{t_0}
\eeqa
\qed


\begin{claim}(Backwards FP 
(or Kolgomorov)  equation)

If $\phi(x)$ is a smooth function
of $x$ with {\bf payoff function}
\beq
u(x, t)=  E[\phi(\rvx_T)|\rvx_t=x]
\eeq
where $t<T$,
then

\beq
-\partial_t u(x, t)= \Omega_xu(x, t)
\quad
\text{with {\bf terminal BC} }u(x, T)= \phi(x)
\label{eq-pde-Omega-u}
\eeq
\beq
-\partial_tP(\XT{x_T}{T}|\XT{x}{t})=
\underbrace{\Omega_x}_
{\calb_x}
P(\XT{x_T}{T}|\XT{x}{t})
\eeq
and
\beq
-\partial_tP_{T|t}= \Omega_{\rvx} P_{T|t}
\eeq
\end{claim}
\proof

Begin by noticing that
$u(\rvx_{\tau},\tau )$
for $t<\tau<T$ is a
martingale. 
\beq
E[u(\rvx_\tau, \tau)|\rvx_t=x]= u(x,t)
\eeq
(This was
proven as an example 
in the section on martingales)
Hence, $du$ has no drift. 

By It\^o's Lemma,

\beq
d u(\rvx_\tau, \tau)=\partial_\tau u(\rvx_\tau, \tau)+
\Omega_x u(\rvx_\tau, \tau)
+ \frac12  qL^2 d\rvB_\tau
\eeq
Setting the drift to zero gives 


\beq
0=\partial_\tau u(\rvx_\tau, \tau)+
\Omega_x u(\rvx_\tau, \tau)
\eeq
Taking the expected value
$E[\cdot|\rvx_t=x]$
of both sides of the last 
equation and using martingale property gives
\beq
-\partial_t u(x, t)=
\Omega_x u(x, t)
\eeq

Next, we express 
 in Dirac notation, 
both sides of 
the last equation
(with $u(x,t)$ replaced by $E[\phi(\rvx_T)|\rvx_t=x]$).

\beqa
-\partial_tE[\phi(\rvx_T)|\rvx_t=x]
&=&
-
\int dx' \;\phi(x')\partial_tP(\XT{x'}{T}|\XT{x}{t})
\\&=&
-\sum_{x'}\av{\phi|x'}\av{x'|\partial_tP_{T|t}|x}_{t}
\\
&=&
-\bra{\phi}
\partial_tP_{T|t}\ket{x}_{t}
\eeqa

\beqa
\Omega_xE[\phi(\rvx_T)|\rvx_t=x]
&=&
\int dx' \;\phi(x')\Omega_xP(\XT{x'}{T}|\XT{x}{t})
\\&=&
\sum_{x'}\av{\phi|x'}
\av{x'|\Omega_\rvx P_{T|t}|x}_{t}
\\
&=&
\bra{\phi}
\Omega_\rvx P_{T|t}\ket{x}_{t}
\eeqa


\qed


Note that the
$P(\XT{x}{t}|\XT{x_0}{t_0})$ in the
forward FP equation is a probability density 
in $(x,t)$,
but the $P(\XT{x_T}{T}|\XT{x}{t})$
in the backwards FP is NOT
a probability density in $(x,t)$.


\section{First and second order statistics}
In this section,
we derive differential equations
for the  first and second order statistics
of an SDE without and with CC.
\subsection{For general SDE}
Suppose the SDE coefficients $f$ and $L$
can depend on the event $(x,t)$.


Let
\beq
m_\mu(t) = E[\rvx_\mu(t)]=
\av{\rvx_\mu(t)}
\eeq

\beq
C_{\mu, \nu}(t,s) = \av{\rvx_\mu(t), \rvx_\nu(s)}
\eeq

\beq
V_{\mu, \nu}(t) = C_{\mu, \nu}(t,t)
\eeq


\beq
dx_\mu = \left[a_\mu(t) + F_{\mu, \nu}(t)x_\nu \right] dt +
L_{\mu, \nu}(t)d\rvB_\nu
\eeq

\beq
R_{\mu, \nu} =
\frac{1}{2}L_{\mu, \alp}Q_{\alp, \beta}L^T_{\beta, \nu} =\frac{1}{2} (LQL^T)_{\mu, \nu}
\eeq

\begin{claim}
\label{cl-mt-Vt}
\beq
\frac{dm}{dt}=
a + Fm
\eeq

\beq
\frac{dV}{dt}=
VF^T +FV^T+ 2R
\eeq

\end{claim}
\proof

\beqa
\frac{dm}{dt}&=&
\av{a + F\rvx}
\\\
&=&
a + Fm
\eeqa

Let $\partial_\alp =\pder{}{x_\alp}$
\beqa
d\av{\rvx_\mu, \rvx_\nu}
&=&
dx_\alp\partial_\alp\av{\rvx_\mu, \rvx_\nu}
+
dx_\alp dx_\beta\frac{1}{2}\partial_\alp\partial_\beta\av{\rvx_\mu, \rvx_\nu}
\\
&=&
\av{\rvx_\mu,
d\rvx_\nu}
+
\av{d\rvx_\mu,
\rvx_\nu}
+ \av{d\rvx_\mu, d\rvx_\nu}
\\
&=&
\av{\rvx_\mu,
F_{\nu,\alp}\rvx_\alp dt}
+
\av{F_{\mu,\alp}\rvx_\alp dt,
\rvx_\nu}
+
2R_{\mu, \nu}dt
\\
&=&
(VF^T+FV^T+2R)dt
\eeqa
\qed

We call a {\bf propagator}, a function $\Psi:\RR^+\times \RR^+\rarrow \RR$
that satisfies
\beq
\pder{\Psi(t, s)}{t}  = F(t) \Psi(t, s)
\eeq

\beq
\begin{array}{l}
\Psi(a, c) = \Psi(a, b)\Psi(b, c)
\\
\Psi^{-1}(a,b) = \Psi(b,a)
\\
\Psi(a,a)=1
\end{array}
\eeq



\begin{claim}
\beq
m(t) = \Psi(t, t_0)m(t_0) +\int_{t_0}^{t}d\tau\;
\Psi(t, \tau)a(t)
\label{eq-mt-psi}
\eeq

\beq
V(t)=\av{\rvx(t), \rvx^T(t)}=\Psi(t, t_0)V(t_0)\Psi^T(t, t_0)
+
\int_{t_0}^t d\tau\; \Psi(t, \tau)R(\tau)
\Psi^T(t, \tau)
\label{eq-Vt-psi}
\eeq


\beq
\av{\rvx(t), \rvx^T(s)} =
\left\{
\begin{array}{ll}
V(t)\Psi^T(s,t) &\text{if } t<s
\\
\Psi(t,s)V(s) & \text{if }t\geq s
\end{array}
\right.
\label{eq-sde-cov-s-t}
\eeq
\end{claim}
\proof
Eqs.(\ref{eq-mt-psi} and (\ref{eq-Vt-psi}))
can be proven simply by taking the time derivative of both sides.
This yields the differential
equations for $m(t)$ and $V(t)$ that were established in Claim \ref{cl-mt-Vt}.
We won't prove Eq.(\ref{eq-sde-cov-s-t}) here.
For a proof,  see
Ref.\cite{sde-applied-book}.
\qed



If $P(x,t=0)$ is a Gaussian, $P(x,t)$ must be Gaussian too,
because the transformation is linear. Therefore,

\beq
P(x,t) = P(x(t))=
\caln(x(t); \mu=m(t), \Sigma^2=V(t))
\eeq

\beq
P(x(t)|x(s)) =\caln(x(t); \mu=m(t|s), \Sigma^2=V(t|s))
\eeq
where

\beq
m(t|s) =
\Psi(t,s)x(s)
+\int_{t_0}^t d\tau \Psi(t, \tau)a(\tau)
\eeq

\beq
V(t|s)=\int_s^td\tau\;
\Psi(t, \tau)
R(\tau)\Psi^T(t, \tau)
\eeq

This transition matrix $P(x(t)|x(s))$, when we discretize time and set $x(t_k)=x_k$, defines the
dynamic bnet of Fig.\ref{fig-1st-2nd-order-bnet}.

\beq

\eeq

\beq
a_k = \int_{t_k}^{t_{k+1}}
d\tau\; \Psi(t_{k+1}, \tau)a(\tau)
\eeq

\beq
\Sigma_k=
\int_{t_k}^{t_{k+1}}d\tau\;
\Psi(t_{k+1}, \tau)R(\tau)\Psi^T(t_{k+1}, \tau)
\eeq
The TPMs,
printed in blue, for the bnet of Fig.\ref{fig-1st-2nd-order-bnet}, are as follows:

\beq \color{blue}

\eeq

\beq \color{blue}

\eeq

\subsection{In case SDE has CC}
When the SDE has CC,
\beq
\Psi(t, s) = e^{(t-s)F}
\eeq


At time $t=\infty$,
a system with CC reaches
a {\bf steady state} (i.e., zero time derivatives of expected values)
with mean value $m_\infty$
and variance $V_\infty$ given by
\beq
0=a + Fm_\infty
\eeq
\beq
0=V_\infty F^T  + F v_\infty^T + 2R
\eeq


\section{Fourier Analysis for CC case}

In this section, we will apply Fourier
analysis to the SDE with CC.

We begin by recalling a few definitions from Fourier
analysis.

\newcommand{\Tintegral}[1]{
\int_{-\;\frac{T}{#1}}^{\frac{T}{#1}}}

The {\bf Dirac delta function} satisfies:

\beq
\delta(t) =\frac{1}{2\pi} \int_{-\infty}^{\infty}d\omega \;e^{i\omega t}
\eeq
The {\bf Fourier transform} of $x(t)$ is
\beq
\TIL{x}(\omega)=\calf[x(t)](\omega)=
\int_{-\infty}^{\infty}dt\; x(t) e^{-i\omega t}
\eeq
The {\bf inverse Fourier transform} of $\TIL{x}(\omega)$ is
\beq
x(t)=\calf^{-1}[\TIL{x}(\omega)](t)=
\frac{1}{2\pi}
\int_{-\infty}^{\infty}d\omega\; \TIL{x}(\omega) e^{i\omega t}
\eeq

Define the {\bf bounded-time Fourier transform} $\TIL{x}_T(\omega)$ by

\beq
\TIL{x}_T(\omega)=\Tintegral{2}
dt\;x(t) e^{-i\omega t}
\eeq


Define the
{\bf power spectral density}
of process $x(t)$ by
\beq
Pow_\rvx(\omega)= \lim_{T\rarrow 0}
\frac{1}{T} E[\TIL{x}_T(\omega)\TIL{x}^\dagger_T(\omega)]
\eeq
In case $\rvx=\dot{\rvB}$= white noise,


\beq
S_{\dot{\rvB}}(\omega) = Q
\eeq

Define the
{\bf autocorrelation function for stationary process $\rvx(t)$} by
\beq
AC_\rvx(\tau)=
E[x(t)x^T(t+\tau)]
\eeq
$\tau$ is called the {\bf time lag}.

\begin{claim}


\beq
\Tintegral{2}dt
\Tintegral{2}ds\;
g(t-s)=
\int_{-T}^{T}d\tau\;g(\tau)(T-|\tau|)
\label{eq-ac-integral-formula}
\eeq
\end{claim}
\proof

Let

\beq
\tau=\frac{t-s}{\sqrt{2}},\quad
\sigma=\frac{t+s}{\sqrt{2}}
\eeq
The absolute value of the Jacobian $|\frac{\partial(s, t)}{\partial(\tau, \sigma)}|$ equals 1.
Hence,

\begin{align}
\Tintegral{2}dt
\Tintegral{2}ds\;
g(t-s)
&=
\Tintegral{\sqrt{2}}d\tau
\int_{-\;\frac{T}{\sqrt{2}}-|\tau|}
^{\frac{T}{\sqrt{2}}+|\tau|}d\sigma \;g(\sqrt{2}\tau)
\quad(\text{See Fig. \ref{fig-integral-region}})
\\
&=
\Tintegral{\sqrt{2}}d\tau
\;g(\sqrt{2}\tau)
(\sqrt{2}T - 2|\tau|)
\\
&=
\int_{-T}
^{T}d\tau'
\;g(\tau')
(T - |\tau'|)\quad(\tau' = \sqrt{2}\tau)
\end{align}

\begin{figure}[h!]
\centering
\includegraphics[width=1.5in]
{stochastic-diff-eqns/integral.png}
\caption{Integration
region
for integral given by Eq.(\ref{eq-ac-integral-formula})}
\label{fig-integral-region}
\end{figure}


\qed

\begin{claim}(Wiener Khinchin theorem (WK))

\beq
Pow_\rvx(\omega)
=
\int_{-\infty}^{\infty}d\tau\; AC_\rvx(\tau)e^{i\omega \tau}
\eeq
If $AC(\tau)=AC(-\tau)$,

\beq
Pow_\rvx(\omega)
=
2\int_{0}^{\infty}d\tau\; AC_\rvx(\tau)\cos(\omega\tau)
\eeq

\end{claim}
\proof

\begin{align}
\int_{-\infty}^{\infty}d\tau\;
AC_\rvx(\tau)
e^{i\omega\tau}
&=
\int_{-\infty}^{\infty}d\tau\;
\av{x(t)x^\dagger(t+\tau)}
e^{i\omega\tau}
\\
&=
\int_{-\infty}^{\infty}d\tau
\int_{-\infty}^{\infty}\frac{d\omega_1}{2\pi}
\int_{-\infty}^{\infty}\frac{d\omega_2}{2\pi}
\av{\TIL{x}(\omega_1)\TIL{x}^\dagger(\omega_2)}
e^{i\omega\tau+i\omega_1t-i\omega_2(t+\tau)}
\end{align}

To get rid of the $t$ dependence on the right-hand side
of the last equation, apply $\frac{1}{T}\int_{-T/2}^{T/2}dt$ to both sides
to get


\begin{align}
\int_{-\infty}^{\infty}d\tau\;
AC_\rvx(\tau)
e^{i\omega\tau}
&=
\int_{-\infty}^{\infty}\frac{d\tau}{T}
\int_{-\infty}^{\infty}\frac{d\omega_1}{2\pi}
\int_{-\infty}^{\infty}d\omega_2
\av{\TIL{x}(\omega_1)\TIL{x}^\dagger(\omega_1)}
e^{i(\omega-\omega_2)\tau}
\delta(\omega_1-\omega_2)
\\
&=
\int_{-\infty}^{\infty}\frac{d\tau}{T}
\int_{-\infty}^{\infty}\frac{d\omega_1}{2\pi}
\av{\TIL{x}(\omega_1)\TIL{x}^\dagger(\omega_1)}
e^{i(\omega-\omega_1)\tau}
\\
&=
\frac{1}{T}
\int_{-\infty}^{\infty}d\omega_1
\av{\TIL{x}(\omega_1)\TIL{x}^\dagger(\omega_1)}
\delta(\omega-\omega_1)
\\
&=
\frac{1}{T}
\av{\TIL{x}(\omega)\TIL{x}^\dagger(\omega)}=
\frac{1}{T}
\av{\TIL{x}(\omega)\TIL{x}^T(\omega)}
\end{align}
Alternative proof:


\beqa
\frac{1}{T}
E[|\TIL{x}_T(\omega)|^2]
&=&\frac{1}{T}
\Tintegral{2}ds
\Tintegral{2}dt\;
E[x(t)x^T(s)]e^{-iw(t-s)}
\\
&=&\frac{1}{T}
\int_{-T}^{T}d\tau\;
AC(\tau)e^{i\omega \tau}(T-|\tau|)
\\
&\rarrow&
\int_{-\infty}^{\infty}d\tau\; AC(\tau)e^{i\omega \tau}
\quad\text{(Use Eq.(\ref{eq-ac-integral-formula}))}
\eeqa
\qed

As a trivial example of the WK theorem, note that
\beq
AC_{\dot{\rvB}}(\tau) = Q\delta(\tau)
\eeq
Hence,  by the WK theorem,

\beq
Pow_\rvx(\omega)=Q
\eeq

Next, let us solve the
SDE with CC using Fourier transforms.
Substituting Fourier transforms for $x(t)$
into this equation
\beq
\frac{d x}{dt}=Fx + L\dot{\rvB}
\eeq
we get

\beq
-i\omega \TIL{x} = F\TIL{x} + L\TIL{\dot{\rvB}}
\eeq
Hence,

\beq
\TIL{x}= -( F+i\omega)^{-1}L\TIL{\dot{\rvB}}
\eeq

\beqa
Pow_{\dot{\rvB}}(\omega) &=& \TIL{x}\TIL{x}^\dagger
\\
&=&
(F+i\omega)^{-1}\underbrace{LQL^\dagger}
_{2R}(F^T-i\omega)^{-1}
\eeqa

\beq
AC_{\rvx}(\tau)=
\int_{-\infty}^{\infty}d\omega\; e^{i\omega \tau}
(F+i\omega)^{-1}(2R)
(F^T-i\omega)^{-1}
\eeq

When steady state is reached,
the expected values (averages) of functions of $\rvx(t)$ cease to vary with time. Thus, we only need to compare $\rvx(t)$
to itself, not to $\rvx(t+\tau)$ with $\tau>0$.
Hence, we only need to know $AC(\tau)$ at $\tau=0$.


For steady state, by the Wiener Khinchin theorem,
\beq
Pow_\rvx(\omega)=AC_\rvx(0) 2\pi\delta(w)
\eeq
where

\beq
AC_\rvx(0)=
\int_{-\infty}^\infty
dw\;(F+i\omega)^{-1}(2R)
(F^T-i\omega)^{-1}
\eeq


\section{Lamperti Transformation}

The {\bf Lamperti Transformation} answers
the following question: If the next two
SDE are satisfied, express $g_\mu$ in terms of $f_\mu$ and $L_{\mu, \nu}$
\beq
dx_\mu = f_\mu(x,t)dt + L_{\mu,\nu}(x,t)d\rvB_\nu
\eeq

\beq
dy_\mu = g_\mu(y,t)dt + d\rvB_\mu
\label{eq-g-def-grisa}
\eeq
Note that $g\neq f/L$
because the product $Ld\rvB$
is not simple multiplication.


\begin{claim} For $n=1$, the function
$g(y,t)$ in Eq.(\ref{eq-g-def-grisa})
is given by

\beq
g(y,t)=
\left.\left(
\frac{f(\XT{x}{t})}{L(\XT{x}{t})}
+
\pder{}{t}
\int_\xi^x \frac{du}{L(\XT{u}{t})}
-\;
\frac{1}{2}
\pder{L(\XT{x}{t})}{x}
\right)\right|_{x\rarrow h^{-1}(y,t)}
\eeq
\end{claim}
\proof
Suppose
\beq
y=h\eqdef\int_\xi^x\frac{du}{L(u,t)}
\eeq
Then

\beqa
dy &=& \pder{h}{t}dt + \frac{dx}{L}
-\;\frac{1}{2L^2}
\pder{L}{x}(dx)^2
\\
&=&
 \pder{h}{t}dt + \frac{fdt +  Ld\rvB}{L}
-\;\frac{1}{2L^2}
\pder{L}{x}\underbrace{(Ld\rvB)^2}_{L^2dt}
\\
&=&
\left(
\pder{h}{t} + \frac{f}{L} -\frac{1}{2}\pder{L}{x}
\right)dt + d\rvB
\eeqa
\qed

\section{Karhunen–Loève series}

In this section
we explain the {\bf Karhunen–Loève (KL) series}. The most intuitive
way of explaining  KL series is
using Dirac bra-ket notation.

Let $\{\ket{t}
 : t\in [0,T]\}$ be a complete orthonormal basis
\beq
\int_0^{T}dt\;\ket{t}{\bra t} = 1,
\;
\av{t|t'}=\delta(t-t')
\eeq
Express $\rvx(t)$ in bra-ket notation
\beq
\rvx(t)=\av{t|\rvx},
\quad
\rvx^*(t)=\av{\rvx|t},
\eeq

Consider the operator
\beq
C_\rvx=E[\ket{\rvx}\bra{\rvx}]
-
E[\ket{\rvx}]E[\bra{\rvx}]
\eeq
The matrix elements
of $C_\rvx$ are:

\beq
C_\rvx(t, t')= \av{t|C_\rvx|t'}
=E[\rvx(t)\rvx^*(t')]-E[\rvx(t)]
E[\rvx^*(t')]
\eeq
Hence, $C_\rvx(t,t')$ is the correlation matrix in the $\{\ket{t}:t\in [0,T]\}$ basis.

Suppose $C_\rvx$ has eigenvalue,
eigenvector pairs $(\lam_n, \ket{\phi_n})$ for
 $n=1,2,3,\ldots$. Hence
\beq
C_\rvx\ket{\phi_n}=\lam_n\ket{\phi_n}
\eeq
where the states $\ket{\phi_n}$
for all $n$, are a complete
orthonormal basis. Then

\beq
\sum_{n=1}^\infty \ket{\phi_n}\bra{\phi_n}=1,
\;
\av{\phi_n|\phi_{n'}}=
\delta(n,n')
\eeq


\beq
C_\rvx = \sum_{n=1}^\infty \ket{\phi_n}\lam_n\bra{\phi_n}
\eeq


If we define
\beq
\phi_n(t) = \av{t|\phi_n},
\;
\phi_n^*(t) = \av{\phi_n|t}
\eeq

\beq
\rvx^*_n = \av{\rvx|\phi_n}=
\int_0^Tdt\;\av{\rvx|t}\av{t|\phi_n},
\quad
\rvx_n = \av{\phi_n|\rvx}=
\int_0^T dt\; 
\av{\phi_n|t}\av{t|x}
\eeq
then

\beq
\ket{\rvx}=\sum_n \rvx_n\ket{\phi_n}
\eeq
Note also that both

\beq
\av{\phi_n|C_\rvx|\phi_m}=
E[\rvx_n\rvx^*_m]-
E[\rvx_n]E[\rvx^*_m]
\eeq
and

\beq
\av{\phi_n|C_\rvx|\phi_m}=\lam_n\delta(n,m)
\eeq
are satisfied. Therefore,


\beq
\boxed
{E[\rvx_n\rvx^*_m]-
E[\rvx_n]E[\rvx^*_m]
=
\lam_n\delta(n,m)}
\eeq






\begin{claim}
(Karhunen–Loève expansion for Brownian motion) 

If $\rvB(t)$ is
Brownian motion and 

\beq
C_\rvB = \sum_{n=0}^{\infty} \ket{\phi_n}\lam_n\bra{\phi_n}
\eeq
then

\beq
\rvB(t) =\sum_{n=0}^\infty z_n\av{t|\phi_n}
\eeq
with

\beq
z_n\sim \caln(\mu=0, \s^2 =\lam_n)
\eeq
and

\beq
\av{t|\phi_n}=\sqrt{\frac{2}
{T}}\sin \omega_n t
\quad \text{ where }
\omega_n = \frac{\pi}{T}(n +\frac{1}{2})=\frac{1}
{\sqrt{\lam_n}}
\eeq

\end{claim}
\proof
\beq
E[\rvB_t\rvB_s]-\underbrace{E[\rvB_t]E[\rvB_s]}_0=
E[\rvB_t\rvB_s]=\min(t,s)
\eeq

\beq
\int_0^T dt\;
\min(s,t)\phi_n(t) = \lam_n
\phi_n(s), \; 0\leq s\leq T
\eeq

\beqa
\int_0^T dt\;
\min(s,t)\sin\omega_n t=
\int_0^s dt\; t\sin\omega_n t
+
\int_s^T dt\;s\sin\omega_n t
\eeqa
\beqa
\int_0^s dt\;t\sin\omega_n t &=&
\left.\frac{\sin\omega_nt}{\omega_n^2}\right|_{t=0}^s
-\left.\frac{t\cos\omega_n t}{\omega_n}\right|_{t=0}^s
\\
&=&
\frac{\sin\omega_ns}{\omega_n^2}
-
\frac{s\cos\omega_ns}{\omega_n}
\eeqa

\beq
\int_s^T dt\; s\sin\omega_n t
=
\left.-\;\frac{s\cos \omega_nt }{\omega_n}\right|_{t=s}^T
=
\frac{s\cos \omega_n s}{\omega_n}
\eeq
Hence

\beq
\int_0^T dt\;
\min(s,t)\sin\omega_n t=
\frac{\sin(\omega_ns)}{\omega_n^2}=
\lam_n \sin(\omega_n s)
\eeq

The $\sqrt{\frac{2}{T}}$
factor in the definition
of $\phi_n(t)$
is necessary to ensure
that $\av{\phi_n|\phi_n}=1$.

$\omega_n$ has the
the indicated values because $\phi_n(t=0)=0$
and $\phi_n(t=T)=1$ 


$\phi_n(t=0)=0$ because $\rvB(0)=0$.
$\phi_n(t=1)=1$ 
because
\beqa
T&=&E[\rvB(T)\rvB(T)]
\\
&=&
\frac{2}{T}
\sum_{m=0}^\infty
\sum_{n=0}^\infty
E[z_mz_n]
\sin{\omega_m T}
\sin{\omega_n T}
\\
&=&
\frac{2}{T}
\sum_{n=0}^\infty
\lam_n
\\
&=&
T\frac{2}{\pi^2}
\underbrace{
\sum_{n=0}^\infty
\frac{1}{(n+1/2)^2}}_{=\frac{\pi^2}{2}=\zeta(2, 1/2)}
\eeqa

\qed


\section{Girsamov Theorem}

In this section,
we prove the {\bf Girsamov theorem}.

\begin{claim}(Girsanov Theorem)
\label{cl-girsanov-theo}

If $\rvx_t,
\rvy_t\in\RR^n$ are
semi-martingales such that
\beq
\left\{
\begin{array}{l}
d\rvx_t = f(\rvx_t, t)dt + d\rvB_t
\\
d\rvy_t = g(\rvx_t, t)dt + d\rvB_t
\end{array}\right.
\eeq
and if we define $\calr$ by
\beq
\calr=
\frac{P(x_{1\ldots N})dx_{1\ldots N}}
{P(y_{1\ldots N})dy_{1\ldots N}}
\eeq
then


\begin{enumerate}[(a)]

\item  \label{enum-R}
\beq
\calr=
\exp
\left\{
\begin{array}{l}
\frac{-1}{2}\int_{\tau=0}^td\tau\; (f(\XT{x_\tau}{\tau})-g(\XT{x_\tau}{\tau}))^TQ^{-1}(f(\XT{x_\tau}{\tau})-g(\XT{x_\tau}{\tau}))
\\
-\int_{\tau=0}^t (f(\XT{x_\tau}{\tau})-g(\XT{x_\tau}{\tau}))^T 
Q^{-1}d\rvy_\tau
\end{array}
\right\}
\eeq
\item \label{enum-exp-val}
\beq
E[h[\rvx_{1\ldots N}]]=
E[\calr h[\rvy_{1\ldots N}]]
\eeq
\item \label{enum-beta}
If we define

\beq
\theta(\XT{x_t}{t}) = (g-f)(\XT{x_t}{t})
\eeq
and

\beq
\rvB_t^{\pm \theta}=\rvB_t\pm
\int_0^t \theta(\XT{x_\tau}{\tau}) d\tau
\eeq
then $\rvB_t^{\pm \theta}$ is Brownian motion
with diffusion
matrix $Q$.
Furthermore,


\beq\text{For $d\rvB_t^{-\theta}$:}\quad
\left\{
\begin{array}{l}
dx_t = g(\XT{x_t}{t})dt + d\rvB_t^{-\theta}
\\
dy_t = (g+\theta)(\XT{x_t}{t})dt + d\rvB_t^{-\theta}\quad\text{($+\theta dt$ shift in drift term)}
\end{array}
\right.
\eeq

\beq
\text{For $d\rvB_t^{+\theta}$:}\quad
\left\{
\begin{array}{l}
dx_t = (f-\theta)(\XT{x_t}{t})dt + d\rvB_t^{+\theta}\quad\text{($-\theta dt$ shift in drift term)}
\\
dy_t = f(\XT{x_t}{t})dt + d\rvB_t^{+\theta}
\end{array}
\right.
\eeq







\end{enumerate}
\end{claim}
\proof

We will
only prove this for $n=1$,
$Q=q$.
The case of general $n$ 
is closely analogous, but 
with matrices instead of scalars.

Proof of \ref{enum-R}:
\beqa
\frac{P(x_{1\ldots N})}
{P(y_{1\ldots N})}
&=&
\frac{-1}{2q}
\int_{\tau=0}^t d\tau\;
[(\dot{x}_\tau)^2
-(\dot{y}_\tau)^2]
\quad \text{(from Eq.(\ref{eq-px-div-pb-source}))}
\\
&=&
\frac{-1}{2q}
\int_{\tau=0}^t d\tau\;
[\dot{x}_\tau
-\dot{y}_\tau]
[\dot{x}_\tau
+\dot{y}_\tau]
\\
&=&
\frac{-1}{2q}
\int_{\tau=0}^t d\tau\;
[f-g][f+g+2\dot{\rvB}]
\\
&=&
\left\{
\begin{array}{l}
\frac{-1}{2q}
\int_{\tau=0}^t d\tau\;
[f-g]^2
\\
+
\frac{-1}{q}
\int_{\tau=0}^t d\tau\;
[f-g]
\underbrace{[g+\dot{\rvB}]}_{\dot{y}_\tau}
\end{array}
\right\}
\eeqa
Proof of \ref{enum-exp-val}:

\beq
dx_{1\ldots N}=
dy_{1\ldots N},
\quad
h[x_{1\ldots N}]=
h[y_{1\ldots N}]
\eeq
Hence

\beqa
E[h[\rvx_{1\ldots N}]]
&=&
\int d\rvx_{1\ldots N}\;
P(x_{1\ldots N})  
h[x_{1\ldots N}]
\\
&=&
\int d\rvy_{1\ldots N} 
P(y_{1\ldots N})  
\calr  h[y_{1\ldots N}]
\\
&=&
E[\calr h[\rvy_{1\ldots N}]]
\eeqa
Proof of \ref{enum-beta}:
\beq
d\rvB_t^{\pm \theta}=
d\rvB_t \pm \theta(\XT{x_t}{t})dt
\eeq
so

\beq
 [d\rvB^{\pm \theta}] ^2
=[d\rvB] ^2 = q\;dt
\eeq



\qed
\hrule
$dP_\rvx=P(x_{1\ldots N})dx_{1\ldots N}$ corresponds to
what is called an infinitesimal probability measure in 
more sophisticated treatments.

\beq
\calr  = \frac{dP_\rvx}{dP_\rvy}=
\frac{P(x_{1\ldots N})dx_{1\ldots N}}
{P(y_{1\ldots N})dy_{1\ldots N}}
=
\frac{P(x_{1\ldots N})}
{P(y_{1\ldots N})}

$P(x_{1\ldots N})/

Note that expectation values
over paths depend on measure $dP_\rvx$:
\beq
E_{dP_\rvx}[h[\rvx_{1\ldots N}]]=\int dx_{1\ldots N}
P(x_{1\ldots N})
h[\rvx_{1\ldots N}]
\eeq

\eeq

\right\}
\eeq
so

E[\calr]&=& 1- \frac{1}{2}\int_{\tau=0}^td\tau\; \theta(\XT{x_\tau}{\tau})^2

\eeqa

\section{Doob's $h$ transform}
Doob's h-transform is as a modification of the drift in the backward equation by conditioning the diffusion on a future event.

Assume
\beq
d\rvx_t=f(\rvx_t, t)
dt +  L(\rvx_t, t)d\rvB_t
\eeq

\beq
\calb_x= \Omega_x=
f(x,t)\partial_x + \frac12 L(x,t)^2
\partial_x^2
\eeq

Suppose $u(x,t)$
and $h(x,t)$ 
are positive functions 
(such as products of probabilities)
that are solutions  to the
backward FP equation

\beq
\left\{
\begin{array}{l}
-\partial_t u(x,t) =
\calb_x u(x, t)
\\
-\partial_t h(x,t) =
\calb_x h(x, t)
\end{array}
\right.
\eeq
Define a new process

\beq
U(x,t)= 
\frac{u(x,t)}{h(x,t)}
\eeq

\begin{claim}
$U(x,t)$ satisfies
\beq
-\partial_tU(x, t)=\calb^{(h)}_x U(x,t)
\eeq
where 
\beq
\calb^{(h)}_x= (\calb_x)|_{f\to f^{(h)}}
\eeq
where

\beq
f^{(h)}(x,t)= f(x,t) + L^2\partial_x  h(x,t)
\eeq
\end{claim}
\proof

\beq
-\partial_x(hU)=  \calb_x (hU)
\label{eq-doob-start}
\eeq

\beq
\partial_x hU =
h\partial_x U 
+ U\partial_x h
\eeq

\beqa
\partial_x^2 hU
&=&
\partial_x[h\partial_x U 
+ U\partial_x h
]
\\
&=&
h\partial_x^2U
+2(\partial_x U)
(\partial_x h)
+U\partial_x^2 h
\eeqa

Expanding both sides of
Eq.(\ref{eq-doob-start}) and using the fact that
$h$ satisfies the FP backwards equation, we get
\beq
-\left[
{h\partial_t U }
+ \cancel{U\partial_t h}
\right]=
\left\{
\begin{array}{l}
f\left[{h\partial_x U} 
+\cancel{ U\partial_x} h\right]
\\
+\frac12 L^2\left[
h\partial_x^2U
+2(\partial_x U)
(\partial_x h)
+\cancel{U\partial_x^2 h}
\right]
\end{array}
\right\}
\eeq

Dividing by $h$

\beqa
-\partial_t U &=&
\underbrace{\left[
f +  L^2 \partial_x(\ln h)
\right]}_{f^{(h)}} \partial_x U
+ \frac12 L^2 \partial_x^2 U
\\
&=&
\calb_x^{(h)}U
\eeqa
\qed

Define {\bf Doob's transform} as the process $\rvx^{(h)}$ 
such that
\beq
d\rvx_t^{(h)}=f^{(h)}(\rvx_t, t)
dt +  L(\rvx_t, t)d\rvB_t
\eeq

Both $u(\rvx_t, t)$ and $h(\rvx_t, t)$ are martingales for $\rvx_t$. $U(\rvx_t, t; h)$ is a martingale for $\rvx^{(h)}$.

In  $n$-dim,
the drift term is

 \beq
qL^2\partial_x h
\to 2R_{\mu,\nu}\pder{}{x_\nu}
\eeq

\hrule
Examples
\begin{enumerate}
\item
Bayes rule

Suppose $t\leq t_m\leq T$. Let

\beq
u(x, t) =\lam P(\XT{x_T}{T}|\XT{x_m}{t_m})
P(\XT{x_{m}}{t_m}|\XT{x}{t})
,\quad
h(x,t)=\lam
P(\XT{x_T}{T}|\XT{x}{t})
,\quad
U(x,t)=\frac{u(x,t)}{h(x,t)}
\eeq

\begin{itemize}
\item Assume $\lam=1:$

In this case, if the process $\rvx_t$ is Markovian 

\beq
u(x, t) = \underbrace{P(\XT{x_T}{T}|\XT{x_m}{t_m},\XT{x}{t})}_{\text{ likelihood}}
\underbrace{P(\XT{x_{m}}{t_m}|\XT{x}{t})}_{\text{prior for event $(\XT{x_m}{tm})$}}
=
\underbrace{P(\XT{x_T}{T},\XT{x_m}{t_m}|\XT{x}{t})}_{\text {posterior}}
\eeq
so

\beq
U(x,t)=P(\XT{x_m}{t_m}| \XT{x_T}{T},\XT{x}{t})
\label{eq-doob-markov}
\eeq



The drift correction is proportional to
$\nabla_x P(\XT{x_T}{T}|\XT{x}{t})$
which makes a lot of sense
if you want to induce
 the system to go from event  $(\XT{x}{t})$ to event
$(\XT{x_T}{T})$.

\item Assume $\lam=1/P(\XT{x_T}{T}|\XT{x_m}{t_m})$:

In this case,

\beq
U(x, t) = \underbrace{P(\XT{x_m}{t_m}|\XT{x}{t})}_{u}
\underbrace{\left[
\frac{P(\XT{x_T}{T}|\XT{x_m}{t_m})}{P(\XT{x_T}{T}|\XT{x}{t})}
\right]}_{=1/h=\text{importance weight}}
\eeq
As with $\lam=1$, if $\rvx_t$ is a Markovian process, then Eq.(\ref{eq-doob-markov}) holds.




\end{itemize}

\item Brownian Endpoint Bridge (Brownian motion conditioned
on a hit an endpoint)

\beq
d\rvx_t=d\rvB_t
\eeq

\beq
h(x, t)=P(\XT{x_T}{T}
|\XT{x}{t})=
\frac{1}{\sqrt{2\pi(T-t)}}
\exp\left(
\frac{-(x-x_T)^2}
{2(T-t)}
\right)
\eeq

\beq
\partial_x\ln h(x,t)=
\frac{x_T-x}{T-t}
\eeq

\beq
d\rvx^{(h)}=
\frac{x_T-x}{T-t} dt + d\rvB_t
\eeq

$\partial_x \ln h$ acts  like a force. $-\ln h \sim (x-x_T)^2/(T-t)$,
a parabola centered $x=x_T$, 
acts like a potential.
The particle is attracted towards $(x_T, T)$.

\end{enumerate}

\begin{claim}
\label{cl-ratio-2-martingales-h}

If $h(\rvx_t, t)$
is a martingale, then
\beq
\frac{h(\rvx_t, t)}
{h(\rvx_0, t_0)}=
\exp\left(
-
\frac12 \int_0^t d\tau\;
\theta_\tau^2
+
\int_0^t \theta_\tau d\rvB_\tau
\right)
\eeq
where
\beq
\theta_t=
L(\rvx_t, t)
\partial_x \ln h(\rvx_t, t)
\eeq
\end{claim}
\proof

Apply It\^o's Lemma to   $\ln h(\rvx_t, t)$
($h(\rvx_t, t)$ is a martingale by assumption, but $\ln h$
is a semi-martingale)

\beq
d\ln h=
(\partial_t \ln h +
\Omega_x \ln h)dt +
L(\partial_x\ln h)d\rvB_t
\eeq
where
\beq
\Omega_x = 
f(x,t)\partial_x
+\frac12 L^2(x,t)\partial_x^2
\eeq

\beq
\partial_x \ln h=
\frac{\partial_x h}{h}
\eeq

\beqa
\partial_x^2 \ln h
&=&
\partial_x
\frac{\partial_x h}{h}
\\
&=&
\frac{-(\partial_x h)^2}{h^2}+
\frac{\partial_x^2h}{h}
\\
&=&
-(\partial_x \ln h)^2 + \frac{1}{h}
\partial_x^2 h
\eeqa

\beqa
\Omega_x \ln h&=&
f\left[\frac{\partial_x h}{h}\right]
+\frac12 L^2
\left[
-(\partial_x \ln h)^2 + \frac{1}{h}
\partial_x^2 h
\right]
\\
&=&
\frac{\Omega_x h}{h}
-
\frac{L^2}{2} 
(\partial_x \ln h)^2
\eeqa

$h(\rvx_t, t)$
is a martingale so
 \beq
\partial_t h +
\Omega_x h=0
\eeq
Hence
\beqa
d\ln h&=&
\left(-\frac{L^2}{2}
(\partial_x \ln h)^2
\right)dt
+ L(\partial_x \ln h)d\rvB_t
\\
&=&
-\;\frac12 \theta^2
dt + \theta d\rvB_t
\eeqa
Now integrate 
both sides

\beq
\ln\left[
\frac{h(\rvx_t, t)}
{h(\rvx_0, 0)}
\right]=
\int_0^t
\left[
-\;\frac12 \theta^2
dt + \theta d\rvB_t
\right]
\eeq
Finally, exponentiate both sides.
\qed

\begin{claim}

If $\rvx_t,
\rvy_t\in\RR^n$ are
semi-martingales such that
\beq
\left\{
\begin{array}{l}
d\rvx_t = f(\rvx_t, t)dt + L(\rvx_t, t)d\rvB_t
\\
d\rvy_t = g(\rvx_t, t)dt + L(\rvx_t, t)d\rvB_t
\end{array}\right.
\eeq
and if we define $\calr$ by
\beq
\calr=
\frac{P(x_{1\ldots N})dx_{1\ldots N}}
{P(y_{1\ldots N})dy_{1\ldots N}}
\eeq
and 

\beq
\theta_t = (f-g)(\rvx_t, t)= L(\rvx_t, t)\partial_x \ln h(\rvx_t, t)
\eeq
for a martingale $h(\rvx_t, t)$, 
then

\beq
\calr = \frac{h(\rvx_t, t)}{h(\rvx_0, t)}
\eeq

\end{claim}
\proof
Just combine Claims \ref{cl-girsanov-theo} and \ref{cl-ratio-2-martingales-h}
\qed



\section{Appendix: Some explicitly solvable SDE examples}
Let $\mu, \nu, \alpha, \beta\in [n]$
\beq
d\rvx_\mu= f_\mu(\rvx, t)dt + L_{\mu, \nu}(\rvx, t)d\rvB_\nu(t)
\eeq


\beq
\pder{P}{t}=
-\;\pder{Pf_\mu}{x_\mu}
 + \frac{\partial^2}
 { \partial x_\mu\partial x_\nu}
 \left(PR_{\mu, \nu}\right)
\eeq

\begin{itemize}

\item
{\bf Brownian motion} ($f_\mu=0$, $R_{\mu,\nu}=D\delta(\mu, \nu)$)

\beq
dx_\mu =d\rvB_\mu
\eeq


\item {\bf Overdamped Langevin Equation}
($f_\mu=-\frac{1}{2}\pder{U}{x_\mu}$, $R_{\mu,\nu}=D\delta(\mu, \nu)$)

\beq
dx_\mu = -\;\frac{1}{2}\pder{U}{x_\mu}dt + d\rvB_\mu
\eeq

\item {\bf Ornstein–Uhlenbeck process (a.k.a. Langevin equation)} ($f_\mu=-\lam x_\mu$, $R_{\mu,\nu}=D\delta(\mu, \nu)$)

This is the same as the Langevin equation, if we identify $\rvx$ with
the velocity of the
particle and $\lam$ with the drag coefficient.


\beq
d\rvx = -\lam \rvx dt + d\rvB
\eeq

\item
{\bf 1-dim ($n=1$) Black-Sholes} ($f=a x$, $R=(b\rvx)^2 q/2$)

\beq
d\rvx = a \rvx dt + b \rvx d\rvB
\eeq

\item {\bf 1-dim ($n=1$) SDE with
($f=a x + c$, $R=(b\rvx + d)^2 q/2$)}


\beq
d\rvx = [a(t)\rvx +c(t)]dt + [b(t)\rvx+ d(t)]d\rvB
\eeq


\beq
x(t) =\Psi(t, t_0)
\left(
x(t_0)
+
\int_{t_0}^{t}ds\;\Psi^{-1}(s, t_0)[c(s)-b(s)]
+
\int_{t_0}^{t}
\Psi^{-1}(s, t_0)d(s)d\dot{\rvB}(s)
\right)
\eeq

\beq
\Psi(t, t_0)=
\exp\left(
\int_{t_0}^t ds\; \left[a(s)-\frac{1}{2}b^2(s)\right]
+
\int_{t_0}^t b(s) d\dot{\rvB}(s)
\right)
\eeq

\end{itemize}
