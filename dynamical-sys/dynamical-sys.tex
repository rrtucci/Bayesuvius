\chapter{Dynamical Systems}
\label{ch-dynamical-sys}

This chapter is based on 
Ref.\cite{dynamical-fuchs}

\beq
\dot{x} = f(x, t)
\eeq

\beq
f(t, x) = -\pder{V}{{x}}
\eeq

\beq
V - V_0= -\int^x_{x_0} dx \; f(x, t)
\eeq


\section{Linear System}

\beq
\dot{x} = \lam x
\quad 
\xymatrix{
&\rvx\ar[d]|{\;\redplus}
\\
&\dot{\rvx}
}
\eeq

\beq
V = -\;\frac{\lam}{2}x^2
\eeq

\section{Quadratic System}

\beq
\dot{x} = \lam x - x^2
\quad 
\xymatrix@R=.5pc{
&\rvx\ar[dd]|{\;\redplus}
\ar[dl]
\\
\rvx^2\ar[dr]|\redminus
\\
&\dot{\rvx}
}
\eeq

\beq
V =-\; \frac{\lam}{2}x^2  + \frac{1}{3} x^3
\eeq 

\begin{figure}[h!]
\centering
\includegraphics[width=4in]

\caption{$(x, \dot{x})$ and $(x, V)$ plots for quadratic
system $\dot{x}=\lam x-x^2$.}
\label{fig-phase-V-quadratic}
\end{figure}

\section{Cubic System}

\beq
\dot{x} = v_0 + \lam x - x^3
\quad 
\xymatrix@R=.5pc{
&\rvx\ar[dd]|{\;\redplus}
\ar[dl]
\\
\rvx^3\ar[dr]|\redminus
\\
&\dot{\rvx}&\ar[l]
}
\eeq
\beq
V = -v_{0}x -\;\frac{\lam}{2}x^2  + \frac{1}{4} x^4
\eeq
