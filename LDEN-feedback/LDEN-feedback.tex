\beqa
\rvx_0^{[t+1]} &=& \underbrace{\sum_{j=0}^1\beta_{0|j}\rvx_j^{[t]}
 }_{\text{from past}}
 +\rvu_0^{[t+1]}

\underbrace{\sum_{j=0}^1\beta_{1|j}\rvx_j^{[t]}
}_{\text{from past}}

\eeqa

$\rvx^{[0]}_j=0$ for all $j$.

\beq
\rvx^{[t+1]}_j
\maparrow{\calz}

\left(
\TIL{\rvx}_j(z)

\rvx^{[0]}_j

\eeq

\beqa

\TIL{\rvx}_j(z)
 +z\TIL{\rvu}_0(z)

z\TIL{\rvx}_1(z) &=&
\sum_{j=0}^1\beta_{1|j}\TIL{\rvx}_j(z)
+  \alp_{1|0}z\TIL{\rvx}_0(z) + z\TIL{\rvu}_1(z)
\eeqa
