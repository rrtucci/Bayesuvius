\section{Kramers-Moyal Expansion}

The {\bf Kramers–Moyal expansion} is a Taylor expansion to all orders in $x$ and first order in $t$ of the Chapman–Kolmogorov equation for a \ul{Markovian process}.


Let 

$p( x, t)=$ denote the probability density of a Markov process $\rvx_t$

$P( x, t+\tau\mid  x', t)=$
the transition probability of $\rvx_t$

The expansion expresses the evolution equation for $p( x, t)$ in terms of the jump moments of the process.



 

For a Markov process  the 
Chapman–Kolmogorov equation implies
\beq
\underbrace{p(x, t+\tau)}
_{P(\XT{x}{t+\tau}\mid\XT{x_0}{t_0})}
=
\int_{-\infty}^{\infty}dx'\;
P(\XT{x}{t+\tau}\mid\XT{x'}{t})
\underbrace{p(x', t)}_{P(\XT{x'}{t}\mid\XT{x_0}{t_0})}
\eeq
Define the jump variable $\Delta$ and kernel $K(\cdot)$
\beq
\Delta = x-x', \quad K_{t+\tau|t}(x|x')=
 P(\XT{x}{t+\tau}\mid\XT{x'}{t})
\eeq
Then
\beq
p( x, t+\tau)
=
\int d\Delta\;
 K(x\mid x-\Delta) 
p(x-\Delta, t) 
\eeq
Define the jump moments $M_n$ by
\beq
M_n(x,\tau)
=
\int d\Delta\;\Delta^n  K(x+\Delta|x)
\eeq
Define the Kramers–Moyal coefficients
\beq
D^{(n)}( x, t)
=
\frac{1}{n!}
\lim_{\tau\to0}
\frac{M_n(x,\tau)}{\tau}.
\eeq


\begin{claim}
\beq
\partial_t p( x, t)
=
\sum_{n=1}^\infty
(-\partial_x)^n
\bigl[
D^{(n)}( x, t)p( x, t)
\bigr].
\eeq
\end{claim}
\noindent{\it proof 1 (low tech):}

\beqa
p( x, t+\tau)
&=&
\int d\Delta\;
\int dx'\; \delta(x'-(x-\Delta))K(x\mid x')p( x', t)
\\
&=&\int dx'\;
\int d\Delta\;
\delta(\Delta-(x-x'))K(x'+\Delta\mid x')p( x', t)
\\&=&
\int dx'
\sum_{n=0}^\infty
\frac{(-\partial_x)^n}{n!}
\left[\delta(x-x')
M_n(x,\tau) 
p( x', t)
\right]
\\
&=&
\sum_{n=0}^\infty
\frac{(-1)^n}{n!}
\partial_x^n
\bigl[M_n(x, \tau)p( x, t)\bigr]
\label{eq-m-zero-on-right}
\eeqa
Note that
\beq
M_0(x,\tau)=\int d\Delta \;
\underbrace{K(x;\Delta, \tau)}_{P( \XT{x} {t+\tau}\mid \XT{x-\Delta}{t})}=1
\eeq
Therefore, in
Eq.(\ref{eq-m-zero-on-right}),
if
we  the $n=0$ term 
from the right to the
left hand side, we get
\beq
p( x, t+\tau)-p( x, t)
=
\sum_{n=1}^\infty
\frac{(-1)^n}{n!}
\partial_x^n
\bigl[M_n(x,\tau)p( x, t)\bigr].
\eeq
If we now divide by $\tau$:
\beq
\frac{p( x, t+\tau)-p( x, t)}{\tau}
=
\sum_{n=1}^\infty
\frac{(-1)^n}{n!}
\partial_x^n
\left[
\frac{M_n(x,\tau)}{\tau}
p( x, t)
\right].
\eeq
Now take $\tau\to 0$.
\qed

If all terms with $n\ge3$ vanish  then
\beq
\partial_t p
=
-\partial_x(D^{(1)}p)
+\partial_x^2(D^{(2)}p) 
\eeq
which is the Fokker–Planck equation.

For an Itô SDE
\beq
d\rvx_t=f(\rvx_t t) dt+L(\rvx_t t) d\rvB_t 
\eeq
one finds
\beq
D^{(1)}=f
\qquad
D^{(2)}=\frac12 L^2 
\eeq
and all higher coefficients vanish.


Brownian diffusions truncate at second order because increments scale as
\beq
\Delta X \sim \sqrt{\tau} 
\eeq
so moments higher than second vanish after division by $\tau$.

\hrule

\noindent{\it proof 2 (high tech):}

the following identity which is formally valid for a
finite translation $\Delta$:
\beq
f(x-\Delta)
=
e^{-\Delta\partial_x}f(x)
,\quad
\left(\av{f|x-\Delta}= \av{f|e^{-\Delta\partial_x}|x}\right)
\eeq
where
\beq
e^{-\Delta\partial_x}
=
\sum_{n=0}^\infty
\frac{(-\Delta)^n}{n!}\partial_x^n
\eeq
In particular, this applies to the 
Dirac delta function:



\beq
\delta(x)=\int_{-\infty}^\infty dk\; \frac{e^{ik}}{2\pi},\quad
\delta(k)=\int_{-\infty}^\infty dx\; \frac{e^{ik}}{2\pi}
\eeq




\beq
\int_{-\infty}^\infty dx\to \quad \sum_x,\quad 
\int_{-\infty}^\infty dk\to \quad \sum_k
\eeq

\beq
\av{k| x}=
\frac{e^{ikx}}{\sqrt{2\pi}},
\quad 
\av{x| k}=
\frac{e^{-ikx}}{\sqrt{2\pi}}
\eeq

\beq
\sum_{x}\ket{x}\bra{x}=1,\quad
\av{x|x'}=\delta(x-x')
\eeq

\beq
\sum_{k}\ket{k}\bra{k}=1,\quad
\av{k|k'}=\delta(k-k')
\eeq






\begin{enumerate}
\item Expanding $\ket{P}$ in $\ket{x}$ basis

\beq
\tilde{P}(k)
=
\int_{-\infty}^\infty dx\; 
\frac{e^{ik}}{\sqrt{2\pi}}P(x),\quad
\left(
\av{k|P}=
\sum_x\av{k| x}\av{x|P}
\right)
\eeq

\item Expanding $\ket{P}$ in $\ket{k}$ basis

\beq
P(x)=
\int_{-\infty}^\infty dk\;
\frac{e^{-ik}}{\sqrt{2\pi}}
\tilde{P}(k)
,\quad
\left(
\av{x|P}=
\sum_k \av{x|k}\av{k|P}
\right)
\eeq

\item Expanding $\ket{P}$ in $\ket{n}$ basis


\beq
\mu_n= \int_{-\infty}^\infty dx\; x^n P(x)
,\quad
\left(\underbrace{\av{n|\ul{\mu}|P}}_{\mu_n}=
\sum_x \underbrace{\av{n|\ul{\mu}|x}}_{x^n}\av{x|P}\right)
\eeq

\beq
\av{k|n}=\frac{(ik)^n}{n!\sqrt{2\pi}}\implies
\av{x|n}= \sum_k\av{x|k}\av{k|n}
=
\int_{-\infty}^\infty dk\; \frac{e^{-ikx}}{\sqrt{2\pi}}\frac{(ik)^n}{n!\sqrt{2\pi}}=\frac{1}{n!}
(-\partial_x)^n\delta(x)
\eeq


\beq
\boxed{
\av{k|n}=\frac{(ik)^n}{n!\sqrt{2\pi}},
\quad
\av{x|n}=
\frac{1}{n!}
(-\partial_x)^n\delta(x)
}
\eeq
\beq
\av{x-x'|n}=\frac{1}{n!}(-\partial_x)^n\delta(x-x')
\eeq

\beq
\int_{-\infty}^\infty dx\; \frac{e^{ikx}}{\sqrt{2\pi}} P(x)=\sum_{n=0}^\infty \frac{(ik)^n}{n!\sqrt{2\pi}}\int_{-\infty}^\infty x^n P(x)=
\sum_{n=0}^\infty \frac{(ik)^n}{n!\sqrt{2\pi}}\mu_n
\eeq




\beq
\sum_x \av{k|x}\av{x|P}=
\sum_{n=0}^\infty
\underbrace{\av{k|n}}_{\frac{(ik)^n}{n!\sqrt{2\pi}}}
\underbrace{\av{n|\ul{\mu}|P}}_{\mu_n}
\eeq


\beq
\boxed{ 
\sum_{n=0}^\infty \ket{n}\bra{n}\ul{\mu}=1}
\eeq

\beq
\av{x|P}=\sum_{n=0}^\infty \av{x|n}\av{n|\ul{\mu}P}=
\sum_{n=0}^\infty \frac{1}{n!}(-\partial_x)^n \delta(x)\mu_n
\eeq


\end{enumerate}


\hrule

Jump
\beq
\Delta = x-x'
\eeq


\beq
\mu(\XT{n}{t}\mid \XT{x'}{t'})= \int_{-\infty}^\infty dx\; (x-x')^n P(
\XT{x}{t}\mid \XT{x'}{t'})
,\quad
\left(\underbrace{\av{n|\ul{\mu}^{x'}P_{t, t'}|x'}}_
{\mu(\XT{n}{t}\mid \XT{x'}{t'})}
=
\sum_x \underbrace{\av{n|\ul{\mu}^{x'}|x}}_{(x-x')^n}
\av{x|P_{t,t'}|x'}\right)
\eeq

\beq
\int_{-\infty}^\infty dx\; \frac{e^{ik(x-x')}}{\sqrt{2\pi}} P(\XT{x}{t}|\XT{x'}{t'})=\sum_{n=0}^\infty \frac{(ik)^n}{n!\sqrt{2\pi}}\int_{-\infty}^\infty (x-x')^n P(\XT{x}{t}|\XT{x'}{t'})=
\sum_{n=0}^\infty \frac{(ik)^n}{n!\sqrt{2\pi}}\mu(\XT{n}{t}| \XT{x'}{t'})
\eeq

\beq
\sum_x\av{k|x-x'}\av{x|P_{t,t'}|x'}=
\sum_{n=0}^\infty\av{k|n}\av{n|\ul{\mu}^{x'}P_{t,t'}|x'}
\eeq



\beq
\boxed{
\sum_{n=0}^\infty \ket{n}\bra{n}\ul{\mu}^{x'}=
\sum_x \ket{x-x'}\bra{x}}
\eeq

\beq
\bra{x-x'}P_{t,t'}\ket{x'}=
\sum_{n=0}^\infty \av{x-x'|n}\av{n|\ul{\mu}^{x'}P_{t,t'}|x'}
=
\sum_{n=0}^\infty
\frac{1}{n!}(-\partial_x)^n\delta(x-x')
\mu(\XT{n}{t}| \XT{x'}{t'})
\eeq


\hrule
\beq
P(\XT{x}{t}|\XT{x_0}{t_0})=
\int dx'\;
P(\XT{x}{t}|\XT{x'}{t-\tau})P(\XT{x'}{t-\tau}|\XT{x_0}{t_0})
,\quad
\left(\av{x|P_{t, t-\tau}|x_0} =
\sum_{x'} 
\av{x|P_{t, t-\tau}|x'}
\av{x'|P_{t-\tau, t_0}|x_0}\right)
\eeq


\beq
P_{t, t_0}=
P_{t, t-\tau}P_{t-\tau, t_0}
\eeq


\beqa
\bra{x-x'}P_{t, t_0}\ket{x_0}&=&
\sum_{n=0}^\infty
\sum_{x'}
\bra{x-x'}P_{t, t-\tau}
\ket{n}\av{n|x'}
\bra{x'}P_{t-\tau, t_0}\ket{x_0}
\\
&=&
\sum_{n=0}^\infty
\sum_{x'}
\frac{1}{n!}(-\partial_x)^n\delta(x-x')
\mu(\XT{n}{t}| \XT{x'}{t-\tau})
\bra{x'}P_{t-\tau, t_0}\ket{x_0}
\\
&=&
\sum_{n=0}^\infty\frac{(-\partial_{x})^n}{n!} 
\left[\mu(\XT{n}{t}| \XT{x}{t-\tau})
\bra{x}P_{t-\tau, t_0}\ket{x_0}\right]
\eeqa

$n=0$ term on right  hand side is just 
$\av{x-x'|P(t-\tau, t_0)|x_0}$ so move it to
left hand side and divide by $\tau$.
Also define

\beq
D^{(n)}(x,t) = \lim_{\tau\to 0}\frac{1}{\tau}\mu(\XT{n}{t}| \XT{x}{t-\tau})
\eeq

\beq
\partial_t P(x,t)=
\sum_{n=1}^\infty
\frac{(-\partial_{x})^n}{n!} 
\left[D^{(n)}(x,t)
P(x,t)\right]
\eeq
\qed
