\item {\bf Maxwell's Relations}


\beq
\begin{array}{ll}
\xymatrix{
db
&
&da
\\
dx\ar[dr]|{a}
\ar[u]|{\partial_x b}
&
&dy\ar[dl]|{b}
\ar[u]|{\partial_y a=\partial_x b}
\\
&dz
&
}\quad\quad
&
\boxed{
\partial_{y}
\underbrace{\partial_{x|y}z}_a
=
\partial_{x}
\underbrace{\partial_{y|x}z}_b
}
\end{array}
\eeq

Example:

\beq F=U-TS, \quad dU=TdS-pdV
\eeq
\beq
dF= dU-TdS-SdT=
-pdV-SdT
\eeq


\beq
\begin{array}{ll}
\xymatrix{
-dp
&
&-dS
\\
dT\ar[dr]|{-S}
\ar[u]|{-\partial_{T|V}p}
&

\ar[u]|{-\partial_{V|T}S}

\boxed{
\partial_{T|V}p=

}
\end{array}
\label{eq-dp-is-ds}
\eeq

\item {\bf Type I (Reciprocal Rule)}
 
\begin{claim}
\beq
\begin{array}{ll}
\xymatrix{
dx\ar[dr]|{a}
\ar[rr]|{b_1}
&&
dy\ar[dl]|{b_2, c_2}
\\
&dz\ar@/_1.5pc/[ur]|{c_1}
&
}\quad\quad
\left\{
\begin{array}{l}
a = \partial_{x|y}z
\\
b_1=\partial_{x|z}y,\quad
b_2=\partial_{y|x} z
\\
c_1=\partial_{z|x}y,\quad
c_2=\partial_{y|x} z
\\
\boxed{
a =- b_1b_2,\quad
c_1c_2=1
}
\end{array}
\right.
\end{array}
\eeq
\end{claim}
\proof

\beq
dz= dx
\underbrace{\partial_{x|y}z}_{a}
+dy
\underbrace{\partial_{y|x}z}_{b_2, c_2}
\eeq
\beq
dy=dx
\underbrace{\partial_{x|z}y}_{b_1}
+
dz
\underbrace{\partial_{z|x}y}_{c_1}
\eeq

\beq
dz=[a +b_1b_2]dx + [c_1c_2]dz
\eeq

\qed


\item {\bf Type II (Cyclic Permutation Rule)}.

This rule
is the same as as
the Type I rule, but with $dz=0$.

\begin{claim}
\beq
\begin{array}{ll}
\xymatrix{
dx\ar[dr]|{a}
\ar[rr]|{b_1}
&&
dy\ar[dl]|{b_2, c_2}
\\
&dz=0\ar@/_1.5pc/[ur]|{c_1}
&
}\quad\quad
\left\{
\begin{array}{l}
a = \partial_{x|y}z
\\
b_1=\partial_{x|z}y,\quad
b_2=\partial_{y|x} z
\\
\boxed{
a =-b_1b_2
}
\end{array}
\right.
\end{array}
\eeq
\end{claim}
\proof See proof of Type I rule.
\qed

Example:

\beq
\begin{array}{ll}
\xymatrix{
dT\ar[dr]|{\alp V}
\ar[rr]|{\partial_{T|V}p}
&&
dp\ar[dl]|{-\kappa_T V}
\\
&dV=0\ar@/_1.5pc/[ur]
&
}\quad\quad
\boxed{\partial_{T|V}p=
\alp/\kappa_T}
\end{array}
\eeq 

\item {\bf Type III Rule (Parallelogram Rule)}


For this rule,
two $dz$ 
must be considered.
We label each of
the 2 $dz$ nodes with a subscript
listing the arrows
entering the node.
The 2 $dz$ 
are numerically equal
(i.e., $d_{xy}z
=
d_{xw}z$ ).

\begin{claim}
\beq
\begin{array}{ll}
\xymatrix@C=1pc{
dy
\ar@/^1pc/[rrrr]|{s_1}
\ar[dr]|{r}
&
&
dx
\ar[ll]
\ar[rr]|{b_1}
\ar[dl]|{\;a'}
\ar[dr]|{a}
&
&
dw
\ar[dl]|{b_2, s_2}
\ar@/_2pc/[llll]
\\
&
d_{yx}z
\ar@{-}[rr]_{=}
&
&
d_{xw}z
}\quad\quad
&
\left\{
\begin{array}{l}
a'=\partial_{x|y}z
\\
a=\partial_{x|w}z
\\
b_1=\partial_{x|y}w
,\quad
b_2=\partial_{w|x}z
\\
r=\partial_{y|x}z
\\
s_1=\partial_{y|x}w,\quad
s_2=\partial_{w|x}z
\\
\boxed{a'=a+b_1 b_2, \quad r=s_1s_2
}
\end{array}
\right.
\end{array}
\eeq
\end{claim}
\proof

\beq
d_{xy}z = d_{xw}z
\implies
dx
\underbrace{\partial_{x|y}z}_{a'}
+ dy
\underbrace{\partial_{y|x}z}_r
=
dx
\underbrace{\partial_{x|w}z}_{a}
+
dw
\underbrace{\partial_{w|x}z}_{b_2,s_2}
\eeq

\beq
dw= dx
\underbrace{\partial_{x|y}w}_{b_1}+
dy
\underbrace{\partial_{y|x}w}_{s_1}
\eeq

\beq
0=
dx\left[
a'
-a
-
b_1
b_2
\right]
+dy\left[
r
-
s_1s_2
\right]
\eeq

\qed
