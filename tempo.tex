\beq
\frac{d\rvx_t}{dt}= a_t + b_t\circ_\alp \dot{\rvB}_t
\eeq

\beq
d\rvx_t= a_t dt + b_t\circ_\alp d\rvB_t
\eeq

\beq
\rvx(T)-\rvx(0)=\int_{0}^T a_t dt  + 
\int_{t=0}^T b_t\circ_\alp d\rvB_t
\eeq

{\bf General Stochastic $\alp$-anticipatory Integral}
For $\alp\in[0, 1]$, 
\beq 
\int_{t=0}^T 
\rvx_t\circ_\alp d\rvB_t
\eqdef 
\lim_{\Delta t\to 0}\sum_i \rvx_{t_i+\alp\Delta t}\Delta_{t_i}^{t_{i+1}}\rvB
\eeq

\begin{itemize}
\item $\alp=0$: {\bf It\^{o} (zero anticipatory) Integral}
\beq
\rvx\circ_0 d\rvB =\rvx\; d\rvB,
\eeq
\item $\alp=1/2$: {\bf Stratonovitch (half anticipatory) integral}
\beq
\rvx\circ_{1/2} d\rvB =\rvx\circ d\rvB
\eeq
\item $\alp=1$: {\bf fully anticipatory integral}
\end{itemize}






\begin{claim}
If

\beq
d\rvx_t= a_t dt + b_t d\rvB_t
\eeq
then

\beq
\int_{t=0}^T
\rvx_t \circ_\alp d\rvB_t
=
\int_{t=0}^T \rvx_t d \rvB_t
+\alp  \int_{t=0}^T  b_t dt
\eeq
\label{cl-ito-conversion-eq}
\end{claim}
\proof

\beq
\rvx_{t_i +\alp \Delta t}
=
\rvx_{t_i} +\alp \Delta_{t_i}^{t_{i+1}}
\rvx
+
\calo
\left([\Delta_{t_i}^{t_{+1} } \rvx]^2\right)
\eeq

\beqa
\sum_i \rvx_{t_i+\alp\Delta t}\Delta_{t_i}^{t_{i+1}}\rvB
&=&
\sum_i  \rvx_{t_i}
\Delta_{t_i}^{t_{i+1}}\rvB
+
\alp\sum_i
\underbrace{\left(\Delta_{t_i}^{t_{i+1}}\rvx\right)}_{b_i \Delta_{t_i}^{t_{i+1}}\rvB}
\left(\Delta_{t_i}^{t_{i+1}}\rvB\right)
\\
&\to&
\int_{t=0}^T \rvx_t d \rvB_t
+\alp  \int_{t=0}^T  b_t dt
\eeqa

\qed

\begin{claim}
\beq
\int_{t=0}^T \rvB_t \;d\rvB_t= \frac{1}{2}(\rvB_T^2 -T)
\eeq
\label{cl-ito-b-db}
\end{claim}
\proof

\beq
2\rvB_t \Delta_t^{t+\Delta t} \rvB=
\underbrace{\rvB_{t+\Delta t}^2 - \rvB_t^2}_{
\Delta_t^{t+\Delta t} (\rvB^2)
} 
- \underbrace{\left[\Delta_t^{t+\Delta  t}\rvB\right]^2}_{\Delta t}
\eeq

\beqa
2\int_{t=0}^T \rvB_t \;d\rvB_t &=&
\lim_{\Delta t\to 0}\sum_i 2\rvB_{t_i}\Delta_{t_i}^{t_i +\Delta t}\rvB
\\
&=&
\rvB_T^2 - \rvB_0^2 -T
\eeqa

\qed

\begin{claim}
\beq
\int_{t=0}^T \rvB_t \circ d\rvB_t= \frac{1}{2}\rvB_T^2 
\eeq
\end{claim}
\proof

From Claims
 \ref{cl-ito-conversion-eq}
and \ref{cl-ito-b-db}, we get

\beq
\int_{t=0}^T
\rvx_t \circ d\rvB_t = \frac{1}{2}(\rvB_T^2-T)+
\underbrace{\alp}_{\frac12}\int_{t=0}^T \underbrace{b_t}_1 dt
\eeq
\qed

\begin{claim}
The It\^{o} integral
is a martigale. The $\alp$-anticipatory
integral for $\alp\neq 0$ is not a martingale.
\end{claim}
\proof

\beq
M_\alp(T)= \int_{t=0}^T
\rvx_t \circ_\alp d\rvB_t
\eeq

\beq
E[
M_0(T)
\mid
\rvB_{[0, T]}
]
=
\frac{1}{2}E[\rvB_T^2 - T\mid \rvB_{[0, T]}]
=
\frac{1}{2}[\rvB_T^2 -T]
=M_0(T)
\eeq

For $\alp\neq 0$,
\beq
E[
M_\alp(T)
\mid
\rvB_{[0, T]}
]\neq M_\alp(T)
\eeq

\qed

It\^o integral is  a Martingale
but it violates the integration
Tables of standard Calculus. The
Stratonovich integral has the opposite
of those 2 qualities. I's not
a Martingale, but it does 
preserve  the integration
Tables of standard Calculus.
