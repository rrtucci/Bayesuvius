\section{FP Equations in Dirac Notation}

When expressed in succinct
Dirac notation,
the derivation  of the forward and
backward FP equations
becomes very clear.

Begin by expressing a PDE in Dirac notation
\beq
\partial_t \phi(x_t)= \Omega_{x}\phi(x_t)
\to \quad 
\partial_t \av{x_t|\phi}= \Omega_{x_t}\av{x_t|\phi}
\eeq
and an expectation value in Dirac notation

\beq
E_{\rvx, t}[\phi(\rvx)]=
\int dx P(x,t)\phi(x)
\to \quad
E_{\rvx, t}[\phi(\rvx)]=\av{P|\phi}
\eeq
Hence

\beq
\av{P|\partial_t |\phi}=
\av{P|\Omega_{\rvx,t}|\phi}
\eeq
Now taking the Hermitian
conjugate of both sides,

\beq
-\av{\phi|\partial_t |P}=
\av{\phi|\Omega^\dagger_{\rvx,t}|P}
\eeq
Setting $\bra{\phi}=\bra{\XT{x}{t}}$ we get
\beq
-\av{\XT{x}{t}|\partial_t |P}=
\av{\XT{x}{t}|\Omega^\dagger_{\rvx,t}|P}
\eeq
Pulling the operators out 
of the bra-ket sandwich

\beq
-\partial_t\av{\XT{x}{t}| P}=
\Omega^\dagger_{x,t}\av{\XT{x}{t}|P}
\eeq
so
\beq
\partial_t\av{\XT{x}{t}| P}=
\calf_{x,t}\av{\XT{x}{t}|P},\quad
\calf_{x,t}=-\Omega^\dagger_{x,t}
\eeq
If we make $P$ a conditional
probability, then
for the forward evolution, we get

\beq
\partial_t
\underbrace{\av{\XT{x}{t}| P|\XT{w}{s}}}_
{P(\XT{x}{t}|\XT{w}{s})}=
\calf_{x,t}\av{\XT{x}{t}|P|\XT{w}{s}},\quad
\calf_{x,t}=-\Omega^\dagger_{x,t}
\eeq
and for the backwards evolution we get

\beq
\partial_t
\underbrace{\av{\XT{y}{u}| P|\XT{x}{t}}}_{
P(\XT{y}{u}|\XT{x}{t})}=
\calb_{x,t}\av{\XT{y}{u}|P|\XT{x}{t}},\quad
\calb_{x,t}=-\Omega_{x,t}
\eeq

Note that the
$P(\XT{x}{t}|\XT{w}{s})$ in the
forward FP equation is a probability density 
in $(x,t)$,
but the $P(\XT{y}{u}|\XT{x}{t})$
in the backwards FP is NOT
a probability density in $(x,t)$.

Transition matrix for
random process $\rvx_t$. Assume $t_0\leq t$

\beq
P_\rvx(\XT{x}{t}|\XT{x_0}{t_0})=
\bra{\XT{x}{t}}P_\rvx\ket{\XT{x_0}{t_0}}
\eeq

Chapman-Kolgomorov
for random process $\rvx_t$

\beq
\sum_{x_b}
P(\XT{x_a}{t_a}|\XT{x_b}{t_b})
P(\XT{x_b}{t_b}|\XT{x_c}{t_c})
=
P(\XT{x_a}{t_a}|\XT{x_c}{t_c})
\eeq

\beq
\bra{x_a} P^\rvx_{t_a|t_b}
\underbrace{\left[\sum_{x_b} \ket{x_b}
\bra{x_b}\right] }_{1} P^\rvx_{t_b|t_c}
\ket{x_c}
=
\bra{x_a} P^\rvx_{t_a|t_c} \ket{x_c}
\eeq

\beq
\boxed{
P^\rvx_{t_a|t_b}
P^\rvx_{t_b|t_c} = 
P^\rvx_{t_a|t_c}}
\eeq


Random process $\rvx_t$ is a martingale if
\beq
E[\rvx_t|\rvx_{t_0}]= \rvx_{t_0} 
\eeq
when $t_0\leq t$.
In particular, applying $\ket{x_0}$ to both sides,
\beq
E[\rvx_t|\rvx_{t_0}=x_0]=x_0
\eeq

\begin{claim}
If $\rvx_t$ is a martingale
and $t_0\leq t$, then

\beq
\boxed {P^\rvx_{t|t_0}= P^\rvx_{t_0|t_0}=1}
\eeq

\end{claim}
\proof
\beqa
E[\rvx_t|\rvx_{t_0}=x_0]
&=&
\sum_{x}x\bra{x}P_{t|t_0}\ket{x_0}
\\
&=&
\rvx\sum_{x}\ket{x}\bra{x} P_{t|t_0}\ket{x_0}
\\
&=& \rvx \underbrace{P_{t|t_0}}_{1}\ket{x_0}
\\
&=&
x_0
\eeqa
\qed

\begin{claim}
Brownian motion is a martingale.
\beq
E[\rvB_t|\rvB_{t_0}=B_0]= B_0
\eeq
\end{claim}
\proof

\beqa
E[\rvB_t|\rvB_{t_0}=B_0]
&=&
\sum_B B \av{B|P|B_0}
\\
&=&
\int_{-\infty}^\infty dB\; B\;
\caln(B; \mu=B_0, \s^2=0)
\\
&=&
B_0
\eeqa

\qed

Let
\beq
\phi(x)=\av{\phi|x}
\eeq

Let $\ket{x}_t$  be a copy of $\ket{x}$ for time $t$.
Likewise, let $\bra{x}_t$  be a copy of $\bra{x}$ for time $t$.
We shall represent a function $\phi(x,t)$
with explicit time dependence by

\beq
\phi(x,t)=\av{\phi|x}_t
\eeq


\begin{claim}
$\phi(\rvx_t,{t})$ is a martingale, for some smooth function $\phi(\cdot)$
and martingale $\rvx_t$.
\beq
E[\phi(\rvx_t,{t})|\rvx_{t_0}=x_0]= \phi(x_0, t_0)
\eeq
\end{claim}
\proof

\beqa
E[\phi(\rvx_t,{t})|\rvx_{t_0}=x_0]
&=&
\bra{\phi}
\sum_{x}\ket{x}_t
\bra{x}_tP_{t|t_0}
\ket{x_0}_{t_0}
\\
&=&
\bra{\phi}P_{t|t_0}\ket{x_0}_{t_0}
\\
&=&
\av{\phi|x_0}_{t_0}
\eeqa
\qed
