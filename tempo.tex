\hrule
Rules for computing 
partial derivative relations in reversible
thermodynamics


\begin{itemize}



\item {\bf Type I (Reciprocal Rule)}

 
\begin{claim}
\beq
\begin{array}{ll}
\xymatrix{
dx\ar[dr]|{a}
\ar[rr]|{b_1}
&&
dy\ar[dl]|{b_2, c_2}
\\
&dz\ar@/_1.5pc/[ur]|{c_1}
&
}\quad\quad
\left\{
\begin{array}{l}
a = \partial_{x|y}z
\\
b_1=\partial_{x|z}y,\quad
b_2=\partial_{y|x} z
\\
c_1=\partial_{z|x}y,\quad
c_2=\partial_{y|x} z
\\
\boxed{
a =- b_1b_2,\quad
c_1c_2=1
}
\end{array}
\right.
\end{array}
\eeq
\end{claim}
\proof

\beq
dz= dx
\underbrace{\partial_{x|y}z}_{a}
+dy
\underbrace{\partial_{y|x}z}_{b_2, c_2}
\eeq
\beq
dy=dx
\underbrace{\partial_{x|z}y}_{b_1}
+
dz
\underbrace{\partial_{z|x}y}_{c_1}
\eeq

\beq
dz=[a +b_1b_2]dx + [c_1c_2]dz
\eeq

\qed


\item {\bf Type II (Cyclic Permutation Rule)}.

This rule
is the same as as
the Type I rule, but with $dz=0$.

\begin{claim}
\beq
\begin{array}{ll}
\xymatrix{
dx\ar[dr]|{a}
\ar[rr]|{b_1}
&&
dy\ar[dl]|{b_2, c_2}
\\
&dz=0\ar@/_1.5pc/[ur]|{c_1}
&
}\quad\quad
\left\{
\begin{array}{l}
a = \partial_{x|y}z
\\
b_1=\partial_{x|z}y,\quad
b_2=\partial_{y|x} z
\\
\boxed{
a =-b_1b_2
}
\end{array}
\right.
\end{array}
\eeq
\end{claim}
\proof See proof of Type I rule.
\qed

Example:

\beq
\begin{array}{ll}
\xymatrix{
dT\ar[dr]|{\alp V}
\ar[rr]|{\partial_{T|V}p}
&&
dp\ar[dl]|{-\kappa_T V}
\\
&dV=0\ar@/_1.5pc/[ur]
&
}\quad\quad
\boxed{\partial_{T|V}p=
\alp/\kappa_T}
\end{array}
\eeq 

\item {\bf Type III Rule (Parallelogram Rule\footnote{I gave it this name. Ref.
%\cite{Fleming-physica-chemistry}
calls it just Type III rule})}


For this rule,
two $dz$ 
must be considered.
We label each of
the 2 $dz$ nodes with a subscript
listing the arrows
entering the node.
The 2 $dz$ 
are numerically equal
(i.e., $d_{xy}z
=
d_{xw}z$ ).

\begin{claim}
\beq
\begin{array}{ll}
\xymatrix@C=1pc{
dy
\ar@/^1pc/[rrrr]|{s_1}
\ar[dr]|{r}
&
&
dx
\ar[ll]
\ar[rr]|{b_1}
\ar[dl]|{\;a'}
\ar[dr]|{a}
&
&
dw
\ar[dl]|{b_2, s_2}
\ar@/_2pc/[llll]
\\
&
d_{yx}z
\ar@{-}[rr]_{=}
&
&
d_{xw}z
}\quad\quad
&
\left\{
\begin{array}{l}
a'=\partial_{x|y}z
\\
a=\partial_{x|w}z
\\
b_1=\partial_{x|y}w
,\quad
b_2=\partial_{w|x}z
\\
r=\partial_{y|x}z
\\
s_1=\partial_{y|x}w,\quad
s_2=\partial_{w|x}z
\\
\boxed{a'=a+b_1 b_2, \quad r=s_1s_2
}
\end{array}
\right.
\end{array}
\eeq
\end{claim}
\proof

\beq
d_{xy}z = d_{xw}z
\implies
dx
\underbrace{\partial_{x|y}z}_{a'}
+ dy
\underbrace{\partial_{y|x}z}_r
=
dx
\underbrace{\partial_{x|w}z}_{a}
+
dw
\underbrace{\partial_{w|x}z}_{b_2,s_2}
\eeq

\beq
dw= dx
\underbrace{\partial_{x|y}w}_{b_1}+
dy
\underbrace{\partial_{y|x}w}_{s_1}
\eeq

\beq
0=
dx\left[
a'
-a
-
b_1
b_2
\right]
+dy\left[
r
-
s_1s_2
\right]
\eeq

\qed

Example 1:

\beq
\begin{array}{ll}
\xymatrix@C=1pc{
dT
\ar@/^1pc/[rrrr]
\ar[dr]|{r}
&
&
dV
\ar[ll]
\ar[rr]|{\partial_{V|T}S}
\ar[dl]|{\partial_{V|T}U}
\ar[dr]|{-p}
&
&
dS
\ar[dl]|{T}
\ar@/_2pc/[llll]
\\
&
d_{TV}U
\ar@{-}[rr]_{=}
&
&
d_{VS}U
}\quad
&
\boxed{
\begin{array}{l}
\partial_{V|T}U
=-p+ T\partial_{V|T}S
\\
 \partial_{T|V}p=
\partial_{V|T}S \quad\text{(Maxwell Rel.)}
\end{array}}
\end{array}
\eeq

Example 2:

\beq
\begin{array}{ll}
\xymatrix@C=1pc{
dT
\ar@/^1pc/[rrrr]
\ar[dr]|{r}
&
&
dp
\ar[ll]
\ar[rr]|{\partial_{p|T}S}
\ar[dl]|{\partial_{p|T}H}
\ar[dr]|{V}
&
&
dS
\ar[dl]|{T}
\ar@/_2pc/[llll]
\\
&
d_{Tp}H
\ar@{-}[rr]_{=}
&
&
d_{pS}H
}\quad
&
\boxed{
\begin{array}{l}
\partial_{p|T}H
=V+ T\partial_{p|T}S
\\
\partial_{p|T}S=
- \partial_{T|p}V
\quad\text{(Maxwell Rel.)}
\end{array}}
\end{array}
\eeq


\end{itemize}
