\section{FP Equations in Dirac Notation}

When expressed in succinct
Dirac notation,
the derivation  of the forward and
backward FP equations
becomes very clear.


\begin{claim}
If
\beq
\partial_t \phi(x_t)= \Omega_{x,t}\phi(x_t)
\eeq
then

\beq
\partial_tP_{t|t_0}= \Omega_{\rvx,t}^\dagger P_{t|t_0}
\eeq
Hence

\beq
\calf_{x,t}=\Omega_{x,t}^\dagger
\eeq

\end{claim}
\proof

We want to express  the expected value of
both sides of this PDE in Dirac notation
\beqa
E[\phi(\rvx_t)|\rvx_0=x_0] &=&
\int dx \;\phi(x)P(\XT{x}{t}|\XT{x_0}{t_0})
\\
&=&\sum_x\av{\phi|x}\av{x|P_{t|t_0}|x_0}_{t_0}
\\
&=&
\av{\phi|P_{t|t_0}|x_0}_{t_0}
\eeqa
Note that the time dependence
in this expected value gets transferred from inside the $\phi()$
to inside the $P()$. This is a crucial observation.

\beqa
E[\partial_t\phi(\rvx_t)|\rvx_0=x_0]
&=&
\int dx \;\phi(x)\partial_tP(\XT{x}{t}|\XT{x_0}{t_0})
\\&=&
\sum_x\av{\phi|x}\av{x|\partial_tP_{t|t_0}|x_0}_{t_0}
\\
&=&
\bra{\phi}
\partial_tP_{t|t_0}\ket{x_0}_{t_0}
\eeqa

\beqa
E[\Omega_{\rvx,t}\phi(\rvx_t)|\rvx_0=x_0]
&=&\sum_x\left[\Omega_{x,t}\av{\phi|x}\right]\bra{x}
P_{t|t_0}\ket{x_0}_{t_0}
\\
&=&\sum_x\av{\phi|x}\Omega^\dagger_{x,t}\bra{x}
P_{t|t_0}\ket{x_0}_{t_0}
\quad \text{(integrate by parts)}
\\
&=&\sum_x\av{\phi|x}\bra{x}
\Omega^\dagger_{\rvx,t}
P_{t|t_0}\ket{x_0}_{t_0}
\\
&=&\bra{\phi}
\Omega^\dagger_{\rvx,t}
P_{t|t_0}\ket{x_0}_{t_0}
\eeqa
\qed
\begin{claim}
If
\beq
\partial_t \phi(x_t)= \Omega_{x,t}\phi(x_t)
\eeq
then

\beq
\partial_tP_{T|t}= \Omega_{\rvx,t} P_{T|t}
\eeq
Hence

\beq
\calb_{x,t}=\Omega_{x,t}
\eeq

\end{claim}
\proof
\qed


Note that the
$P(\XT{x}{t}|\XT{w}{s})$ in the
forward FP equation is a probability density 
in $(x,t)$,
but the $P(\XT{y}{u}|\XT{x}{t})$
in the backwards FP is NOT
a probability density in $(x,t)$.







\beq
\sum_x = \int dx
\eeq

Transition matrix for
random process $\rvx_t$. Assume $t_0\leq t$

\beq
P_\rvx(\XT{x}{t}|\XT{x_0}{t_0})=
\bra{\XT{x}{t}}P_\rvx\ket{\XT{x_0}{t_0}}
\eeq



Random process $\rvx_t$ is a {\bf martingale} if
\beq
E[\rvx_t|\rvx_{t_0}]= \rvx_{t_0} 
\eeq
when $t_0\leq t$.
In particular, applying $\ket{x_0}$ to both sides,
\beq
E[\rvx_t|\rvx_{t_0}=x_0]=x_0
\eeq

\begin{claim}
If $\rvx_t$ is a martingale
and $t_0\leq t$, then

\beq
\boxed {P^\rvx_{t|t_0}= P^\rvx_{t_0|t_0}=1}
\eeq

\end{claim}
\proof
\beqa
E[\rvx_t|\rvx_{t_0}=x_0]
&=&
\sum_{x}x\bra{x}P_{t|t_0}\ket{x_0}
\\
&=&
\rvx\sum_{x}\ket{x}\bra{x} P_{t|t_0}\ket{x_0}
\\
&=& \rvx \underbrace{P_{t|t_0}}_{1}\ket{x_0}
\\
&=&
x_0
\eeqa
\qed

\begin{claim}
Brownian motion is a martingale.
\beq
E[\rvB_t|\rvB_{t_0}=B_0]= B_0
\eeq
\end{claim}
\proof

\beqa
E[\rvB_t|\rvB_{t_0}=B_0]
&=&
\sum_B B \av{B|P_{t|t_0}|B_0}
\\
&=&
\int_{-\infty}^\infty dB\; B\;
\caln(B; \mu=B_0, \s^2=0)
\\
&=&
B_0
\eeqa
The mean of a normal distribution is
its mean.

\qed

Let
\beq
\phi(x)=\av{\phi|x}
\eeq

Let $\ket{x}_t$  be a copy of $\ket{x}$ for time $t$.
Likewise, let $\bra{x}_t$  be a copy of $\bra{x}$ for time $t$.
We shall represent a function $\phi(x,t)$
with explicit time dependence by

\beq
\phi(x,t)=\av{\phi|x}_t
\eeq

Note that $x_t=x(t)$ so $\ket{x_t}\neq \ket{x}_t= \ket{\rvx_t=x}$, strictly speaking.
We won't use $\ket{x_t}$.

\begin{claim}
$\phi(\rvx_t,{t})$ is a martingale, for some smooth function $\phi(\cdot)$
and martingale $\rvx_t$.
\beq
E[\phi(\rvx_t,{t})|\rvx_{t_0}=x_0]= \phi(x_0, t_0)
\eeq
\end{claim}
\proof

\beqa
E[\phi(\rvx_t,{t})|\rvx_{t_0}=x_0]
&=&
\bra{\phi}
\sum_{x}\ket{x}_t
\bra{x}_tP_{t|t_0}
\ket{x_0}_{t_0}
\\
&=&
\bra{\phi}P_{t|t_0}\ket{x_0}_{t_0}
\\
&=&
\av{\phi|x_0}_{t_0}
\eeqa
\qed
