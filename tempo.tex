\section{Laplace Equation (Hyperbolic)}

\subsection{Notation}

1-dim

$\max(x, x')=$ ramp with height $x$ when plotted as a
function of $x'$.

\beq
\partial_x\min(x,x')=
\left\{
\begin{array}{ll}
\partial_x x & \text{if $x<x'$}
\\
\partial_x x' & \text{if $x>x'$}
\end{array}
\right\}
=\indi(x<x')
\eeq

\beq
1=\int_{-\infty}^\infty dx\;\partial_x\indi(x<x')=\int_{-\infty}^\infty dx\; \delta(x-x')
\eeq
so
\beq
 \partial_x\indi(x<x')=\delta(x-x')
\eeq

Define
\beq
\call_{x}= - \partial_x^2,\quad
\call_{x} G(x,x')=\delta(x-x')
\eeq





2-dim

$\vec{r}=(x,y)=x\hat{x} + y\hat{y}$, \quad $\nabla =(\partial_x, \partial_y)$,
 $\nabla^2=\partial_x^2 + \partial_y^2$

\beq
\engdim{\partial_x^2}\engdim{G}=
\engdim{\delta(x)}^d
\implies 
\engdim{G}=L^{2-d}
\eeq

$r=|\vec{x}-\vec{x'}|$
\begin{itemize}

\item $d=1$: $G\sim r$, grows with $r$
\item $d=2$: $G\sim \ln(r)$
\item $d=3$: $G\sim 1/r$, decreases with $r$
\end{itemize}

assume $x>0$
\beq
|x-x'|=
\left\{
\begin{array}{ll}
-(x-x')&\text{if $x'>x$}
\\
x-x'&\text{if $-x<x'<x$}
\\
x-x'&\text{if $x'<-x$}
\end{array}
\right\},\quad
|x+x'|=
\left\{
\begin{array}{ll}
x+x'&\text{if $x'>x$}
\\
x'+x&\text{if $-x<x'<x$}
\\
-(x+x')&\text{if $x'<-x$}
\end{array}
\right\}
\eeq

\beq
\frac{1}{2}|x-x'|
+\frac{1}{2}|x+x'|
=
\underbrace{\left\{
\begin{array}{ll}
x'&\text{if $x'>x$}
\\
x&\text{if $-x<x'<x$}
\\
-x'&\text{if $x'<-x$}
\end{array}
\right\}}_{
=\max(x,x') \text{ if $x'>0$}}
\eeq

\beq
\frac{1}{2}|x-x'|
-\frac{1}{2}|x+x'|
=
\underbrace{\left\{
\begin{array}{ll}
-x&\text{if $x'>x$}
\\
-x'&\text{if $-x<x'<x$}
\\
x&\text{if $x'<-x$}
\end{array}
\right\}}_{
=-\min(x,x') \text{ if $x'>0$}}
\eeq



 

\subsection{General Solution}


\begin{claim}
Suppose
\beq
\call_{x}= -\partial_x^2,\quad
\call_{x} G(x, x')=\delta(x-x')
\eeq
The solution to the equation
\beq
\call_{x} \eta (x) = S(x)\quad
\eeq
for $x>0$
with  Dirichlet, Neumann or Robin boundary conditions satisfies the {\bf general solution}:

\beqa
\ket{\eta}
&=&
\underbrace{G{P_\rvx}\ket{S}}_{\text{Source Term $\cala_S$}}
\\
&&+\underbrace{
G\left[-\ket{x'}\partial_{x'}\bra{x'}
+
\ket{x'}\cev{\partial}_{x'}\bra{x'}\right]_{x'=0}
\ket{\eta}}_{\text{Boundary Conditions (BC) term $\cala_{BC}$}}
\eeqa
or, in non-Dirac notation,

\beqa
\eta(x)
&=&
\int_0^\infty dx'\;G(x,x')S(x')
\\
&&+
G(x,0) \left[-\partial_{x'}
+ \cev{\partial}_{x'}
\right]\eta(0)
\eeqa
\end{claim}
\proof
\qed


\subsection{Point Sources and their images}


Free space Geeen's function

1-dim
\beq
G^+(x, x')= -\;\frac{1}{2}|x-x'| = g_0(x-x')
\eeq
2-dim
\beq
G^+(\vec{r},\vec{r}\;')= -\;\frac{1}{2\pi}\ln |\vec{r}
-\vec{r}\;'| = g_0(\vec{r}-\vec{r}\;')
\eeq
3-dim
\beq
G^+(\vec{r},\vec{r}\;')= -\;\frac{1}{4\pi |\vec{r}
-\vec{r}\;'|} = g_0(\vec{r}-\vec{r}\;')
\eeq


\beq
G^-(x, x')=G(-x, x')
\eeq






\subsection{Dirichlet BC}

\beq
G^D(x, x')= G^+ -G^-
=\min(x, x')
\eeq

$\min(x,x')$ has slope$+1$  for $x'\in [-x, x]$
and is constant for $x'<-x$ and $x'>x$.
This kink at $x'=\pm x$
leads to a jump in the first derivative $\partial_{x'}G$ and to a delta in the 
second derivative $\partial_{x'}^2G$.
Hence $\call G=\delta(x-x') + \delta(x+x')$
Since we are working in the $x'>0$ quadrant,
the mirror image delta function $\delta(x+x')$ can be 
ignored.


\subsection{Neumann BC}
\beq
G^N(x, x')=G^++G^- =
-\max(x, x')
\eeq

$\max(x,x')$ is constant  for $x'\in [-x, x]$
and has slope $\pm 1$ for $x'<-x$ and $x'>x$.
This kink at $x'=\pm x$
leads to a jump in the first derivative $\partial_{x'}G$ and to a delta in the 
second derivative $\partial_{x'}^2G$.
Hence $\call G=\delta(x-x') + \delta(x+x')$
Since we are working in the $x'>0$ quadrant,
the mirror image delta function $\delta(x+x')$ can be 
ignored.


\subsection{Mixed (Robin)}

\beq
G^R= G^+ - G^- +
\underbrace{2k\int_0^\infty 
dX\;e^{-k X} g_0(x+x'+X)}_{J}
\label{eq-robin-superposition-X}
\eeq
where $k= \frac{a_1}{a_0}$.


\begin{claim}
According to  Eq.(\ref{eq-robin-superposition-X}),

\beq 
G^D= G^+ - G^- = \min (x,x')
\eeq
\beq
G^N=-\max(x, x') + \frac{1}{k}
\eeq
\end{claim}
\proof


\beq
\int dx\;xe^{ax}
=
\frac{e^{ax}}{a}\left(x-\frac{1}{a}\right)
\eeq

 $X_1=x+x'>0$

\begin{align}
J&=
-k\int_0^\infty 
dX\;e^{-k X}
|X+X_1|
\\
&=
-k\int_{X_1}^{\infty}
dX\;e^{-k (X-X_1)}X
\\
&=
-e^{k X_1}e^{-k X_1}\left(X_1+\frac{1}{k}\right)
\\
&=-\left(X_1+\frac{1}{k}\right)
\end{align}

\qed










Free space Green's function
\beq
G_0\twoevent{x}{t}{x'}{t'}=
g_0(x-x', t-t)
\eeq

\beq
G_\mu\twoevent{x}{t}{x'}{t'}=
g_\mu(x, x'; t-t) \quad\text{for $\mu = D, N, R$}
\eeq


$\tau=t-t'>0$
\beq
\call[g_0(x-x', \cdot)](s)=
\tilde{g}_0(x-x',s)]=\int_0^\infty d\tau\; e^{-s\tau}g_0(x-x', \tau)
\eeq

\beq
\call[g_\mu(x,x'; \cdot)](s)=
\tilde{g}_\mu(x, x';s)]=\int_0^\infty d\tau\; e^{-s\tau}g_\mu(x-x'; \tau)
\quad\text{for $\mu = D, N, R$}
\eeq

Diffusion
\beq
K= \sqrt{\frac{s}{D}}
\eeq
wave
\beq
K=s 
\eeq

\beq 
K_0= \frac{a_0}{a_1}
\eeq

robin
\beq
[a_0 + a_1\partial_x]\eta(x,t)=0
\eeq




\begin{claim}
\beq
\tilde{g}_R(x, x';s)=
\tilde{g}_0(x-x', s)
+\tilde{R}(s)\tilde{g}_0(x+x', s)
\eeq

\beq
\tilde{g}_0(x-x', s)=
\frac{1}{2DK}
e^{-K|x-x'|}
\eeq

\beq
\tilde{R}(s)=
\frac{K-K_0}{K+K_0}
\eeq

$R(\tau)=$ boundary response kernel
Hence, in th time domain, 
\beq
G^R= g_0(x-x', \tau)
+
\int_0^{\tau}
d\tau'\; R(\tau')
g_0(x+x', \tau -\tau')
\eeq
\end{claim}
\proof
Neumann, $K_0=0$
$\tilde{R}(s)=1$ so 
$R(\tau)=\delta(\tau)$

\beq G^R\rarrow 
g_0(x-x', \tau)
+
g_0(x+x', \tau -\tau')
=G^N
\eeq

Dirichlet, $K\rarrow \infty$, $\tilde{R}\to -1$

\beq G^R\rarrow 
g_0(x-x', \tau)
-
g_0(x+x', \tau -\tau')
=G^D
\eeq

\qed
