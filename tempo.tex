Transition matrix for
random process $\rvx_t$. Assume $t_0\leq t$

\beq
P_\rvx(\XT{x}{t}|\XT{x_0}{t_0})=
\bra{\XT{x}{t}}P_\rvx\ket{\XT{x_0}{t_0}}
\eeq



Random process $\rvx_t$ is a {\bf martingale} if
\beq
E[\rvx_t|\rvx_{t_0}]= \rvx_{t_0} 
\eeq
when $t_0\leq t$.
In particular, applying $\ket{x_0}$ to both sides,
\beq
E[\rvx_t|\rvx_{t_0}=x_0]=x_0
\eeq

\begin{claim}
If $\rvx_t$ is a martingale
and $t_0\leq t$, then

\beq
\boxed {P^\rvx_{t|t_0}= P^\rvx_{t_0|t_0}=1}
\eeq

\end{claim}
\proof
\beqa
E[\rvx_t|\rvx_{t_0}=x_0]
&=&
\sum_{x}x\bra{x}P_{t|t_0}\ket{x_0}
\\
&=&
\rvx\sum_{x}\ket{x}\bra{x} P_{t|t_0}\ket{x_0}
\\
&=& \rvx \underbrace{P_{t|t_0}}_{1}\ket{x_0}
\\
&=&
x_0
\eeqa
\qed

\begin{claim}
Brownian motion is a martingale.
\beq
E[\rvB_t|\rvB_{t_0}=B_0]= B_0
\eeq
\end{claim}
\proof

\beqa
E[\rvB_t|\rvB_{t_0}=B_0]
&=&
\sum_B B \av{B|P_{t|t_0}|B_0}
\\
&=&
\int_{-\infty}^\infty dB\; B\;
\caln(B; \mu=B_0, \s^2=0)
\\
&=&
B_0
\eeqa
The mean of a normal distribution is
its mean.

\qed

Let
\beq
\phi(x)=\av{\phi|x}
\eeq

Let $\ket{x}_t$  be a copy of $\ket{x}$ for time $t$.
Likewise, let $\bra{x}_t$  be a copy of $\bra{x}$ for time $t$.
We shall represent a function $\phi(x,t)$
with explicit time dependence by

\beq
\phi(x,t)=\av{\phi|x}_t
\eeq

Note that $x_t=x(t)$ so $\ket{x_t}\neq \ket{x}_t= \ket{\rvx_t=x}$, strictly speaking.
We won't use $\ket{x_t}$.

\begin{claim}
$\phi(\rvx_t,{t})$ is a martingale, for some smooth function $\phi(\cdot)$
and martingale $\rvx_t$.
\beq
E[\phi(\rvx_t,{t})|\rvx_{t_0}=x_0]= \phi(x_0, t_0)
\eeq
\end{claim}
\proof

\beqa
E[\phi(\rvx_t,{t})|\rvx_{t_0}=x_0]
&=&
\bra{\phi}
\sum_{x}\ket{x}_t
\bra{x}_tP_{t|t_0}
\ket{x_0}_{t_0}
\\
&=&
\bra{\phi}P_{t|t_0}\ket{x_0}_{t_0}
\\
&=&
\av{\phi|x_0}_{t_0}
\eeqa
\qed
