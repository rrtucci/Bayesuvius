\begin{claim}

If $h(\rvx_t, t)$
is a martingale, then
\beq
\frac{h(\rvx_t, t)}
{h(\rvx_0, t_0)}=
\exp\left(
-
\frac12 \int_0^t d\tau\;
\theta_\tau^2
+
\int_0^t \theta_\tau d\rvB_\tau
\right)
\eeq
where
\beq
\theta_t=
L(\rvx_t, t)
\partial_x \ln h(\rvx_t, t)
\eeq
\end{claim}
\proof

Apply It\^o's Lemma to   $\ln h(\rvx_t, t)$
($h(\rvx_t, t)$ is a martingale by assumption, but $\ln h$
is a semi-martingale)

\beq
d\ln h=
(\partial_t \ln h +
\Omega_x \ln h)dt +
L(\partial_x\ln h)d\rvB_t
\eeq
where
\beq
\Omega_x = 
f(x,t)\partial_x
+\frac12 L^2(x,t)\partial_x^2
\eeq

\beq
\partial_x \ln h=
\frac{\partial_x h}{h}
\eeq

\beqa
\partial_x^2 \ln h
&=&
\partial_x
\frac{\partial_x h}{h}
\\
&=&
\frac{-(\partial_x h)^2}{h^2}+
\frac{\partial_x^2h}{h}
\\
&=&
-(\partial_x \ln h)^2 + \frac{1}{h}
\partial_x^2 h
\eeqa

\beqa
\Omega_x \ln h&=&
f\left[\frac{\partial_x h}{h}\right]
+\frac12 L^2
\left[
-(\partial_x \ln h)^2 + \frac{1}{h}
\partial_x^2 h
\right]
\\
&=&
\frac{\Omega_x h}{h}
-
\frac{L^2}{2} 
(\partial_x \ln h)^2
\eeqa

$h(\rvx_t, t)$
is a martingale so
 \beq
\partial_t h +
\Omega_x h=0
\eeq
Hence
\beqa
d\ln h&=&
\left(-\frac{L^2}{2}
(\partial_x \ln h)^2
\right)dt
+ L(\partial_x \ln h)d\rvB_t
\\
&=&
-\;\frac12 \theta^2
dt + \theta d\rvB_t
\eeqa
Now integrate 
both sides

\beq
\ln\left[
\frac{h(\rvx_t, t)}
{h(\rvx_0, 0)}
\right]=
\int_0^t
\left[
-\;\frac12 \theta^2
dt + \theta d\rvB_t
\right]
\eeq
Finally, exponentiate both sides.
\qed
