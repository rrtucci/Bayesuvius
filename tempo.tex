Define a {\bf semi-martingale}
as a random process that satisfies the 
{\bf Stochastic Differential Equation} (SDE)\footnote{The 
acronym SDE complements the acronyms ODE (Ordinary Differential Equation)
and PDE (Partial Differential Equation)}
\beq
\frac{d\rvx_t}{dt}= a_t + b_t\circ_\alp \dot{\rvB}_t
\eeq
where $a_t$ and $b_t$ are reasonably  smooth
function of $t$ and $\rvx_t$. 
The definition of the product $\circ_\alp$ of $b_t$ and
the white noise $\dot{\rvB}_t$, requires 
some finesse and will be given below.
Hence
a semi-martingale is the 
sum of a smooth function and a zig-zaggy function.
An alternative expressions for this SDE is

\beq
d\rvx_t= a_t dt + b_t\circ_\alp d\rvB_t
\eeq
This SDE is solved at least formally, once 
we define a stochastic integral, by

\beq
\rvx(T)-\rvx(0)=\int_{0}^T a_t dt  + 
\int_{t=0}^T b_t\circ_\alp d\rvB_t
\eeq

For any $\alp\in[0, 1]$, define the {\bf General Stochastic $\alp$-anticipatory Integral} of the  semi-martingale $\rvx_t$ by

\beq 
\int_{t=0}^T 
\rvx_t\circ_\alp d\rvB_t
\eqdef 
\lim_{\Delta t\to 0}\sum_i \rvx_{t_i+\alp\Delta t}\Delta_{t_i}^{t_{i+1}}\rvB
\eeq

\begin{itemize}
\item $\alp=0$: {\bf It\^{o} (zero anticipatory) Integral}

In this case we will omit the $\circ_\alp$
\beq
\rvx\circ_0 d\rvB =\rvx\; d\rvB,
\eeq
\item $\alp=1/2$: {\bf Stratonovitch (half anticipatory) integral}

In this case we will omit the $\alp$
\beq
\rvx\circ_{1/2} d\rvB =\rvx\circ d\rvB
\eeq
\item $\alp=1$: {\bf fully anticipatory integral}
\end{itemize}


Define the {\bf quadratic variation of $\rvx_t$ and $\rvy_t$}
by
\beq
\av{\rvx_t, \rvy_t}_0^T=
\lim_{\Delta t\to 0}\sum_i\left[\Delta_{t_i}^{t_i+\Delta t}\rvx\right]
\left[\Delta_{t_i}^{t_i+\Delta t}\rvy\right]
\eeq


\begin{claim}
If

\beq
d\rvx_t= a_t dt + b_t d\rvB_t
\eeq
then

\beq
\int_{t=0}^T
\rvx_t \circ_\alp d\rvB_t
=
\int_{t=0}^T \rvx_t d \rvB_t
+\alp\;  \underbrace{\int_{t=0}^T  b_t dt}_{\av{\rvx_t, \rvB_t}_0^T}
\eeq
\label{cl-ito-conversion-eq}
\end{claim}
\proof

\beq
\rvx_{t_i +\alp \Delta t}
=
\rvx_{t_i} +\alp \Delta_{t_i}^{t_{i+1}}
\rvx
+
\calo
\left([\Delta_{t_i}^{t_{+1} } \rvx]^2\right)
\eeq

\beqa
\sum_i \rvx_{t_i+\alp\Delta t}\Delta_{t_i}^{t_{i+1}}\rvB
&=&
\sum_i  \rvx_{t_i}
\Delta_{t_i}^{t_{i+1}}\rvB
+
\alp\sum_i
\underbrace{\left(\Delta_{t_i}^{t_{i+1}}\rvx\right)}_{b_i \Delta_{t_i}^{t_{i+1}}\rvB}
\left(\Delta_{t_i}^{t_{i+1}}\rvB\right)
\\
&\to&
\int_{t=0}^T \rvx_t d \rvB_t
+\alp  \int_{t=0}^T  b_t dt
\eeqa

\beqa
\av{\rvx_t, \rvB_t}_0^T&=&
\lim_{\Delta t\to 0}\sum_i
\left[\Delta_{t_i}^{t_i+\Delta t}\rvx\right]
\left[\Delta_{t_i}^{t_i+\Delta t}\rvB\right]
\\
&=&
\int_0^T dt\; b_t
\eeqa
\qed

\begin{claim}
\beq
\int_{t=0}^T \rvB_t \;d\rvB_t= \frac{1}{2}(\rvB_T^2 -T)
\eeq
\label{cl-ito-b-db}
\end{claim}
\proof

{stochastic-diff-eqns/ito-integral-square.png}
\caption{Geometry involved in calculation of $\int \rvB_t d\rvB_t$.}

From Fig.\ref{fig-ito-integral-square}
\beq
2\rvB_t \Delta_t^{t+\Delta t} \rvB=
\underbrace{\rvB_{t+\Delta t}^2 - \rvB_t^2}_{
\Delta_t^{t+\Delta t} (\rvB^2)
} 
- \underbrace{\left[\Delta_t^{t+\Delta  t}\rvB\right]^2}_{\Delta t}
\eeq

\beqa
2\int_{t=0}^T \rvB_t \;d\rvB_t &=&
\lim_{\Delta t\to 0}\sum_i 2\rvB_{t_i}\Delta_{t_i}^{t_i +\Delta t}\rvB
\\
&=&
\rvB_T^2 - \rvB_0^2 -T
\eeqa

\qed

\begin{claim}
\beq
\int_{t=0}^T \rvB_t \circ d\rvB_t= \frac{1}{2}\rvB_T^2 
\eeq
\end{claim}
\proof

From Claims
 \ref{cl-ito-conversion-eq}
and \ref{cl-ito-b-db}, we get

\beq
\int_{t=0}^T
\rvx_t \circ d\rvB_t = \frac{1}{2}(\rvB_T^2-T)+
\underbrace{\alp}_{\frac12}\int_{t=0}^T \underbrace{b_t}_1 dt
\eeq
\qed

\begin{claim}
The It\^{o} integral
is a martigale. The $\alp$-anticipatory
integral for $\alp\neq 0$ is not a martingale.
\end{claim}
\proof

\beq
M_\alp(T)= \int_{t=0}^T
\rvx_t \circ_\alp d\rvB_t
\eeq

\beq
E[
M_0(T)
\mid
\rvB_{[0, T]}
]
=
\frac{1}{2}E[\rvB_T^2 - T\mid \rvB_{[0, T]}]
=
\frac{1}{2}[\rvB_T^2 -T]
=M_0(T)
\eeq

For $\alp\neq 0$,
\beq
E[
M_\alp(T)
\mid
\rvB_{[0, T]}
]\neq M_\alp(T)
\eeq

\qed

The It\^o integral is  a martingale
but it violates the integration
rules of standard Calculus. The
Stratonovich integral does the opposite. It's not
a Martingale, but it does 
preserve  the integration
rules of standard Calculus.

Martingale random processes respect Causality; i.e., 
their current expectation value does not
depend on the future. The
$\alp$ in a stochastic integral (SI)
measures how much the SI
violates Causality. SIs with
$\alp=0$ (It\^{o}) are the only truly causal ones.
Microscopic Physics should
of course obey Causality so
observable
quantities
should be 
expressible using Causal SI's only. 
However, there are
many situations
in which non-causal SI with fractional $\alp$
are useful: for example,
if we want to impose temporal
boundary conditions, or if we do some
temporal averaging that
includes future values.
