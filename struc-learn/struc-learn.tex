{\bf linear regression}

\beq
\rvy = \beta_0 +\sum_{j=1}^N\beta_j \rvx_j
\eeq

For $k=1, \ldots, N$,
\beq
\av{\rvx_k, \rvy}
=
\sum_{j=1}^N\beta_j\av{\rvx_k, \rvx_j}
\eeq

$\rvx=(\rvx_1, \ldots, \rvx_N)^T$
\beq
\av{\rvx, \rvy}=
\av{\rvx, \rvx^T}\beta
\eeq

\beq
\beta=
\av{\rvx, \rvx^T}^{-1}\av{ \rvx, \rvy}
\eeq

cat(x;\pi.)
=\pi_x
=\prod_k\pi_k^{\indi(k=x)}
\eeq


\beq
Dir(\pi.;\alp.)=
\indi(\pi_+=1)
\Gamma(\alp_+)
\prod_k
\frac{\pi_k^{\alp_k-1}}

\eeq


\beq
cat(x;\pi.)Dir(\pi.;\alp.)
=
\caln(!\pi.)Dir(\pi.;\alp.')
\eeq

\beq
\alp'_k = \alp_k + \indi(x=k)
\eeq

\beq
P(x|\pi.)=
cat(x;\pi.)
\eeq

\beq
P(\pi.)=Dir(\pi.;\alp.)
\eeq

\beq
P(x|\pi.)P(\pi.)=
\caln(!\pi.)
P(\pi.|x)
\eeq

\beqa
P(\pi.|x)
&=&\caln(!\pi.)
cat(x; \pi.) Dir(\pi.;\alp.)
\\
&=&
Dir(\pi.;\alp'.)
\eeqa
