\chapter{Green Functions for PDE}
\label{ch-green-fun-pde}

\section{Introduction}
Let
\beq
\call = \partial_t - D \partial^2_x
\eeq
The Diffusion Equation (DE) in one space dimension $x$,
 with a source $S(x,t)$, is
\beq
\call \eta(x,t) = S(x,t)
\eeq
In this chapter will illustrate the use of Green
functions by solving the DE partial
differential equation (PDE)
for all $x>0$ and $t>0$, with  
given initial
conditions at $t=0$
and given boundary conditions (BC) of 3 possible types
at $x=0$.

\section{Boundary Condition Types}
Types of boundary conditions (BC) that we will consider:

\begin{itemize}
\item {\bf Dirichlet BC}

\beq \eta(0, t)= D_\rvt(t)
\eeq

\item
{\bf von Neumann BC}
\beq
\partial_x\eta(0,t)= N_\rvt(t)
\eeq
For example, if there were an impermeable wall at $x=0$,

\beq
\partial_x\eta(0,t)=0
\eeq

\item {\bf Mixed (Robin) BC}

\beq
[a \partial_x\eta + b \eta](0, t)=R_\rvt(t)
\eeq
\end{itemize}
If the boundary condition depends on $\eta(x,t)$
only, then it is called a {\bf homogeneous BC} (i.e., $D_\rvt=N_\rvt=R_\rvt==0$)  .
Otherwise, it is said to be an {\bf inhomogeneous BC}






\section{Notation},
I will often use specialized 
notation in this chapter to
express with more clarity 
the basic Green's functions equations and their
derivations. This section
will describe such notation.

An event
 will be presented 
either as a row vector or a column vector:
\beq
(x,t)\leftrightarrow \oneevent{x}{t}
\eeq
Functions $f()$ of one event, and functions $G()$
of two events
will be expressed with their events either as row or column vectors
\beq
f(x,t) \leftrightarrow f\oneevent{x}{t}
\eeq
\beq
G(x,t;x',t')\leftrightarrow G\twoevent{x}{t}{x'}{t'}
\eeq

Another notation that we will often use for 
expressing functions of one or two
events, is Dirac bra-ket notation.
We describe that notation next.

Define
\beq
\call_{x,t}= \partial_t -D \partial_x^2,\quad
\call_{x,t} G\twoevent{x}{t}{x'}{t'}=\stackt{\delta(x-x')}{\delta(t-t')}
\eeq


\beq
\call^\dagger_{x,t}= -{\partial}_t -D {\partial}_x^2,
\quad \call^\dagger_{x,t}G^\dagger\twoevent{x'}{t'}{x}{t}=\stackt{\delta(x-x')}{\delta(t-t')}
\eeq




\beq
P_\rvx = \int_0^\infty dx'\; \ket{\rvx=x'}\bra{\rvx=x'}
\eeq


\beq
P_\rvt = \int_0^t dt'\; \ket{\rvt=t'}\bra{\rvt=t'}
\eeq

\beq
P_0 = \ket{0}\bra{0}
\eeq

\beq
\partial_{\rvx} =
\int_0^\infty dx'\; \ket{\rvx=x'}\partial_{x'}\bra{\rvx=x'}
\eeq

\beq
\partial_{\rvt} =
\int_0^\infty dt\; \ket{\rvt=t'}\partial_{t'}\bra{\rvt=t'}
\eeq

\beq
\rv{\call}= \partial_\rvt-D\partial^2_\rvx
\eeq

\beq
f(x)\cev{\partial_x} a(x)= a(x) \partial_x f(x)
\eeq
We will also use the following suggestive notation
for dual tensor product spaces



\beq \stackt{A}{B}= A\otimes B\eeq

\beq \kett{x}{t}=\ket{x}\otimes\ket{t},
\quad 
\brat{x}{t}=\bra{x}\otimes\bra{t}
\eeq


\section{Master Equation}
\begin{claim}
The solution to the equation
\beq
\call \eta (x,t) = S(x,t)\quad
\eeq
for $x>0$ and $t>0$
with initial conditions
\beq
\eta(x,0)= \eta_\rvx(x)
\eeq
and Dirichlet, von Neumann or Robin boundary conditions satisfies the {\bf master equation}:

\beqa
\ket{\eta}
&=&
\underbrace{G\stackt{P_\rvx}{P_\rvt}\ket{S}}_{\text{Source Term}}
\\
&&\underbrace{+G\stackt{P_\rvx}{P_0}\kett{\eta_\rvx}{0}}_{\text
{Initial Conditions (IC) term}}
\\
&&\underbrace{-D
G\stackt{\left[-\ket{x'}\partial_{x'}\bra{x'}
+
\ket{x'}\cev{\partial}_{x'}\bra{x'}\right]_{x'=0}}{P_\rvt}
\ket{\eta}}_{\text{Boundary Conditions (BC) term}}
\eeqa
\end{claim}
\proof

Define $I$, $I_\rvx$ and $I_\rvt$  by
\beq
I = \brat{x}{t}G\rv{\call}
\ket{\eta}-
\brat{x}{t}\rv{\call^\dagger}G^\dagger
\ket{\eta}
\eeq

\beq
I_\rvt =
\brat{x}{t}G
\partial_\rvt
\ket{\eta}-
\brat{x}{t}
(-{\partial}_\rvt)G^\dagger
\ket{\eta}
\eeq

\beq
I_\rvx =
\brat{x}{t}G\partial_\rvx^2
\ket{\eta}-
\brat{x}{t}
\partial_\rvx^2G^\dagger
\ket{\eta}
\eeq
Hence
\beq
\boxed
{I = I_\rvt -D I_\rvx}
\eeq
\hrule
Note that we can simplify the
expression for $I$ as follows:



\beq
\rv{\call}\ket{\eta}=\ket{S},
\quad
\rv{\call^\dagger}G^\dagger=1
\eeq

\beq
\boxed
{I = \brat{x}{t}G\ket{S}-\brat{x}{t}\ket{\eta}
}
\eeq
\hrule
Note that we can simplify the
expression for $I_\rvt$ as follows:


\beqa
I_\rvt &=&
\int_0^t dt'
\brat{x}{t}G
\stackt{1}{\ket{t'}\partial_{t'}\bra{t'}}
\ket{\eta}
+
\brat{x}{t}G
\stackt{1}{\ket{t'}\cev{\partial}_{t'}\bra{t'}}
\ket{\eta}
\eeqa
where we used

\beq
 \bra{t'}G^\dagger\ket{t}=\bra{t}G\ket{t'}
\implies
\partial_{t'} \bra{t'}G^\dagger\ket{t}
=
\bra{t}G\ket{t'}\cev{\partial}_{t'}
\eeq

\beq
\partial_{t'} (\ket{t'}\bra{t'})=
\ket{t'}\partial_{t'}\bra{t'}
+\ket{t'}\cev{\partial}_{t'}\bra{t'}
\eeq


\beqa
I_\rvt &=&
\int_0^t dt'
\brat{x}{t}G
\stackt{1}{\partial_{t'} (\ket{t'}\bra{t'})}
\ket{\eta}
\eeqa

\beq
\int_0^t
dt'\;\partial_{t'} (\ket{t'}\bra{t'})
=
\ket{t}\bra{t}-\ket{0}\bra{0}
\eeq


\beq
\boxed{I_\rvt =
\brat{x}{t}G
\stackt{P_\rvx}{\ket{t}\bra{t}-\ket{0}\bra{0}}
\ket{\eta}}
\eeq
\hrule
Note that we can simplify the
expression for $I_\rvx$ as follows:

\beq
I_\rvx =
\int_0^\infty dx'
\brat{x}{t}G
\stackt{\ket{x'}\partial_{x'}^2\bra{x'}}{1}
\ket{\eta}
-
\brat{x}{t}
G\stackt{\ket{x'}\cev{\partial}_{x'}^2\bra{x'}}{1}
\ket{\eta}
\eeq

\beq
\ket{x'}\partial_{x'}^2\bra{x'}
-\ket{x'}\cev{\partial}_{x'}^2\bra{x'}=
\partial_{x'}\left(\ket{x'}\partial_{x'}\bra{x'}\right)
-\left(\ket{x'}\cev{\partial}_{x'}\bra{x'}\right)\cev{\partial}_{x'}
\eeq

\beq
I_\rvx=
\int_0^\infty dx'
\brat{x}{t}G
\stackt{\partial_{x'}(\ket{x'}\partial_{x'}\bra{x'})
-\left(\ket{x'}\cev{\partial}_{x'}\bra{x'}\right)\cev{\partial}_{x'}}{1}
\ket{\eta}
\eeq

\begin{align}
\int_0^\infty dx'\left[
\partial_{x'}(\ket{x'}\partial_{x'}\bra{x'})
-\left(\ket{x'}\cev{\partial}_{x'}\bra{x'}\right)\cev{\partial}_{x'}\right]
&=
\left[\ket{x'}\partial_{x'}\bra{x'}
-
\ket{x'}\cev{\partial}_{x'}\bra{x'}\right]_0^\infty
\\
&=
-\left[\ket{x'}\partial_{x'}\bra{x'}
-
\ket{x'}\cev{\partial}_{x'}\bra{x'}\right]_{x'=0}
\end{align}

\beq
\boxed{I_\rvx=
\brat{x}{t}G
\stackt{\left[-\ket{x'}\partial_{x'}\bra{x'}
+
\ket{x'}\cev{\partial}_{x'}\bra{x'}\right]_{x'=0}}{P_\rvt}
\ket{\eta}}
\eeq
or, in non-Dirac notation,

\beq
I_\rvx =
\int_0^tdt'\;G\twoevent{x}{t}{0}{t'} \left[-\partial_{x'}
+ \cev{\partial}_{x'}
\right]\eta\oneevent{0}{t'}
\eeq

\qed

The BC term 
of the Master Equation 
must be specialized for
each of the 3 possible
types of BC. We show how to do  that below.

All Green's functions must satisfy
the DE with delta function sources

\beq
\call_{x,t} G\twoevent{x}{t}{x'}{t'}=\delta(x-x')\delta(t-t')
\eeq
In addition, they must satisfy BC
that match the BC satisfied by the
solution $\eta(x,t)$.
Below, we specify what BC the Green's function must satisfy.

Let $P(x;\s^2)$
represent the Normal distribution:
\beq
P(x;\s^2)=\frac{1}{\sqrt{2\pi\s^2}}
\exp\left(\frac{-x^2}{2\s^2}\right)
\eeq

The $G^\pm$  Green's function
will be used as a building blocks
to construct the Green's functions
appropriate to the various BC.

\beqa
G^\pm\twoevent{x}{t}{x'}{t'}
&=&
\frac{1}{\sqrt{4\pi D(t-t')}}
\exp\left[-\;\frac{(x\pm x')^2}{4D(t-t')}\right]
\\
&=& P(x\pm x'; \kappa(t-t'))
\label{eq-def-green-pm}
\eeqa
where $\kappa =2D$.

$G^\pm$ are not  delta functions, but in 
the limit $\kappa \rarrow 0$, and $t\approx t'$,
they can be replaced by delta functions
\beq
G^\pm\twoevent{x}{t}{x'}{t'}\rarrow \delta(x'-x)
\eeq

A consequence of the Master Equation is that

\beq
G\twoevent{0}{t}{0}{t'}=0\implies\quad  G\twoevent{0}{t}{0}{t'}\cev{\partial}_{x'} 
\propto \delta(t-t')
\eeq
and 
\beq
G\twoevent{0}{t}{0}{t'}\cev{\partial}_{x'}=0
\implies\quad
G\twoevent{0}{t}{0}{t'} \propto \delta(t-t')
\eeq
This Green's function duality,
together the final form for 
the BC term of the Master Equation, implies that
the Green's Function
must satisfy the homogeneous
version of the inhomogeneous BC
satisfied by $\eta(x,t)$


\section{Dirichlet BC}

\beq
\eta(0, t)= D_\rvt(t)
\eeq

Choose Green's function $G_D$ so that
it satisfies the DE with delta function source and obeys
the same BC, but homogeneous
\beq
G_D\twoevent{0}{t}{x'}{t'}=0
\eeq

\beqa
\text{BC term}&=&
-D\brat{x}{t}G_D
\stackt{\left[
\ket{x'}\cev{\partial}_{x'}\bra{x'}\right]_{x'=0}}{P_\rvt}
\ket{\eta}
\\
&=&
-D\int_0^t dt'\;
G_D\twoevent{x}{t}{0}{t'}\cev{\partial}_{x'}
D_\rvt(t')
\eeqa

The $G^\pm$ Green's functions
are defined  by Eq.(\ref{eq-def-green-pm}).
The Green's  function 
for the Dirichlet BC is

\beq
G_D= G^+ - G^-
\eeq




At the boundary
\beq
\left.
\partial_{x'} G_D
(x,t;x',t')
\right|_{x'=0}
=
\frac{x}{\sqrt{2\kappa (t-t')^3}}
\exp\left(
\frac{-x^2}{2\kappa(t-t')}
\right)
\eeq
This $G_D$ was found by the method of images: two
free space solutions $G^\pm$ to $\ul{\call}G^\pm=1$
were superposed so as  to satisfy
$G_D=0$ at the boundary $x=0$.
\hrule
At first glance, this result appears to imply
that

\beq
\lim_{x\rarrow 0}(-D)
\int_0^{t} dt'\;
[\partial_{x'} G_D(x,t;x',t')]_{x'=0}
=\delta(t-t')
\eeq
This is indeed the case. Let's prove
explicitly for the Green's function $G_D=G^--G^+$

Suppose

\beq
G^-(x,t;x',t')= P(x-x'; \kappa (t-t'))
\eeq
where $D=\kappa/2$.
Define

\beq
y=\frac{x}{\sqrt{\kappa(t-t')}}=\frac{x}{\ell}
\eeq
Then
\beqa
-D[\partial_{x'} G^-(x,t;x',t')]_{x'=0}&=&
\left(\frac{-\kappa}{2}\right)
\left(\frac{-x}{\ell^2}\right)\frac{1}{\ell\sqrt{2\pi}}\exp\left(
\frac{-x^2}{2\ell^2}\right)
\\
&=&
\frac{x}{2(t-t')}\frac{1}{\ell\sqrt{2\pi}}\exp\left(
\frac{-y^2}{2}\right)
\eeqa

\beq
\ln y = -\frac{1}{2}\ln (t-t')+\cdots\implies \frac{dy}{y}=
-\;\frac{d(t-t')}{2(t-t')}=
\;\frac{dt'}{2(t-t')}
\eeq
Thus,

\beq
dt'\;\frac{x}{2(t-t')}= dy\;\frac{x}{y}=
dy\sqrt{\kappa(t-t')}=dy\ell
\eeq
Hence
\beq
-D\int_0^t dt'[\partial_{x'} G^-(x,t;x',t')]_{x'=0}=
\int_{\frac{x}{\sqrt{\kappa t}}}^\infty
dy \frac{1}{\sqrt{2\pi}}\exp\left(
\frac{-y^2}{2}\right)=\frac{1}{2}
\eeq


It's illuminating to prove this with delta
functions and their derivatives. To do so, assume
\beq
 G^-(x,t;x',t') =\frac{1}{\ell} \delta\left(\frac{x-x'}{\ell}\right),
\eeq
Then

\begin{align}
-D\int_0^t dt'[\partial_{x'} G^-(x,t;x',t')]_{x'=0}
&=
-D\int_0^t \frac{dt'}{\ell^2} \delta'\left(\frac{x}{\ell}\right)
\\
&=
-D\int_0^t\frac{dt'}{\ell^2}\frac{1}{\eps}\left[
\delta\left(\frac{x}{\ell}+\frac{\eps}{2}\right)-
\delta\left(\frac{x}{\ell}-\frac{\eps}{2}\right)
\right]
\\
&=
-
\int_{\frac{x}{\sqrt{\kappa t}}}^\infty \frac{dy}{y}\frac{1}{\eps}\left[
\delta\left(y+\frac{\eps}{2}\right)-
\delta\left(y-\frac{\eps}{2}\right)
\right]
\\
&= \int_{0}^\infty
dy\;\delta\left(y\right)= \frac{1}{2}
\end{align}
Hence, the derivative
of a delta function equals the difference of
2 delta functions. We integrate over half of
one of the two delta functions.


\section{von Neumann BC}
\beq
\partial_x\eta(0, t)= N_\rvt(t)
\eeq

Choose Green's function $G_N$ so that
it satisfies the DE with delta function source and obeys
the same BC, but homogeneous
\beq
G_N\twoevent{x}{t}{0}{t'}\cev{\partial}_{x'}=0
\eeq

\beqa
\text{BC term}&=&
D\brat{x}{t}G_N
\stackt{\left[\ket{x'}\partial_{x'}\bra{x'}
\right]_{x'=0}}{P_\rvt}
\ket{\eta}
\\
&=&
D\int_0^t dt'\;
G_N\twoevent{x}{t}{0}{t'}
N_\rvt(t')
\eeqa

The $G^\pm$ Green's functions
are defined  by Eq.(\ref{eq-def-green-pm}).
The Green's  function 
for the von Neumann BC is

\beq
G_N= G^+ + G^-
\eeq
Note that for the Dirichlet BC problem, $G^+$ and $G^-$
are subtracted instead if added.





\section{Mixed (Robin) BC}
