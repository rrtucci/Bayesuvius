\chapter{Green Functions for PDE}
\label{ch-green-fun-pde}

\section{Introduction}
Let
\beq
\call = \partial_t - D \partial^2_x
\eeq
The Diffusion Equation (DE) in one space dimension $x$,
 with a source $S(x,t)$, is
\beq
\call \eta(x,t) = S(x,t)
\eeq
In this chapter will illustrate the use of Green
functions by solving the DE partial
differential equation (PDE)
for all $x>0$ and $t>0$, with  
given initial
conditions at $t=0$
and given boundary conditions (BC) of 3 possible types
at $x=0$.

\section{Boundary Condition Types}
Types of boundary conditions (BC) that we will consider:

\begin{itemize}
\item {\bf Dirichlet BC} (absorbing wall)

\beq \eta(0, t)= D_\rvt(t)
\eeq

\item
{\bf von Neumann BC} (reflecting wall)
\beq
\partial_x\eta(0,t)= N_\rvt(t)
\eeq
For example, if there were an impermeable wall at $x=0$,

\beq
\partial_x\eta(0,t)=0
\eeq

\item {\bf Mixed (Robin) BC}

\beq
[a_0 \eta + a_1\partial_x\eta  ](0, t)=R_\rvt(t)
\eeq
\end{itemize}
If the boundary condition depends on $\eta(x,t)$
only, then it is called a {\bf homogeneous BC} (i.e., $D_\rvt=N_\rvt=R_\rvt=0$)  .
Otherwise, it is said to be an {\bf inhomogeneous BC}.

\section{Notation},
I will often use specialized 
notation in this chapter to
express with more clarity 
the basic Green's functions equations and their
derivations. This section
will describe such notation.

An event
 will be presented 
either as a row vector or a column vector:
\beq
(x,t)\leftrightarrow \oneevent{x}{t}
\eeq
Functions $f()$ of one event, and functions $G()$
of two events
will be expressed with their events either as row or column vectors
\beq
f(x,t) \leftrightarrow f\oneevent{x}{t}
\eeq
\beq
G(x,t;x',t')\leftrightarrow G\twoevent{x}{t}{x'}{t'}
\eeq

Another notation that we will often use for 
expressing functions of one or two
events, is Dirac bra-ket notation.
We describe that notation next.

Define
\beq
\call_{x,t}= \partial_t -D \partial_x^2,\quad
\call_{x,t} G\twoevent{x}{t}{x'}{t'}=\stackt{\delta(x-x')}{\delta(t-t')}
\eeq


\beq
\call^\dagger_{x,t}= -{\partial}_t -D {\partial}_x^2,
\quad \call^\dagger_{x,t}G^\dagger\twoevent{x'}{t'}{x}{t}=\stackt{\delta(x-x')}{\delta(t-t')}
\eeq




\beq
P_\rvx = \int_0^\infty dx'\; \ket{\rvx=x'}\bra{\rvx=x'}
\eeq


\beq
P_\rvt = \int_0^t dt'\; \ket{\rvt=t'}\bra{\rvt=t'}
\eeq

\beq
P_0 = \ket{0}\bra{0}
\eeq

\beq
\partial_{\rvx} =
\int_0^\infty dx'\; \ket{\rvx=x'}\partial_{x'}\bra{\rvx=x'}
\eeq

\beq
\partial_{\rvt} =
\int_0^\infty dt\; \ket{\rvt=t'}\partial_{t'}\bra{\rvt=t'}
\eeq

\beq
\rv{\call}= \partial_\rvt-D\partial^2_\rvx
\eeq

\beq
f(x)\cev{\partial_x} a(x)= a(x) \partial_x f(x)
\eeq
We will also use the following suggestive notation
for dual tensor product spaces



\beq \stackt{A}{B}= A\otimes B\eeq

\beq \kett{x}{t}=\ket{x}\otimes\ket{t},
\quad 
\brat{x}{t}=\bra{x}\otimes\bra{t}
\eeq

\section{Master Equation}
\begin{claim}
Suppose
\beq
\call_{x,t}= \partial_t -D \partial_x^2,\quad
\call_{x,t} G\twoevent{x}{t}{x'}{t'}=\stackt{\delta(x-x')}{\delta(t-t')}
\eeq
The solution to the equation
\beq
\call_{x,t} \eta (x,t) = S(x,t)\quad
\eeq
for $x>0$ and $t>0$
with initial conditions
\beq
\eta(x,0)= \eta_\rvx(x)
\eeq
and Dirichlet, von Neumann or Robin boundary conditions satisfies the {\bf master equation}:

\beqa
\ket{\eta}
&=&
\underbrace{G\stackt{P_\rvx}{P_\rvt}\ket{S}}_{\text{Source Term $\cala_S$}}
\\
&&+\underbrace{G\stackt{P_\rvx}{P_0}\kett{\eta_\rvx}{0}}_{\text
{Initial Conditions (IC) term  $\cala_{BC}$}}
\\
&&+\underbrace{D
G\stackt{\left[-\ket{x'}\partial_{x'}\bra{x'}
+
\ket{x'}\cev{\partial}_{x'}\bra{x'}\right]_{x'=0}}{P_\rvt}
\ket{\eta}}_{\text{Boundary Conditions (BC) term $\cala_{BC}$}}
\eeqa
or, in non-Dirac notation,

\beqa
\eta\oneevent{x}{t}
&=&
\int_0^tdt'
\int_0^\infty dx'\;G\twoevent{x}{t}{x'}{t'}S\oneevent{x'}{t'}
\\
&&+\int_0^\infty dx'\;
G\twoevent{x}{t}{x'}{0}\eta_\rvx(x')
\\
&&+D
\int_0^tdt'\;G\twoevent{x}{t}{0}{t'} \left[-\partial_{x'}
+ \cev{\partial}_{x'}
\right]\eta\oneevent{0}{t'}
\eeqa
\end{claim}
\proof

Define $I$, $I_\rvx$ and $I_\rvt$  by
\beq
I = \brat{x}{t}G\rv{\call}
\ket{\eta}-
\brat{x}{t}\rv{\call^\dagger}G^\dagger
\ket{\eta}
\eeq

\beq
I_\rvt =
\brat{x}{t}G
\partial_\rvt
\ket{\eta}-
\brat{x}{t}
(-{\partial}_\rvt)G^\dagger
\ket{\eta}
\eeq

\beq
I_\rvx =
\brat{x}{t}G\partial_\rvx^2
\ket{\eta}-
\brat{x}{t}
\partial_\rvx^2G^\dagger
\ket{\eta}
\eeq
Hence
\beq
\boxed
{I = I_\rvt -D I_\rvx}
\eeq
\hrule
Note that we can simplify the
expression for $I$ as follows:



\beq
\rv{\call}\ket{\eta}=\ket{S},
\quad
\rv{\call^\dagger}G^\dagger=1
\eeq

\beq
\boxed
{I = \brat{x}{t}G\ket{S}-\brat{x}{t}\ket{\eta}
}
\eeq
\hrule
Note that we can simplify the
expression for $I_\rvt$ as follows:


\beqa
I_\rvt &=&
\int_0^t dt'
\brat{x}{t}G
\stackt{1}{\ket{t'}\partial_{t'}\bra{t'}}
\ket{\eta}
+
\brat{x}{t}G
\stackt{1}{\ket{t'}\cev{\partial}_{t'}\bra{t'}}
\ket{\eta}
\eeqa
where we used

\beq
 \bra{t'}G^\dagger\ket{t}=\bra{t}G\ket{t'}
\implies
\partial_{t'} \bra{t'}G^\dagger\ket{t}
=
\bra{t}G\ket{t'}\cev{\partial}_{t'}
\eeq

\beq
\partial_{t'} (\ket{t'}\bra{t'})=
\ket{t'}\partial_{t'}\bra{t'}
+\ket{t'}\cev{\partial}_{t'}\bra{t'}
\eeq


\beqa
I_\rvt &=&
\int_0^t dt'
\brat{x}{t}G
\stackt{1}{\partial_{t'} (\ket{t'}\bra{t'})}
\ket{\eta}
\eeqa



\beq
\int_0^t
dt'\;\partial_{t'} (\ket{t'}\bra{t'})
=
\ket{t}\bra{t}-\ket{0}\bra{0}
\eeq
\beq
G\twoevent{x}{t}{x'}{t}= 0 \quad \text{if $x\neq x'$} \text{(no instant propagation)}
\eeq
 so


\beq
\boxed{I_\rvt =
-\brat{x}{t}G
\stackt{P_\rvx}{\ket{0}\bra{0}}
\ket{\eta}}
\eeq
\hrule
Note that we can simplify the
expression for $I_\rvx$ as follows:

\beq
I_\rvx =
\int_0^\infty dx'
\brat{x}{t}G
\stackt{\ket{x'}\partial_{x'}^2\bra{x'}}{1}
\ket{\eta}
-
\brat{x}{t}
G\stackt{\ket{x'}\cev{\partial}_{x'}^2\bra{x'}}{1}
\ket{\eta}
\eeq

\beq
\ket{x'}\partial_{x'}^2\bra{x'}
-\ket{x'}\cev{\partial}_{x'}^2\bra{x'}=
\partial_{x'}\left(\ket{x'}\partial_{x'}\bra{x'}\right)
-\left(\ket{x'}\cev{\partial}_{x'}\bra{x'}\right)\cev{\partial}_{x'}
\eeq

\beq
I_\rvx=
\int_0^\infty dx'
\brat{x}{t}G
\stackt{\partial_{x'}(\ket{x'}\partial_{x'}\bra{x'})
-\left(\ket{x'}\cev{\partial}_{x'}\bra{x'}\right)\cev{\partial}_{x'}}{1}
\ket{\eta}
\eeq

\begin{align}
\int_0^\infty dx'\left[
\partial_{x'}(\ket{x'}\partial_{x'}\bra{x'})
-\left(\ket{x'}\cev{\partial}_{x'}\bra{x'}\right)\cev{\partial}_{x'}\right]
&=
\left[\ket{x'}\partial_{x'}\bra{x'}
-
\ket{x'}\cev{\partial}_{x'}\bra{x'}\right]_0^\infty
\\
&=
-\left[\ket{x'}\partial_{x'}\bra{x'}
-
\ket{x'}\cev{\partial}_{x'}\bra{x'}\right]_{x'=0}
\end{align}

\beq
\boxed{I_\rvx=
\brat{x}{t}G
\stackt{\left[-\ket{x'}\partial_{x'}\bra{x'}
+
\ket{x'}\cev{\partial}_{x'}\bra{x'}\right]_{x'=0}}{P_\rvt}
\ket{\eta}}
\eeq


\qed

The BC term 
of the Master Equation 
must be specialized for
each of the 3 possible
types of BC. We show how to do  that below.

Careful: Note that
\beq
\partial_t G\twoevent{x}{t}{x'}{t'}=-\partial_{t'}G\twoevent{x}{t}{x'}{t'}
\eeq
and
\beq
\partial_x G\twoevent{x}{t}{x'}{t'}=-\partial_{x'}G\twoevent{x}{t}{x'}{t'}
\eeq

\begin{claim}(space-like propagation for $t=t'=0$)

\beq
G\twoevent{x}{0}{x'}{0}= \delta(x-x')
\eeq
\end{claim}
\proof
\beq
\int_{0}^{t}dt'\;
G\twoevent{x}{t}{x'}{t'}[\underbrace{\cev{\partial}_{t}}_{-\cev{\partial}_{t'}} -D\cev{\partial}_{x'}^2]
=\int_{0}^{t}dt'\;
\delta(x-x')\delta(t-t')
\eeq
so integrating the time derivative term

\beq
-G\twoevent{x}{t}{x'}{t}
+G\twoevent{x}{t}{x'}{0}
-D\int_0^t dt'\;
G\twoevent{x}{t}{x'}{t'}\cev{\partial}^2_{x'}=\delta(x-x')
\eeq
For for an infinitesimal $t=\eps$,
only one term survives 
on the left hand side. Hence
\beq
G\twoevent{x}{\eps}{x'}{0}=\delta(x-x')
\eeq

\qed

Consider the
the master equation and its IC term in particular:
\beq
\eta(x,t)= \cala_S(x,t) + \cala_{IC}(x,t) + \cala_{BC}(x,t)
\eeq

\beq
\cala_{IC}(x,t)=\int_0^\infty dx'\;
G\twoevent{x}{t}{x'}{0}\eta_\rvx(x')
\eeq
When $t=0$,

\beqa
\cala_{IC}(x,0)=\int_0^\infty dx'\;
\underbrace{G\twoevent{x}{0}{x'}{0}}_{\delta(x-x')}
\eta_\rvx(x')
\\
&=& \eta_\rvx(x)
\eeqa


\begin{claim} (time-like propagation for $x=x'=0$)
\label{cl-time-like-prop}

\beq
D \partial_{x'}G\twoevent{0}{t}{0}{t'} 
=
-D\partial_x G\twoevent{0}{t}{0}{t'}=
 \delta(t-t')
\eeq 
\end{claim}
\proof



Note that

\beq
\int_{0}^{\infty}dx'\;
G\twoevent{x}{t}{x'}{t'}[\underbrace{\cev{\partial}_{t}}_{-\cev{\partial}_{t'}} -D\cev{\partial}_{x'}^2]
=\int_{0}^{\infty}dx'\;
\delta(x-x')\delta(t-t')
\eeq
so integrating the second derivative

\beq
-\int_{0}^{\infty}dx'\;G\twoevent{0}{t}{x'}{t'}\cev{\partial}_{t'}+
DG\twoevent{0}{t}{0}{t'}\cev{\partial}_{x'}=\delta(t-t')
\eeq
$G$ peaks when $x=x'$ so when $x=0$,
we need only integrate over $[0,\eps]$
for infinitesimal $\eps$.

\beq
-\int_{0}^{\eps}dx'\;G\twoevent{0}{t}{x'}{t'}\cev{\partial}_{t'}+
DG\twoevent{0}{t}{0}{t'}\cev{\partial}_{x'}=\delta(t-t')
\eeq
A consequence of this is that
As long as $t\neq t'$, $G\twoevent{0}{t}{x'}{t'}$
is a finite function of $x'$ so
the integral over $x'$ is proportional  to
$\eps$ and may be assumed to vanish. 
\qed

Consider the
the master equation and its BC term in particular:
\beq
\eta(x,t)= \cala_S(x,t) + \cala_{IC}(x,t) + \cala_{BC}(x,t)
\eeq
\beq
\cala_{BC}=
D
\int_0^tdt'\;G\twoevent{x}{t}{0}{t'} \left[-\partial_{x'}
+ \cev{\partial}_{x'}
\right]\eta\oneevent{0}{t'}
\eeq
The form of $\cala_{BC}$ and Claim
\ref{cl-time-like-prop} imply that
the Green's Function
must satisfy the homogeneous
version of the inhomogeneous BC
satisfied by $\eta(x,t)$.

Indeed, if Dirichlet BC
are satisfied

\beq
\eta(0,t)=D_\rvt(t),\quad G\twoevent{0}{t}{x'}{t'}=0
\eeq
then $\cala_{BC}(0, t)$ reduces to
\beqa
\cala_{BC}(0,t)&=&
\int_0^tdt'\;\underbrace{DG\twoevent{0} {t}{0}{t'}[\cev{\partial}_{x'}]}_{\delta(t-t')}
\eta\oneevent{0}{t'}
\\
&=&
D_\rvt(t)
\eeqa

Likewise, if von Neumann BC
are satisfied

\beq
\partial_x\eta(0,t)=N_\rvt(t),\quad \partial_x G\twoevent{0}{t}{x'}{t'}=0
\eeq
then $\partial_x\cala_{BC}(0, t)$ reduces to
\beqa
\partial_x\cala_{BC}(0, t)&=&
\int_0^tdt'\;\underbrace{D\partial_x G\twoevent{0}{t}{0}{t'}}_{-\delta(t-t')} \left[-\partial_{x'}
\right]\eta\oneevent{0}{t'}
\\
&=&
N_\rvt(t)
\eeqa





\section{Point sources and their images}

All Green's functions must satisfy
the DE with delta function sources

\beq
\call_{x,t} G\twoevent{x}{t}{x'}{t'}=\delta(x-x')\delta(t-t')
\quad \text{for all $x>0$ and $t>0$}
\label{eq-LG-deltas}
\eeq
In addition, they must satisfy BC
that match the BC satisfied by the
solution $\eta(x,t)$.
Below, we specify what BC the Green's function must satisfy.

Let $P(x;\s^2)$
represent the Normal Distribution
with mean zero and standard deviation $\s$:
\beq
P(x;\s^2)=\frac{1}{\sqrt{2\pi\s^2}}
\exp\left(\frac{-x^2}{2\s^2}\right)
\eeq
Let 
\beqa
G^\pm\twoevent{x}{t}{x'}{t'}
&=&
\frac{1}{\sqrt{4\pi D(t-t')}}
\exp\left[-\;\frac{(x\mp x')^2}{4D(t-t')}\right]
\\
&=& P(x\mp x'; \kappa(t-t'))
\label{eq-def-green-pm}
\eeqa
where $\kappa =2D$. $G^\pm$ is centered at $x=\pm x'$
The $G^\pm$.  Green's functions
will be used as a building blocks
to construct the Green's functions
appropriate to the various BC.

\caption{Contour plots of the function $G^+\twoevent{x}{t}{x'}{t'} = P(x-x';\kappa(t-t'))$}

countour plots for $G^+$. Note that 
$G^+$ is zero along the line $t-t'=0$
except at the point $(x-x', t-t')=(0,0)$ 
where it \qt{spikes} and becomes infinite. Note also that we define
$G=0$
for all $x-x'<0$. The spike at the origin is the reason why 
$\call_{x,t}G^+$ gives a delta function.

Below, we will claim
that the Green's function $G^D$ for the Dirichlet
BC and the Green's function $G^N$
for the von Neumann BC, are  equal to

\beq
G^D =G^+-G^-, \quad  G^N=G^++ G^-
\eeq
Note that the $G^-$ spike
occurs at $x=-x'<0$ so
\beq
\call_{x,t}G^+= \delta(x-x')\delta(t-t'),\quad \call_{x,t}G^-= 0, \quad \text{for all $x>0$ and $t>0$}
\eeq
Hence, both $G^D$ and $G^N$ satisfy
Eq.(\ref{eq-LG-deltas}) 
in the region of interest (i.e., $x>0$ and $t>0$).

Fig.\ref{fig-de-integration-region}
shows the integration region
for the Master Equation and the spikes of $G^\pm$.



\section{Dirichlet BC}

\beq
\eta(0, t)= D_\rvt(t)
\eeq

Choose Green's function $G^D$ so that
it satisfies the DE with delta function source and obeys
the same BC, but homogeneous
\beq
G^D\twoevent{0}{t}{x'}{t'}=0
\eeq
The solution is

\beq
G^D= G^+ - G^-
\eeq




At the boundary
\beq
\left.
\partial_{x'} G^D
(x,t;x',t')
\right|_{x'=0}
=
\frac{x}{\sqrt{2\kappa (t-t')^3}}
\exp\left(
\frac{-x^2}{2\kappa(t-t')}
\right)
\eeq
This $G^D$ was found by the method of images: two
free space solutions $G^\pm$ to $\ul{\call}G^\pm=1$
were superposed so as  to satisfy
$G^D=0$ at the boundary $x=0$.

\begin{claim}
Prove that
\beq
\lim_{x\rarrow 0}D
\int_0^{t} dt'\;
[\partial_{x'} G^D(x,t;x',t')]_{x'=0}
=\delta(t-t')
\eeq
using the explicit form of $G^D= G^+-G^-$.
\end{claim}
\proof

Suppose

\beq
G^+(x,t;x',t')= P(x-x'; \kappa (t-t'))
\eeq
where $D=\kappa/2$.
Define

\beq
y=\frac{x}{\sqrt{\kappa(t-t')}}=\frac{x}{\ell}
\eeq
Then
\beqa
D[\partial_{x'} G^+(x,t;x',t')]_{x'=0}&=&
\left(\frac{\kappa}{2}\right)
\left(\frac{(-1)^2 x}{\ell^2}\right)\frac{1}{\ell\sqrt{2\pi}}\exp\left(
\frac{-x^2}{2\ell^2}\right)
\\
&=&
\frac{x}{2(t-t')}\frac{1}{\ell\sqrt{2\pi}}\exp\left(
\frac{-y^2}{2}\right)
\eeqa

\beq
\ln y = -\frac{1}{2}\ln (t-t')+\cdots\implies \frac{dy}{y}=
-\;\frac{d(t-t')}{2(t-t')}=
\;\frac{dt'}{2(t-t')}
\eeq
Thus,

\beq
dt'\;\frac{x}{2(t-t')}= dy\;\frac{x}{y}=
dy\sqrt{\kappa(t-t')}=dy\ell
\eeq
Hence
\beq
D\int_0^t dt'[\partial_{x'} G^+(x,t;x',t')]_{x'=0}=
\int_{\frac{x}{\sqrt{\kappa t}}}^\infty
dy \frac{1}{\sqrt{2\pi}}\exp\left(
\frac{-y^2}{2}\right)=\frac{1}{2}
\eeq
\section{von Neumann BC}
\beq
\partial_x\eta(0, t)= N_\rvt(t)
\eeq

Choose Green's function $G^N$ so that
it satisfies the DE with delta function source and obeys
the same BC, but homogeneous
\beq
G^N\twoevent{x}{t}{0}{t'}\cev{\partial}_{x'}=0
\eeq
The solution is

\beq
G^N= G^+ + G^-
\eeq
Note that for the Dirichlet BC problem, $G^+$ and $G^+$
are subtracted instead if added.


\section{Mixed (Robin) BC}
\beq
a_0\eta(0, t) + a_1\partial_x \eta(0, t)= R_\rvt(t)
\eeq

Choose Green's function $G^R$ so that
it satisfies the DE with delta function source and obeys
the same BC, but homogeneous
\beq
[a_0 +  a_1 \partial_x]G^R\twoevent{0}{t}{x'}{t'}=0
\eeq
The solution is

\beq
G^R= G^+ + G^-
-2k\int_0^\infty 
dX\;e^{-k X} G^+\twoevent{x}{t}{x'+X}{t'}
\eeq
where $k= \frac{a_0}{a_1}$.

Dirichlet limit, $k\rarrow \infty$:

\beq
k e^{-kX}\rarrow \delta(X)
\eeq

\beq
G^R\rarrow G^+ + G^- -2G^+= G^+-G^- = G^D
\eeq

von Neumann limit, $k=0$:

\beq
G^R=G^++G^-= G^N
\eeq
