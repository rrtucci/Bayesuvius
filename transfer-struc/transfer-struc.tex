\chapter{Transfer of Structural Knowledge}
\label{ch-transfer-struc}

This chapter is based on Ref.\cite{nic-transfer}.

$G. = (G_1, G_2, \ldots, G_n)$

$\theta. = (\theta_1, \theta_2, \ldots, \theta_n)$

$D. = (D_1, D_2, \ldots, D_n)$

\beq
P(G.|D.)=\caln(!G.)P(D.|G.)P(G.)
\eeq

\beq
P(D.|G.)=\prod_i P(D_i|G_i)
\eeq

\beq
P(D_i|G_i)=
\sum_{\theta_i}
P(D_i|G_i, \theta_i)
P(\theta_i|G_i)
\eeq

\beq
P(G_1, G_2)=\caln(!G.)
[P(G_1)P(G_2)]^{\frac{1}{1+\delta}}
(1-\delta)^{|G_1\Delta G_2|}
\eeq


If $\delta=0$, 

\beq
P(G_1, G_2)=P(G_1)P(G_2)
\eeq

If $\delta=1$,

\beqa
P(G_1, G_2)&=&
\sqrt{P(G)P(G)}\;\delta(G_1, G_2)
\\
&=&P(G)\;\delta(G_1, G_2)
\eeqa
