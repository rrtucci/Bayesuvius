\beq
P(G.|D.)=\caln(!G.)P(D.|G.)P(G.)
\eeq

\beq
P(D.|G.)=\prod_i P(D_i|G_i)
\eeq

\beq
P(D_i|G_i)=
\sum_{\theta_i}
P(D_i|G_i, \theta_i)
P(\theta_i|G_i)
\eeq

\beq
P(G_1, G_2)=\caln(!G.)
[P(G_1)P(G_2)]^{\frac{1}{1+a}}
(1-a)^{|G_1\Delta G_2|}
\eeq


If $a=0$, 

\beq
P(G_1, G_2)=P(G_1)P(G_2)
\eeq

If $a=1$,

\beqa
P(G_1, G_2)&=&
\sqrt{P(G)P(G)}\;\delta(G_1, G_2)
\\
&=&P(G)\;\delta(G_1, G_2)
\eeqa
