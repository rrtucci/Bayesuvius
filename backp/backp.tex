\section{Toy Example}
This example
comes from Ref.\cite{backp-toy}.

Consider the following system of 4 equations:
\beq
\left\{
\begin{array}{l}
b= W_1a
\\
c=W_2a
\\
d=W_3b+W_4c
\\
\cale=10-d
\end{array}
\right.
\label{eq-backp-toy-eqs}
\eeq

To calculate $\\nabla_W \cale$ where
 $W=(W_1, W_2, W_3, W_4)$, we can use the
chain rule for partial derivatives.
This yields


\begin{subequations}
\beq
\pder{\cale}{W_4}=\pder{\cale}{d}\pder{d}{W_4}=(-1)(c)
\eeq

\beq
\pder{\cale}{W_3}=\pder{\cale}{d}\pder{d}{W_3}=(-1)(b)
\eeq

\beq
\pder{\cale}{W_2}=\pder{\cale}{d}\pder{d}{c}\pder{c}{W_2}
=(-1)(W_4)(a)
\eeq

\beq
\pder{\cale}{W_1}=\pder{\cale}{d}\pder{d}{b}
\pder{b}{W_1}
=
(-1)(W_3)(a)
\eeq
\end{subequations}

Now note that the system of equations
Eq.(\ref{eq-backp-toy-eqs}) can
be represented graphically by the bnet (with
deterministic nodes) of
Fig.\ref{fig-toy-backp-bnet}.
The calculation
of $\\nabla_W\cale$ can also be represented
graphically with the aid of the defining bnet
Fig.\ref{fig-toy-backp-bnet}. This is done in Fig.\ref{fig-toy-backp-deriv-bnet}.
As illustrated by that figure,
the derivative $\partial_{W_i}\cale$ for $i=1,2,3,4$
is the product of the derivatives along the
arrows from $W_i$ to $\cale$.

\section{General Theory}

\subsection{Jacobians}

Suppose
 $f:\RR^{nx}\rarrow \RR^{nf}$
and

\beq
y=f(x)
\;.
\eeq
Then the Jacobian $\pder{y}{x}$
is defined as the matrix with entries\footnote{
Mnemonic for remembering
order of indices: 
Visualize the Jacobian $\pder{f}{x}$ as
the matrix $\partial f_i/\partial x_j=M_{i,j}$
with row index $i$ and column index $j$.}

\beq
\left[\pder{y}{x}\right]_{i,j}=
\pder{y_i}{x_j}
\;.
\eeq




Jacobian of function composition.
Suppose $f:\RR^{nx}\rarrow\RR^{nf}$,
$g:\RR^{nf}\rarrow \RR^{ng}$. If

\beq
y=g\circ f (x)
\;,
\eeq
then

\beq
\pder{y}{x}dx
=
\pder{g}{f}
\pder{f}{x}dx
\;.
\eeq
Right-hand side
of the last equation
is a product
of two matrices,
so order of the product is important. $dx$ is
a column vector.

Let
\beq
y=f^4\circ f^3 \circ f^2 \circ f^1 (x)
\;.
\eeq
This function composition chain can be
represented by the bnet
Fig.\ref{fig-backp-abc}$(a)$
with TPMs

\beq\color{blue}
P(f^\mu|f^{\mu-1})=
\indi(f^\mu=f^\mu(f^{\mu-1}))
\;
\eeq
for $\mu=1, 2,3,4$.


Note that
\beqa
\pder{y}{f^3}
\pder{f^3}{f^2}
\pder{f^2}{f^1}
\pder{f^1}{x}dx
&=&
\pder{y}{f^3}
\pder{f^3}{f^2}
\pder{f^2}{f^1}
df^1
\\
&=&
\pder{y}{f^3}
\pder{f^3}{f^2}
df^2
\\
&=&
\pder{y}{f^3}df^3
\\
&=&
dy
\;.
\eeqa
This forward propagation can be
represented by the bnet
Fig.\ref{fig-backp-abc}$(b)$
with node TPMs

\beq\color{blue}
P(df^{\mu+1}\mid
df^{\mu})=
\indi(df^{\mu+1}=
\pder{f^{\mu+1}}{f^{\mu}}df^\mu
)
\eeq
for $\mu=1,2,3$.

Note that

\beqa
\pder{x}{f^1}
\pder{f^1}{f^2}
\pder{f^2}{f^3}
\pder{f^3}{y}dy
&=&
\pder{x}{f^1}
\pder{f^1}{f^2}
\pder{f^2}{f^3}df^3
\\
&=&
\pder{x}{f^1}
\pder{f^1}{f^2}df^2
\\
&=&
\pder{x}{f^1}df^1
\\
&=&
dx
\eeqa
This back propagation can be
represented by the bnet
Fig.\ref{fig-backp-abc}$(c)$
with node TPMs

\beq\color{blue}
P(
df^{\mu}
\mid
df^{\mu+1}
)=
\indi(
df^{\mu}
=
df^{\mu+1}
\pder{f^{\mu}}{f^{\mu+1}})
\;
\eeq
for $\mu=2,1,0$.

$\pder{f^{\mu+1}}{f^{\mu}}$
is a Jacobian matrix
so the order of multiplication
matters. In forward prop,
it pre-multiplies,
and in back prop it post-multiplies.

The TPMs, printed in blue,
for a NN bnet,
as given in Chapter \ref{ch-nn},
are as follows.

For all hidden layers $\lam=0, 1, \dots,
\Lambda-2$,

\beq\color{blue}
P(h^{\lam}_i\mid h^{\lam-1})=
\delta\left(h^{\lam}_i,
\phi_i^\lam(\sum_j
 W^{\lam}_{i|j}
h^{\lam-1}_j + b^{\lam}_i)\right)
\eeq
for $i=0, 1, \ldots, nh(\lam)-1$.
For the output visible layer $\lam=\Lambda-1$:

\beq\color{blue}
P(Y_i\mid h^{\Lambda-2}_.)=
\delta \left(Y_i,
\phi_i^{\Lambda-1}(\sum_j W^{\Lambda-1}_{i|j}
h^{\Lambda-2}_j + b^{\Lambda-1}_i)\right)
\;
\eeq
for $i=0, 1, \ldots, ny-1$.


For each $\lam$, replace the matrix
 $W^\lam_{\cdot|\cdot}$
by
the augmented matrix
$[b^\lam., W^\lam_{\cdot|\cdot}]$
so that the new
 $W^\lam_{\cdot|\cdot}$ satisfies

\beq
W^{\lam}_{i|0}=b_i^{\lam}
\;
\eeq

Let the nodes $\rvh_0^\lam$
for all $\lam$ and $\rvY_0$ be
root nodes (so no arrows
pointing into them).
For each $\lam$, draw arrows from
$\rvh_0^\lam$ to all other nodes
in that same layer.
Draw arrows from
$\rvY_0$ to all other nodes
in that same layer.

After performing the
above steps,
the TPMs,
printed in blue,
for the NN bnet,
 are as follows:

For all hidden layers $\lam=0, 1, \dots,
\Lambda-2$,

\beq\color{blue}
P(h^\lam_0)=\delta(h^\lam_0,1)
\;,
\eeq
and

\beq\color{blue}
P(h^{\lam}_i\mid h^{\lam-1}_., h_0^\lam=1)=
\delta\left(h^{\lam}_i,
\phi_i^\lam(\sum_{j}
 W^{\lam}_{i|j}
h^{\lam-1}_j)\right)
\eeq
for $i=1, \ldots, nh(\lam)-1$.
For the output visible layer $\lam=\Lambda-1$:

\beq\color{blue}
P(Y_0)=\delta(Y_0,1)
\;,
\eeq
and

\beq\color{blue}
P(Y_i\mid h^{\Lambda-2}_., Y_0=1)=
\delta \left(Y_i,
\phi_i^{\Lambda-1}(\sum_j W^{\Lambda-1}_{i|j}
h^{\Lambda-2}_j)\right)
\;
\eeq
for $i=1, 2, \ldots, ny-1$.

From here on, we will rename $y$ above
by $Y=\HAT{y}$ and
consider samples $y[i]$ for
$i=0, 1, \ldots, nsam-1$.
The Error (a.k.a. loss or cost function) is

\beq
\cale =\frac{1}{nsam}
\sum_{\sigma=0}^{nsam-1}\sum_i
|Y_i-y_i[\sigma]|^2
\eeq
To perform simple gradient descent,
one uses:

\beq
(W_{i|j}^\lam)'=W_{i|j}^\lam
-\eta\pder{\cale}{W^\lam_{i|j}}
\;.
\eeq
One has

\beq
\pder{\cale}{W^\lam_{i|j}}=
\frac{1}{nsam}
\sum_{\sigma=0}^{nsam-1}
2(Y_i-y_i[\sigma])\pder{Y_i}{W^\lam_{i|j}}
\;.
\eeq
Define $z ^\lam_{i}$ thus


\beq
[z ^\lam(h^{\lam-1})]_{i}=\sum_{j'}
 W^{\lam}_{i|j'}h^{\lam-1}_{j'}
\;.
\label{eq-calb}
\eeq
Then

\beqa
\pder{Y_k}{W^\lam_{i|j}}&=&
\pder{Y_k}{z ^\lam_{i}}
\pder{z _{i}^\lam}{W^\lam_{i|j}}
\\
&=&\pder{Y_k}{z ^\lam_{i}}
h^{\lam-1}_j
\eeqa

\beqa
\pder{\cale}{W^\lam_{i|j}}&=&
\pder{\cale}{z ^\lam_{i}}
\pder{z ^\lam_{i}}{W^\lam_{i|j}}
\\
&=&
\underbrace{\pder{\cale}{z ^\lam_{i}}}_{\delta^\lam_i}
h^{\lam-1}_j
\;.
\eeqa
This suggests that
we can calculate
the derivatives of the error
$\cale$
with respect to the weights
$W^\lam_{i|j}$ in two
stages, using an intermediate
quantity $\delta^\lam_{i}$:

\beq
\left\{
\begin{array}{l}
\delta^\lam_{i}=\pder{\cale}{z ^\lam_{i}}
\\
\pder{\cale}{W^\lam_{i|j}}=\delta^\lam_{i} h^{\lam-1}_j
\end{array}
\right.
\eeq

To
apply what we learned in the earlier
General Theory
section of this chapter,
consider a NN with 4
layers (3 hidden, and the
output visible one).
See Fig.\ref{fig-backp-nn}. The
TPMs, printed in blue,
for the bnet $(c)$
for back propagation, are
as follows:

\beq\color{blue}
P(
dY \mid dz ^{\lam}
)=
\indi\left(
dY
=
\pder{Y}{z ^{\lam+1}}
\pder{z ^{\lam+1}}{\phi^{\lam}}
\pder{\phi^{\lam}}{z ^{\lam}}
dz ^{\lam}
\right)
\;.
\label{eq-backp-nn-1}
\eeq
From Eq.(\ref{eq-calb})

\beq
z ^{\lam+1}_{i}(\phi^{\lam})=
\sum_{r} W^{\lam+1}_{i|r}\phi^{\lam}_{r}
\eeq
so

\beq
\pder{z _{i}^{\lam+1}}{\phi_{j}^{\lam}}
=W^{\lam+1}_{i|j}
\;.
\eeq
One also has

\beq
\pder{\phi^{\lam}_i}{z ^{\lam}_j}
=
D\phi_i^\lam(z _i^\lam)\delta(i,j)
\eeq
where $D{\phi}^\lam_i(z)$
is the derivative of
$\phi^\lam_i(z)$.
Therefore, Eq.(\ref{eq-backp-nn-1})
implies

\beq\color{blue}
P(
dY\mid dz ^{\lam}
)=
\indi\left(
dY
=
\sum_{i,j}
\pder{Y}{z _i^{\lam+1}}
W^{\lam+1}_{i|j}
D{\phi}^\lam_j(z ^\lam_j)dz ^\lam_j
\right)
\;,
\eeq

\beq\color{blue}
P(
d\cale\mid dz ^{\lam}
)=
\indi\left(
d\cale
=
\sum_i
\pder{\cale}{z _i^{\lam+1}}
W^{\lam+1}_{i|j}
D{\phi}^\lam_j(z ^\lam_j)dz _j^\lam
\right)
\;,
\eeq

\beq\color{blue}
\boxed{
P(
\delta^{\lam}_j
\mid
\delta^{\lam+1}_j
)=
\indi(
\delta^{\lam}_j
=
\sum_i
\delta_i^{\lam+1}
D{\phi}^\lam_j(z ^\lam_j))
W^{\lam+1}_{i|j}
)
}
\;.
\eeq

First delta of iteration, belonging to output
layer $\lam=\Lambda-1$:

\beqa
\delta^{\Lambda-1}_j&=&
\pder{\cale}{z ^{\Lambda-1}_j}
\\
&=&
\frac{1}{nsam}
\sum_{\sigma=0}^{nsam-1}\sum_{i=0}^{ny-1}
2(Y_i-y_i[\sigma])
D{\phi}^{\Lambda-1}_i(z ^{\Lambda-1}_i)\delta(i,j)
\\
&=&
\frac{1}{nsam}
\sum_{\sigma=0}^{nsam-1}
2(Y_j-y_j[\sigma])
D{\phi}^{\Lambda-1}_j(z ^{\Lambda-1}_j)
\eeqa

Cute expression for
derivative of sigmoid function:
\beq
D\smoid(x)=
\smoid (x)(1-\smoid(x))
\eeq
