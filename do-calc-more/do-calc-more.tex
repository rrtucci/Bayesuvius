\begin{claim} (from Ref.\cite{hunermund2021})

If 
$
\xymatrix{
&\rvw\ar[d]\ar[ddr]\ar@/^1.5pc/@{<-->}[ddr]|\rvu
\\
&\rvh\ar[dr]
\\
\rvc\ar[rr]\ar[uur]\ar@/_1.5pc/@{<-->}[dr]
&&\rvy
\\
&\rve\ar[ru]\ar[lu]
}$
then
\beqa
P(y|\cald \rvc = c)&=&
\sum_e P(y|c,e)P(e)
\\
&=&
\xymatrix{
c\ar[rr]
&&y
\\
&\sum e\ar[ru]
}
\eeqa
\end{claim}
\proof

\begin{longtable}{l}
\color{red}
$P(y|\cald\rvc=c)=$
\\
$
\xymatrix{
&\sum w\ar[d]\ar[ddr]\ar@/^1.5pc/@{<-->}[ddr]|{\sum u}
\\
&\sum h\ar[dr]
\\
\cald\rvc=c\ar[rr]\ar[uur]
&&y
\\
&\sum e\ar[ru]
}
\xymatrix{
\\
\\
=}
\xymatrix{
\\
\cald\rvc=c\ar[rr]
&&y
\\
&\sum e\ar[ru]
}$
\\
\color{red}
$=\sum_e
P(y|\cald\rvc=c, e)
P(e|\cald\rvc=c)$
\\
\xymatrix{
\\=}
$
\xymatrix{

\ar[dr]
&&y
\\
&\sum e\ar[ru]
}$
\\
\color{red}
$=\sum_e

\xymatrix{
